use std::fmt::{self, Display};

struct Pair<T>{
    x: T,
    y: T,
}
impl<T> Pair<T> {
    fn new(x:T, y:T) -> Self {
        Self {x,y} 
    } 
}
impl<T: Display + PartialOrd> Pair<T> {
    fn cmp_display(&self){
        if self.x >= self.y {
            println!("The largest member is x = {}",self.x);
        }
        else {
            println!("The largest member is y = {}",self.y);
        }
    }
}
impl<T: Display> ToString for Pair<T> {
    fn to_string(&self) -> String{
        format!("[{},{}]",self.x, self.y)
    }
}
fn main() {
   let point = Pair::new(6,4); 
   point.cmp_display();
   println!("{}",point.to_string());
}

여기서 Pair struct는 generic을 가진다. 그래서 메서드를 추가하거나 trait을 implement 하기 위해서는 impl<T> ... Pair<T> 처럼 generic을 표시해주어야 한다.

  1. 첫번째 impl 구문은 Pair struct에 메서드를 추가한다.

  2. 두번째 impl 구문은 첫번째와 마찬가지로 Pair struct에 메서드를 추가한다. 하지만 이 메서드는 generic 변수 T가 Display와 PartialOrd trait을 가졌을 경우에만 조건적으로 struct에 추가된다.

  3. 세번째 impl 구문은 Pair struct에 trait을 implement 한다. generic 변수 T가 Display trait을 가졌을 때만 ToString trait을 추가한다.
