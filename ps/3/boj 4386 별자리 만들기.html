코드

#include <cmath>
#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
struct pos {
    double x, y;
};
pos star[101];

struct edge{
    int u, v;
    double w;
};
vector<vector<edge>> graph(101, vector<edge>());

struct box{
    int v;
    double dist;
    bool operator()(box& a, box& b){
        return a.dist > b.dist;
    }
};

void solve(){
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> star[i].x >> star[i].y;
    }
    for(int i = 0; i<n; i++){
        for(int j = i+1; j<n; j++){
            double weight = sqrt(pow(star[i].x - star[j].x, 2) + pow(star[i].y - star[j].y, 2));
            graph[i].push_back({i, j, weight}); 
            graph[j].push_back({j, i, weight}); 
        }
    }
    priority_queue<box, vector<box>, box> pq;
    pq.push({0, 0});  
    int done[101] = {0,};
    double ans = 0;
    while(!pq.empty()){
        box cur = pq.top();
        pq.pop();
        if(done[cur.v]==0){
            ans += cur.dist;
            done[cur.v] = 1;
        } else {
            continue;
        }
        for(edge next: graph[cur.v]){
            if(done[next.v]==0){
                pq.push({next.v, next.w});
            }
        }
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

별의 최대 개수가 100개 이므로, 모든 별의 좌표를 입력받고 모든 별에 대해서 다른 모든 별에 대한 거리를 graph로 만들어 줄 수 있다.

이후 이 graph를 가지고, mst를 만들어주며 비용을 구해주면 된다.
