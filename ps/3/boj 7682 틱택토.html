코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
string board;
int nX, nO, nB;
int complete_X, complete_O;
void count(){
    for(int i = 0; i<3; i++){
        for(int j = 0; j<3; j++){
            if(board[i*3+j]=='X')
                nX += 1;
            else if(board[i*3+j]=='O')
                nO += 1;
            else
                nB += 1;
        }
    }
}
void search(){
    for(int i = 0; i<3; i++){
        if(board[i*3+0]=='X' && board[i*3+1]=='X' && board[i*3+2]=='X'){
            complete_X += 1;
        } else if(board[i*3+0]=='O' && board[i*3+1]=='O' && board[i*3+2]=='O'){
            complete_O += 1;
        }

        if(board[0*3+i]=='X' && board[1*3+i]=='X' && board[2*3+i]=='X'){
            complete_X += 1;
        } else if(board[0*3+i]=='O' && board[1*3+i]=='O' && board[2*3+i]=='O'){
            complete_O += 1;
        }
    }
    if(board[0]=='X' && board[4]=='X' && board[8]=='X'){
        complete_X += 1;
    } else if(board[0]=='O' && board[4]=='O' && board[8]=='O'){
        complete_O += 1;
    }
    if(board[2]=='X' && board[4]=='X' && board[6]=='X'){
        complete_X += 1;
    } else if(board[2]=='O' && board[4]=='O' && board[6]=='O'){
        complete_O += 1;
    }
}

void solve(string ttt){
    complete_X = complete_O = nX = nO = nB = 0;
    board = ttt; 
    count();
    search();
    bool is_valid = false;
    if(nB > 0){
        // blank exists
        if(nX==nO && complete_O==1 && complete_X==0){
            is_valid = true;
        } else if(nX==nO+1 && complete_X==1 && complete_O==0){
            is_valid = true;
        }
    } else {
        // blank not exists
        if(nX==nO+1 && (complete_X==1 || complete_X==2) && complete_O==0){
            is_valid = true; 
        } else if(nX==nO+1 && complete_X==0 && complete_O==0){
            is_valid = true;
        }
    }
    if(is_valid){
        cout << "valid" << '\n';
    } else {
        cout << "invalid" << '\n';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    string str;
    while(1){
        cin >> str;
        if(str == "end")
            break;
        solve(str); 
    }
    return 0;
}

풀이

가능한 경우의 수를 나누어서 가능 여부를 확인해 주면 된다. 나는 다음과 같이 경우의 수를 나누었다. n(X), n(O)를 각각 X의 개수, O의 개수라고 한다.

  • 빈칸이 있는 경우
    • n(X) == n(O)이고, O로 된 완성된 한 줄만 존재하면 valid (O가 마지막으로 놓은 경우)
    • n(X) == n(O) + 1이고, X로 된 완성된 한 줄만 존재하면 valid (X가 마지막으로 놓은 경우)
  • 빈칸이 없는 경우 (9칸이므로 X에서 끝이 난다)
    • n(X) == n(O) + 1이고, X로 된 완성된 한 줄 또는 두 줄이 존재하면 valid
    • n(X) == n(O) + 1이고, 완성된 줄이 없는 경우 valid

이외의 경우의 수는 모두 invalid이다.

느낀점

경우의 수를 나누어서 풀어야 한다.
