#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m;
struct edge {
    int u, v, w;
    void print(){
        cout << u << ' ' << v << ' ' << w << '\n';
    }
};
vector<vector<edge>> graph(1001, vector<edge>());
struct compare {
    bool operator()(edge& a, edge& b){
        return a.w > b.w;
    }
};

void solve(){
    cin >> n >> m;
    int u, v, w;
    for(int i = 0; i<m; i++){
        cin >> u >> v >> w;
        graph[u].push_back({u, v, w});
        graph[v].push_back({v, u, w});
    }
    vector<edge> ans;
    int done[1001] = {0,};
    int res[1001] = {0,};
    fill(res, res + 1001, INT_MAX);
    priority_queue<edge, vector<edge>, compare> pq;
    pq.push({-1, 1, 0});
    while(!pq.empty()){
        edge cur = pq.top();
        pq.pop();
        if(done[cur.v]!=0)
            continue;
        done[cur.v] = 1;
        if(cur.u != -1 )
            ans.push_back(cur); //dijkstra에서 cur의 weight은 현재까지의 경로의 길이이기 때문에 의미가 없다.
        for(edge n: graph[cur.v]){
            if(res[n.v] > n.w + cur.w){
                res[n.v] = n.w + cur.w;
                pq.push({n.u, n.v, n.w + cur.w});
            }
        }
    }
    cout << ans.size() << '\n';
    for(edge t: ans){
        cout << t.u << ' ' << t.v << '\n';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 첫번째 조건은 서로 다른 두 컴퓨터 간에 통신이 가능하도록 최소개수의 회선만 복구해야 한다고 되어 있다. 이는 말 그대로 해석하여, n개의 vertex가 있을 때 n-1개의 회선만 복구해야 한다는 의미이다.
  • 두번째 조건은 슈퍼컴퓨터가 다른 컴퓨터와 통신하는데 걸리는 최소 시간이, 원래의 네트워크에서 통신하는 데 걸리는 최소 시간보다 커지면 안 된다고 되어 있다. 이는 dijkstra로 해결해야 하는 것을 의미한다.
  • 출력은 복구한 회선들이다. 이는 graph에서 edge를 출력해야 하는 것과 같다
  • 이 문제는 간단히 슈퍼컴퓨터에서 다른 모든 컴퓨터까지 dijkstra로 탐색을 하고, 그 경로를 출력하기만 하면 된다.
  • 왜냐하면 dijkstra는 일단 vertex들을 한번씩만 방문을 한다. 이 의미는 모든 vertex를 방문하는 데 n-1 개의 edge만 거친다는 의미이다. 이렇게 방문한 vertex까지의 경로는 최단 경로가 된다. 따라서 dijkstra로 탐색하며 경로를 출력하면 두가지 조건을 모두 만족시킬 수 있게 된다.

느낀점

  • 오래 고민을 했는데 결국 풀지 못했다. 나는 '최소개수'의 의미를 n-1로 생각하지 못하고, n-1보다 커질 수도 있다고 생각을 했다. 그래서 mst로 탐색을 하며, edge를 고를 때 dijkstra로 탐색을 해서 두번째 조건을 만족시키는 경우 계속 탐색을 하도록 했다. 돌이켜 보면 결국 mst도 n-1개의 edge를 고르는 것이라고 할 수 있다. 또한 내 풀이 자체도 이상하다고 생각한다. dijkstra에 대해 제대로 이해하지 못해서 풀지 못했다고 생각을 한다.
  • 위의 dijkstra 코드에서 done 배열만이 아닌, 추가로 vertex들 까지의 현재 까지의 최소 경로 길이 배열을 추가하여 pq에 넣을 때 최소 경로 길이 배열의 값보다 작은 경우만 넣어주면은 알고리즘 동작 시간이 반으로 줄어든다. 따라서 dijkstra에서는 이렇게 하는 것이 좋을 듯 하다.
    • mst에서는 이렇게 하지 못할 것이라고 생각한다.
