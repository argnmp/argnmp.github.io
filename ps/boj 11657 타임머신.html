#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct edge{
    int u, v, w;
};

int n, m;
vector<vector<edge>> graph(501, vector<edge>());

void solve(){
    cin >> n >> m;    
    int u, v, w;
    for(int i = 0; i<m; i++){
        cin >> u >> v >> w;
        graph[u].push_back({u, v, w});
    }

    ll dist[501]; 
    fill(&dist[0], &dist[501], LONG_LONG_MAX);
    dist[1] = 0;
    for(int i = 0; i<n; i++){
        for(int u = 1; u<=n; u++){
            if(dist[u]!=LONG_LONG_MAX){
                for(edge t: graph[u]){
                    if(dist[u] + t.w < dist[t.v])
                        dist[t.v] = dist[u] + t.w;
                }
            }
        } 
    }

    bool is_inf = false;
    for(int u = 1; u<=n; u++){
        if(dist[u]!=LONG_LONG_MAX){
            for(edge t: graph[u]){
                if(dist[u] + t.w < dist[t.v]){
                    is_inf = true;
                    break;
                }
            }
        }
    }
    if(is_inf)
        cout << -1;
    else {
        for(int i = 2; i<=n; i++){
            if(dist[i]==LONG_LONG_MAX)
                cout << -1 << '\n';
            else
                cout << dist[i] << '\n';
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}
