#include <iostream>
#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
priority_queue<int> maxq;
priority_queue<int, vector<int>, greater<int>> minq;

void solve(){
  cin >> n;   
  int k, maxqt = INT_MAX, minqt = INT_MAX;
  for(int i = 0; i<n; i++){
    cin >> k;
    if(i%2==0) maxq.push(k);
    else minq.push(k);

    if(i != 0 && maxq.top() > minq.top()){
      maxqt = maxq.top();
      minqt = minq.top();
      maxq.pop();
      minq.pop();
      maxq.push(minqt);
      minq.push(maxqt);
    }
    if(i==0){
      cout << maxq.top() << "\n";
    } else {
      cout << min(maxq.top(), minq.top()) << "\n";
    }
  }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • priority queue를 이용해서 반을 나눠서 작은 쪽은 max priority queue로, 큰 쪽은 min priority queue로 가운데 값을 찾아야 한다

느낀점

  • 아이디어를 도출해 내지 못했다
  • priority queue를 사용하는 법을 잊어버렸다.
    • priority_queue
  • 시간초과가 나길래 이유를 못찾았는데, endl 대신 '\n' 을 사용하니 해결이 되었다.
