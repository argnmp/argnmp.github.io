#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
struct matrix {
    int r, c;
};
int dp[501][501] = {0,}; 
void solve(){
    cin >> n;
    vector<matrix> mats;
    int r, c;
    for(int i = 0; i<n; i++){
        cin >> r >> c;
        mats.push_back({r,c});
    }
    for(int d = 1; d<n; d++){
        for(int i = 0; i<n-d; i++){
            // i ~ i+d
            dp[i][i+d] = INT_MAX;
            for(int j = i; j<i+d; j++){
                dp[i][i+d] = min(dp[i][i+d], dp[i][j] + dp[j+1][i+d] + mats[i].r * mats[j].c * mats[i+d].c);
            }
        }
    }
    cout << dp[0][n-1];
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • dp로 푸는 문제이다. 점화식을 세워서 풀이하면 된다.
  • dp 값을 계산하는 순서는 행렬 사이의 거리가 작은거 부터 큰 순으로 계산하면 된다.
    • 이를 계산할 때 거리를 늘려가면서 순회를 해야 하는데, 이 부분에서 조금 헷갈린다. 거리별 순회 이 코드를 이용하면 된다.
  • boj 11066 파일 합치기 과 동일한 문제이다.
