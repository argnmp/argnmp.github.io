#include <iostream>
#include <algorithm>
#include <climits>

#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m;
int mem[101]={0,};
int c[101]={0,};
// dp[c][i] = 비용이 c이고, i번째 앱까지 비활성화 했을 때, 최대 메모리 확보량
int dp[101][10001]={0,};
void solve(){
    cin >> n >> m;
    for(int i=1; i<=n; i++){
        cin >> mem[i];
    } 
    for(int i=1; i<=n; i++){
        cin >> c[i];
    }
    for(int i=1; i<=n; i++){
        for(int j=0; j<=10000; j++){
            if(j>=c[i]){
                dp[i][j] = max(dp[i-1][j-c[i]] + mem[i], dp[i-1][j]);
            }
            else {
                dp[i][j] = dp[i-1][j];
            }
        }
    }
    for(int i = 0; i<=10000; i++){
        if(dp[n][i]>=m){
            cout << i;
            return;
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • dp로 푸는 문제이다. knapsack과 같은 방식으로 풀이하면 된다.
  • dp[101][10001] 배열을 선언해서 dp[i][j] 를 j 비용을 사용 시 i번째 앱까지 (활성화 or 비활성화) 했을 경우 얻을 수 있는 메모리의 최댓값 으로 정의해서 풀면 된다.
  • dp 배열을 모두 구하고 dp[n][k] 에서 k를 0부터 10000까지 늘려가면서 해당 dp 배열의 값이 m 이상일 경우 가장 작은 k 값을 구하면 된다.
