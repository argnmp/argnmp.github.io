코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int g, p;
int uf[100001] = {0,};
int uf_find(int t){
    if(uf[t]==t)
        return t;
    return uf[t] = uf_find(uf[t]);
}
void uf_union(int a, int b){
    int ra = uf_find(a);
    int rb = uf_find(b);
    if(a > b)
        swap(a,b);
    uf[b] = a;
}

void solve(){
    for(int i = 0; i<100001; i++){
        uf[i] = i;
    }
    cin >> g >> p;
    int gate;
    int i;
    for(i = 1; i<=p; i++){
        cin >> gate; 
        int root = uf_find(gate);
        if(root == 0){
            break;
        }
        uf_union(root-1, root);
    }
    cout << i-1;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

union-find를 이용해서 풀어야 한다. 해당 비행기에 대해서, 도킹이 가능한 게이트의 범위가 주어진다. 4가 주어졌다면 1~4 번 게이트에 비행기를 도킹할 수 있다.

일단 중요한 아이디어는, 숫자가 가장 큰 게이트 부터 비행기를 도킹하는 것이다. 4가 주어졌을 때, 4부터 1까지 내려가면서 게이트에 도킹이 가능한지 확인을 하고, 가능하다면 그 게이트에 도킹을 한다.

하지만 이 방식을 그대로 사용하면, O(n^2)의 시간복잡도를 가지기 때문에 시간초과에 걸리게 된다. 이를 해결하기 위해서 union-find가 필요하다.

게이트가 주어지면 해당 게이트의 루트 노드에 비행기를 도킹하는 방식이라고 할 수 있다. 이 과정에서 union-find의 최적화 과정을 통해 검색횟수가 훨씬 줄어든다.

각 게이트를 uf배열의 각 원소로 생각하고, 처음에는 자기 자신을 부모 노드로 가지고 있는다. 그리고 입력을 받고, uf_find를 통해 루트 노드를 검사한다. 만약 루트 노드가 0이라면 1번 게이트까지 모두 비행기가 도킹되어 있는 것이기 때문에 더 이상 비행기를 받을 수 없는 상태가 된다. 만약 루트 노드가 0이 아니라면 해당 루트 노드에 비행기를 도킹한다고 생각하고, uf_union(root-1, root)를 통해서 비행기를 도킹할 수 있는 루트 노드로 갱신을 해준다.

느낀점

union-find로 풀어야 한다는 것은 알았지만, 풀이를 생각해내지 못했다.
