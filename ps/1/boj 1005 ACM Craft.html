첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, k, b;
int cost[1001] = {0,};
vector<vector<int>> graph(1001, vector<int>());
int dp[1001] = {0,};
int calc(int node){
    if(dp[node]!=-1)
        return dp[node];
    int max_cost = 0;
    for(int depend: graph[node]){
        max_cost = max(max_cost, calc(depend));
    }
    dp[node] = max_cost + cost[node];
    return max_cost + cost[node];
}
void solve(){
    memset(cost, 0, sizeof(cost));
    memset(dp, -1, sizeof(dp));
    graph = vector(1001, vector<int>());
    cin >> n >> k;
    for(int i = 1; i<=n; i++){
        cin >> cost[i];
    }
    int u, v;
    for(int i = 0; i<k; i++){
        cin >> u >> v; 
        graph[v].push_back(u);
    }
    cin >> b;
    cout << calc(b) << '\n'; 
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--)
        solve(); 
    return 0;
}

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, k, b;
int cost[1001] = {0,};
vector<vector<int>> graph;
int in[1001] = {0,};
int max_cost[1001] = {0,};
void solve(){
    memset(cost, 0, sizeof(cost));
    graph = vector(1001, vector<int>());
    memset(in, 0, sizeof(in));
    memset(max_cost, 0, sizeof(max_cost));
    cin >> n >> k;
    for(int i = 1; i<=n; i++){
        cin >> cost[i];
    }
    int u, v;
    for(int i = 0; i<k; i++){
        cin >> u >> v;
        graph[u].push_back(v);
        in[v] += 1;
    }
    queue<int> q; 
    for(int i = 1; i<=n; i++){
        if(in[i]==0)
            q.push(i); 
    }
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        for(int next: graph[cur]){
            max_cost[next] = max(max_cost[next], max_cost[cur] + cost[cur]);
            in[next] -= 1;
            if(in[next]==0){
                q.push(next);
            }
        }
    }
    cin >> b;
    cout << max_cost[b] + cost[b] << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--)
        solve(); 
    return 0;
}

풀이

나는 이 문제를 풀 때, topological sort를 반대로 적용했다.

출처: https://www.acmicpc.net/problem/1005, 백준 온라인 저지

위의 화살표를 반대로 하여 그래프의 edge를 추가했다. 그리고 건설해야 하는 건물의 번호에서 부터 재귀적으로 함수를 호출한다.

건설해야 하는 건물의 번호가 4일 때 이 건물을 건설하는데 걸리는 시간은, 의존성이 있는 건물(2와 3)들 중 가장 큰 시간에 자신의 건설시간을 더한 값이다.

이러한 연산을 의존성이 있는 건물들에 대해 재귀적으로 호출하여 계산한다.

위의 그림에서 1의 경우는 의존성이 없기 때문에 건설시간은 10초가 될 것이다.

재귀적으로 호출만 한다면 공통 경로가 존재하기 때문에 시간초과가 발생한다. 따라서, 한번 계산이 완료된 건물에 대해서는 건설 시간을 저장해두었다가 다시 사용하는 방식을 사용한다.

이렇게 푼 방식이 첫번째 풀이이다.

이와 달리 복잡하게 재귀를 생각하지 않아도, topological sort를 사용해서 바로 풀 수가 있다. boj 2252 줄 세우기

topological sort를 하기 위한 그래프를 먼저 생성한다. 이후 in edge의 개수가 0인 vertex가 queue에서 pop이 되며 탐색을 하는데, pop될 때마다 연결된 vertex의 in edge를 업데이트 해주고, 추가로 현재 까지의 건물 건설 비용을 갱신하며 저장해주면 된다.

이러한 갱신이 가능한 이유는 in edge의 개수가 0이 되는 순간, 해당 건물의 건설 시간이 고정되기 때문이다.

이렇게 푼 방식이 두번째 풀이이다.

느낀점

예전에 풀었던 문제라 익숙했지만, 두번째 풀이가 떠오르지 않아서 첫번째 풀이 대로 풀었다. 두번째 풀이 처럼 알고리즘 자체를 이용하는 방식이 더 좋을 것 같다.
