코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct tpl {
    // increasing order
    int e1, e2, e3;
    void align(){
        if(e2 < e1) swap(e1, e2);
        if(e3 < e1) swap(e1, e3);
        if(e3 < e2) swap(e2, e3);
    }
    tpl exec(int i){
        tpl next = tpl {e1, e2, e3};
        if(i%3==0){
            next.e2 -= next.e1;
            next.e1 += next.e1;
        }else if(i%3==1){
            next.e3 -= next.e1;
            next.e1 += next.e1;
        }else {
            next.e3 -= next.e2;
            next.e2 += next.e2;
        }
        return next;
    }
    void print(){
        cout << e1 << ' ' << e2 << ' ' << e3 << endl;
    }
};
int done[1000][1000] = {0,};
void solve(){
    tpl input; 
    cin >> input.e1 >> input.e2 >> input.e3;
    input.align();
    queue<tpl> q;
    q.push(input);
    done[input.e2][input.e3] = 1;
    while(!q.empty()){
        tpl cur = q.front();
        q.pop();
        if(cur.e1 == cur.e2 && cur.e2 == cur.e3){
            cout << 1;
            return;
        }
        for(int i = 0; i<3; i++){
            tpl next = cur.exec(i);
            next.align();
            if(done[next.e2][next.e3]==0){
                done[next.e2][next.e3] = 1;
                q.push(next);
            } 
        }
    } 
    cout << 0;
    return;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 모든 경우에 대해 연산을 수행해야 하는 문제이다. 하지만 중복이 되는 경우가 많기 때문에 이러한 경우의 수 들을 제외해주어야 한다.
  • a, b, c 모두 1 이상 500 이하의 수가 주어질 수 있고, 따라서 돌을 교환할 때, 최대 998까지 될 수 있다. 이 값들을 모두 기억하기 위해서는 int arr[1000][1000][1000] 정도 크기의 배열이 필요한데, 이는 메모리 초과가 발생하는 크기이다. 그러나 세개의 값들을 모두 기록할 필요가 없다. 왜냐하면, 돌의 총 개수는 변하지 않기 때문에, 두 개의 값을 알면 다른 한 개의 값은 정해질 수 밖에 없다. 따라서 두 개의 값만 기록하여 탐색을 하면 된다.
