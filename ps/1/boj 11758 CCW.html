코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct coord{
    int x, y;
};
/*
 * u = p1->p2 = (p2.x - p1.x, p2.y - p1.y, 0)
 * v = p1->p3 = (p3.x - p1.x, p3.y - p1.y, 0)
 * u x v = (p2.x - p1.x)*(p3.y - p1.y) - (p2.y - p1.y)*(p3.x - p1.x)
 */
int ccw(coord p1, coord p2, coord p3){
    return (p2.x - p1.x)*(p3.y - p1.y) - (p2.y - p1.y)*(p3.x - p1.x);
}
void solve(){
    coord p1, p2, p3; 
    cin >> p1.x >> p1.y;
    cin >> p2.x >> p2.y;
    cin >> p3.x >> p3.y;
    int ans = ccw(p1, p2, p3);
    if(ans < 0)
        cout << -1;
    else if(ans > 0)
        cout << 1;
    else
        cout << 0;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

백터의 외적을 이용해서 풀어야 하는 문제이다.

p1, p2, p3 순서대로 연결했을 때, 반시계 방향인지 시계방향인지 평행한지를 알아내야 한다. 출처: https://gaussian37.github.io/math-algorithm-ccw/ Jinsol Kim 님의 블로그에 있는 다이어그램입니다.

위의 첫번째 사진 처럼 p1, p2, p3가 배열되어 있을 때, 이 순서대로 선분을 그으면(사진의 화살표와 다름) 시계방향이 된다. 이를 판단하기 위해서 p1을 기준으로 p2, p3를 향하는 벡터 v1, v2(사진의 화살표)를 그려본다. v1과 v2를 외적하면 (v1 x v2), 간단히 오른손 법칙으로 생각해 볼 때 벡터는 음의 방향을 향하게 된다. 외적의 계산 값이 음수일 때에는 시계 방향이라고 판단할 수 있다.

위의 세번째 사진처럼 p1, p2, p3가 배열되어 있을 때, 이 순서대로 선분을 그으면(사진의 화살표와 다름) 반시계방향이 된다. 마찬가지로 이를 판단하기 위해서 p1을 기준으로 p2, p3를 향하는 벡터 v1, v2(사진의 화살표)를 그려본다. v1과 v2를 외적하면 (v1 x v2), 간단히 오른손 법칙으로 생각해 볼 때 백터는 양의 방향을 향하게 된다. 외적의 계산 값이 양수일 때에는 반시계 방향이라고 판단할 수 있다.

위의 두번째 사진처럼 p1, p2, p3가 배열되어 있는 경우, 두 벡터가 그리는 면적이 0이므로 외적의 계산 값이 0이 된다. 따라서 0일 때에는 일직선으로 판단할 수 있다.

이를 토대로 계산을 해보면 다음과 같다.

struct coord{
    int x, y;
};

풀이에서와 같이 p1, p2, p3가 위의 구조체로 정의되어 있다고 가정한다. (z 값은 0으로 생각한다.)

각 벡터는 다음과 같이 표현된다.

  • v1은 p1 에서 p2를 향하는 벡터이다. 따라서 v1 = (p2.x - p1.x, p2.y - p1.y, 0) 이다.
  • v2는 p1에서 p3를 향하는 벡터이다. 따라서 v2 = (p3.x - p1.x, p3.y - p1.y, 0) 이다.

v1 와 v2의 외적은 다음과 같이 표현된다.

    1. 외적은 v1 x v2 = (v1.y*v2.z - v1.z*v2.y, v1.x*v2.z - v1.z*v2.x, v1.x*v2.y - v1.y*v2.x) 으로 3차원 좌표로 계산한다.
    1. p1, p2, p3는 모두 평면상의 좌표이므로, v1과 v2의 z 값이 0 이었다.
    1. 따라서 v1 x v2 = (0, 0, v1.x*v2.y - v1.y*v2.x) = (0, 0, (p2.x - p1.x)*(p3.y - p1.y) - (p2.y - p1.y)*(p3.x - p1.x) 이다.

p1, p2, p3가 모두 평면 상에 있기 때문에, 결과로 나오는 벡터는 수직방향이 되어서 x값과 y값은 0이 된다. z값은 오른손 법칙의 엄지와 같은 의미를 가지고, 이 값에 따라 시계, 반시계, 일직선이 정해진다.
