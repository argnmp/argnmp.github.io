#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
int from, to;
struct edge {
    int v, c;
};
struct box {
    int u, v, c;
    bool operator()(box& a, box& b){
        return a.c < b.c; 
    }
};
vector<vector<edge>> graph(100001, vector<edge>());
vector<vector<edge>> mst_graph(100001, vector<edge>());

int ans = INT_MAX;
int visited[100001] = {0,};
bool calc(int node){
    // cout << node << endl;
    visited[node] = 1;
    for(const auto& next: mst_graph[node]){
        // cout << next.v <<":" << next.c << endl;
        if(next.v==to){
            ans = min(ans, next.c); 
            return true;
        }
        if(visited[next.v]==0 && calc(next.v)){
            ans = min(ans, next.c); 
            return true;
        }
    }
    return false;
}
void solve(){
    cin >> n >> m;
    int u, v, c; 
    for(int i = 0; i<m; i++){
        cin >> u >> v >> c;
        graph[u].push_back({v, c});
        graph[v].push_back({u, c});
    }
    cin >> from >> to;

    priority_queue<box, vector<box>, box> pq; 
    int done[100001] = {0,};
    done[from] = 1;
    for(const auto& t: graph[from]){
        pq.push({from, t.v, t.c});
    }
    while(!pq.empty()){
        box cur = pq.top();
        pq.pop();
        if(done[cur.v]==0){
            done[cur.v] = 1;
            mst_graph[cur.u].push_back({cur.v, cur.c});
            mst_graph[cur.v].push_back({cur.u, cur.c});
            // cout << cur.u << "->" << cur.v << endl;
        }
        else {
            continue;
        }
        for(const auto& next: graph[cur.v]){
            if(done[next.v]==0){
                pq.push({cur.v, next.v, next.c});
            }
        } 
    }
    calc(from);
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제에서 공장이 있는 두 섬을 연결하는 경로는 항상 존재한다고 하였다.

따라서 주어진 그래프를 가지고, minimum spanning tree를 생성하는 로직을 이용해서 load가 큰 edge부터 연결하도록, 즉 maximum spanning tree를 만들어 준다.

나는 위에서 prim's algorithm을 사용했기 때문에, max priority queue를 이용했다. 항상 load가 큰 edge가 연결되므로 from부터 mst를 만들어 나가면 to까지 가는 유일한 경로는 load들이 큰 edge로만 구성이 된다.

이후 from에서 to까지 가는 경로에서 가장 최소의 load를 찾아주면 답이 된다. 경로를 찾는 방법은 dfs와 백트래킹을 이용하면 된다.

두가지 개선점이 있다.

  1. mst를 만들고 나중에 다시 경로를 찾아가는 것이 아닌, prim's algorithm으로 탐색을 하는 과정에서 to를 만난 순간의 edge의 load가 바로 답이 된다. 현재까지 탐색한 edge들 중 load가 최솟값으로 to까지의 경로가 연결되었기 때문이다.
  2. union-find를 이용한 kruskal algorithm이 좀 더 직관적이다.

두번째 풀이: bfs + parametric search

다른 풀이를 검색해보았는데, 이분 탐색을 이용해서 from에서 to까지 연결 가능한 load의 최댓값을 찾아줄 수 있다.
