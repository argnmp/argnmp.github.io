#include <cmath>
#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
struct coord {
    double x, y;
};
/*
 * u = (p2.x-p1.x, p2.y-p1.y, 0)
 * v = (p3.x-p1.x, p3.y-p1.y, 0)
 * u x v = (0, 0, )
 */
double ccw(coord p1, coord p2, coord p3){
    return (p2.x-p1.x)*(p3.y-p1.y) - (p2.y-p1.y)*(p3.x-p1.x);
}
void solve(){
    cin >> n; 
    coord c;
    cin >> c.x >> c.y;
    coord a, b;
    cin >> a.x >> a.y;
    double ans = 0;
    for(int i = 0; i<n-2; i++){
        cin >> b.x >> b.y;
        ans += ccw(c, a, b)/2; 
        swap(a, b);
    }
    printf("%.1lf", abs(round(ans*10)/10));
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

boj 11758 CCW 에서 벡터의 외적을 사용한 것 처럼, 이 문제에서도 벡터의 외적을 사용하면 된다.

평면의 두 벡터를 외적한 결과물은, 이 벡터들이 이루는 평행사변형의 넓이와 같다. 이 넓이를 2로 나누면, 벡터의 시작점과 끝점들이 이루는 삼각형의 넓이를 구할 수 있다.

https://ip99202.github.io/posts/%EB%B0%B1%EC%A4%80-2166-%EB%8B%A4%EA%B0%81%ED%98%95%EC%9D%98-%EB%A9%B4%EC%A0%81/ 이 링크에 있는 볼록다각형과 오목다각형을 각각 A, B라 하겠다.

그림 A와 B모두, a -> b -> c -> d -> e -> f 순서대로 점이 주어진다. 그렇기 때문에, (a, b, c), (a, c, d), (a, d, e), (a, e, f) 순서대로 외적을 하면 삼각형의 넓이를 얻을 수 있다.

그림 A의 경우 위의 순서대로 외적을 하면, 결과 값이 모두 음수가 나온다.(시계방향이므로) 하지만 그림 B의 경우 위의 순서대로 외적을 하면, (a, b, c)는 양수, (a, c, d)는 음수가 나온다. 필요한 것은 삼각형(a, c, d)에서 삼각형(a, b, c)를 뺀 넓이 이므로, 순서대로 모두 외적한 결과 값을 더하고 마지막 최종 결과값에 절댓값을 만들어주어야 한다.

느낀점

이전에 풀어본 문제이다. 마지막에 절댓값을 만들어주어야 한다는 것을 생각하지 못하였다.
