코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
struct pair_hash{
    inline size_t operator()(const pair<int, string>& p) const{
        return hash<int>()(p.first) ^ hash<string>()(p.second);
    }
};
unordered_map<int, string> key_map;
int g_idx = 1;

vector<vector<int>> graph(15001, vector<int>());

struct box{
    int next;
    string key;

    bool operator<(const box& a) const
    {
        return key < a.key;
    }
};

void in_order(int node, int level){
    for(int i = 0; i<level*2; i++){
        cout << '-';
    } 
    if(level != -1){
        cout << key_map[node] << '\n';
    }
    vector<box> search;
    for(int next: graph[node]){
        search.push_back({next, key_map[next]});
    }
    sort(search.begin(), search.end());
    for(box n: search){
        in_order(n.next, level + 1);
    }
    return;
}

void solve(){
    cin >> n;
    int t;
    for(int i = 0; i<n; i++){
        cin >> t;
        string s;
        int prev = 0;
        for(int j = 0; j<t; j++){
            cin >> s;
            int s_idx;
            bool is_exist = false;
            for(int next: graph[prev]){
                if(s == key_map[next]){
                    s_idx = next;
                    is_exist = true;
                    break;
                }
            }
            if(!is_exist){
                s_idx = g_idx++;
                key_map.insert({s_idx, s});
                graph[prev].push_back(s_idx);
            }
            prev = s_idx;
        }
    }
    in_order(0, -1);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

트리 그래프를 만들어주어야 한다.

로봇 개미가 이동한 경로를 입력받는 과정에서, 이전 방과 현재 입력 받은 방을 edge로 이어준다. 이 과정에서 만약 이미 이어져 있다면 새로 연결하지 않는다. 이미 이어져 있지 않은 경우는 unique하기 때문에 key(이름)에 graph의 vertex를 새롭게 할당해준다.

이렇게 트리 그래프를 만들면 방의 이름은 같지만 다른 vertex에 해당하는 경우가 생길 수 도 있다. 그래서 방의 이름을 index로 사용하지 않고, 별도의 int 형 index를 사용한다. 나중에 역으로 index를 통해 key값을 출력해야 하므로 unordered_map을 사용해서 저장해둔다.

이후 출력은 트리의 in order과 같기 때문에, 이 순서대로 출력해주면 된다.

느낀점

처음에 같은 방 이름에 대한 처리를 제대로 하지 않아서, 푸는데 시간이 약간 걸렸다.
