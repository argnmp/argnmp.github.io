코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int uf[200001] = {0,};
int uf_size[200001] = {0,};
int uf_find(int v){
    if(uf[v]==v){
        return v;
    }
    uf[v] = uf_find(uf[v]);
    return uf[v];
}

// return current size
int uf_union(int a, int b){
    int ra = uf_find(a);
    int rb = uf_find(b); 
    if(ra != rb){
        uf[rb] = ra;
        uf_size[ra] += uf_size[rb];
    }
    return uf_size[ra];
}
void solve(){
    for(int i = 0; i<=200000; i++){
        uf[i] = i;
        uf_size[i] = 1;
    }
    int idx_cursor = 0;
    unordered_map<string, int> idx_map; 
    int f;
    cin >> f; 
    string a, b;
    for(int i = 0; i<f; i++){
        cin >> a >> b;
        int a_idx, b_idx;
        if(idx_map.find(a)==idx_map.end()){
            idx_map.insert({a, idx_cursor});
            idx_cursor += 1;
        }
        a_idx = idx_map[a];
        if(idx_map.find(b)==idx_map.end()){
            idx_map.insert({b, idx_cursor});
            idx_cursor += 1;
        }
        b_idx = idx_map[b];

        int size = uf_union(a_idx, b_idx);
        cout << size << '\n';
    }    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int test;
    cin >> test;
    while(test--){
        solve(); 
    }
    return 0;
}

풀이

union-find를 활용하는 문제이다. boj 1717 집합의 표현 root node에서 계속 하위 네트워크에 속한 node의 개수를 기록해주어야 한다. 이러한 기록은 따로 배열을 만들어서, union 시 서로 다른 집합에 속해있는 경우에만 union 대상의 집합에 속한 노드의 개수를 더해주면 된다. 그리고 친구 관계가 최대 100000이므로, node의 개수는 최대 200000까지 될 수 있다. 또한, unordered_map을 사용해서 이름과 uf배열에서의 index를 매핑해주면 된다.

느낀점

union-find를 알고 있어서 풀 수 있었다.
