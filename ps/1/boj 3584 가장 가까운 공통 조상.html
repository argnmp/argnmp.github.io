문제

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int parent[10001] = {0,};
int n;
void solve(){
    memset(parent, 0, sizeof(parent));
    cin >> n;
    int u, v;
    for(int i = 1; i<n; i++){
        cin >> u >> v;
        parent[v] = u;    
    }
    int a, b;
    cin >> a >> b;
    unordered_set<int> par;
    int cur = a;
    while(cur!=0){
        par.insert(cur);
        cur = parent[cur];
    }
    cur = b;
    while(cur!=0){
        if(par.find(cur)!=par.end()){
            cout << cur << '\n';
            return;
        }
        cur = parent[cur];
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--)
        solve(); 
    return 0;
}

풀이

트리에서 두 vertex의 가장 가까운 공통 조상을 찾는 문제이다. 트리에서 edge의 개수는 n-1개 이다. (여기서 n은 전체 vertex의 개수)

한 vertex의 부모 vertex를 모두 unordered_set 에 저장해둔 뒤, 다른 vertex에서 부모 vertex들을 찾아 올라가며 set에 있는지 확인하면 된다.

느낀점

트리 문제를 이전에 풀어보아서 어렵지 않게 풀 수 있었다.
