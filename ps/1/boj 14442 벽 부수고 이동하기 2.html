#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, k;
char board[1000][1000] = {0,};
int dx[4] = {0,-1,1,0};
int dy[4] = {-1,0,0,1};
struct pos {
    int x, y;
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<m; }
    pos adj(int i){
        return {x+dx[i%4], y+dy[i%4]};
    }
    void print(){
        cout << x << ',' << y << '\n';
    }
};
struct box {
    pos p;
    int dist;
    int wall;
};
int visited[1000][1000][11] = {0,};
void solve(){
    cin >> n >> m >> k;
    for(int i=0; i<n; i++){
        for(int j = 0; j<m; j++){
            char ch = cin.get();
            while(ch=='\n' || ch=='\r' || ch==' '){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    }
    queue<box> q; 
    q.push({{0,0}, 0, k});
    visited[0][0][k] = 1;
    while(!q.empty()){
        box cur = q.front();
        q.pop();
        if(cur.p.x==n-1 && cur.p.y==m-1){
            cout << cur.dist + 1;
            return;
        }
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe()){
                if(board[next.x][next.y]=='0' && visited[next.x][next.y][cur.wall]==0){
                    visited[next.x][next.y][cur.wall] = 1;
                    q.push({next, cur.dist + 1, cur.wall});
                } else if(board[next.x][next.y]=='1' && cur.wall > 0 && visited[next.x][next.y][cur.wall-1]==0){
                    visited[next.x][next.y][cur.wall-1] = 1;
                    q.push({next, cur.dist + 1, cur.wall-1});
                }
                
            }
        }
    }
    cout << -1;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

int visited_wall_max[1000][1000] = {0,};
void solve(){
    memset(visited_wall_max, -1, sizeof(visited_wall_max));
    cin >> n >> m >> k;
    for(int i=0; i<n; i++){
        for(int j = 0; j<m; j++){
            char ch = cin.get();
            while(ch=='\n' || ch=='\r' || ch==' '){

            }
            board[i][j] = ch;
        }
    }
    queue<box> q; 
    q.push({{0,0}, 0, k});
    visited_wall_max[0][0] = k;
    while(!q.empty()){
        box cur = q.front();
        q.pop();
        if(cur.p.x==n-1 && cur.p.y==m-1){
            cout << cur.dist + 1;
            return;
        }
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe()){
                if(board[next.x][next.y]=='0' && visited_wall_max[next.x][next.y] < cur.wall){
                    visited_wall_max[next.x][next.y] = cur.wall;
                    q.push({next, cur.dist + 1, cur.wall});
                } else if(board[next.x][next.y]=='1' && cur.wall > 0 && visited_wall_max[next.x][next.y] < cur.wall-1){
                    visited_wall_max[next.x][next.y] = cur.wall-1;
                    q.push({next, cur.dist + 1, cur.wall-1});
                }
                
            }
        }
    }
    cout << -1;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

bfs를 사용해서 [0,0]에서 [n-1, m-1]에 도달해야 하는 문제이다. vertex 간의 거리가 같기 때문에 bfs로 탐색할때 먼저 방문하는 경우가 최단거리가 된다.

이 문제에서는 방문한 vertex를 다시 방문할 수 있다. 단 해당 vertex에 방문 했을 시 부술 수 있는 벽의 개수를 기록해서, 이 벽의 개수로 방문하지 않았던 경우만 다시 방문할 수 있다.

따라서 visited 배열을, visited[x좌표][y좌표][부술 수 있는 벽의개수]로 설정해서 탐색하면 된다.

이 풀이가 기본 풀이에 해당된다.

탐색 시간을 조금 더 최적화 할 수 있다. 부술 수 있는 벽의 개수를 10으로 특정 vertex를 방문했을 때, 부술 수 있는 벽의 개수를 3으로 특정 vertex를 다시 방문할 필요가 없다. 왜냐하면 10으로 방문했을 때가 더 좋기 때문이다.

이 풀이는 최적화된 풀이에 해당된다. visited_wall_max 배열을 이용해서, 같은 vertex에 방문 시 부술 수 있는 벽의 개수가 이전에 방문했을 때보다 더 많은 경우에만 방문하도록 한다.
