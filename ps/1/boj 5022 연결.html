코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m;
int dx[4] = {0,0,1,-1};
int dy[4] = {1,-1,0,0};
struct pos{
    int x, y, dist;
    bool safe(){ return 0<=x&&x<=n&&0<=y&&y<=m; }
    pos adj(int i){
        return {x+dx[i%4], y+dy[i%4]};
    }
    void print(){
        cout << x << ',' << y << endl;
    }
};

int calc(pos a1, pos a2, pos b1, pos b2){
    int board[101][101] = {0,};
    int a_dist = INT_MAX;
    int b_dist = INT_MAX;
    // connect a1 to a2 using bfs
    int done[101][101] = {0,};

    queue<pair<pos, vector<pos>>> q;
    q.push({a1, vector<pos>()}); 
    done[a1.x][a1.y] = 1;
    while(!q.empty()){
        pos cur = q.front().first;  
        vector<pos> path = q.front().second;
        q.pop();
        if(cur.x==a2.x && cur.y==a2.y){
            path.push_back(cur);
            for(pos t: path){
                board[t.x][t.y] = 1;
            }
            a_dist = path.size()-1;
            break;
        }
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i);
            if(next.x==b1.x&&next.y==b1.y)
                continue;
            if(next.x==b2.x&&next.y==b2.y)
                continue;
            if(next.safe() && done[next.x][next.y]==0){
                done[next.x][next.y] = 1; 
                vector<pos> new_path;
                for(int i = 0; i<path.size(); i++){
                    new_path.push_back(path[i]);
                }
                new_path.push_back(cur);
                q.push({next, new_path});
            }
        } 
    }
    if(a_dist==INT_MAX)
        return INT_MAX;

    memset(done, 0, sizeof(done));
    queue<pos> rq;
    b1.dist = 0;
    rq.push(b1);
    done[b1.x][b1.y] = 1;
    while(!rq.empty()){
        pos cur = rq.front();
        rq.pop();
        if(cur.x==b2.x && cur.y==b2.y){
            b_dist = cur.dist;
            return a_dist + b_dist;
        }
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i);
            if(next.safe() && board[next.x][next.y]==0 && done[next.x][next.y]==0){
                done[next.x][next.y] = 1;
                next.dist = cur.dist + 1;
                rq.push(next);
            }
        }
    }
    
    return INT_MAX;
}
void solve(){
    cin >> m >> n;  
    pos a1, a2, b1, b2; 
    cin >> a1.y >> a1.x;
    cin >> a2.y >> a2.x;
    cin >> b1.y >> b1.x;
    cin >> b2.y >> b2.x;
    int res = min(calc(a1,a2,b1,b2), calc(b1,b2,a1,a2));
    if(res==INT_MAX){
        cout << "IMPOSSIBLE" << '\n';
    }
    else {
        cout << res << '\n';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

a1과 a2, b1과 b2를 연결해야 하는데, 연결한 전선의 길이의 합이 최소가 되어야 한다. 모든 경우의 수를 다 따져봐야 할 것 같지만, 두 연결 중 하나를 최단 거리로 만들고 다른 하나를 앞서 연결한 전선을 지나칠 수 없게 하며 최단 거리로 연결하여 답을 구할 수 있다. a를 먼저 연결하고 그다음 b를 연결하는 경우와, b를 먼저 연결하고 그다음 a를 연결하는 경우를 모두 수행해보고, 전선의 합이 더 작은 것이 답이 된다.

가장 주의할 사항은 다음과 같다. a를 먼저 연결하고 b를 연결한다고 했을 때, a를 연결시 사용되는 최단거리가 b의 양 끝점을 지나지 않도록 해야한다는 것이다. 나는 이것을 생각하지 못해서, 다른 방법을 사용했지만 결국 틀렸다.

vertex를 잇는 모든 edge의 weight가 동일하기 때문에, 최단 거리를 구할 때 bfs를 사용해도 된다.

느낀점

나는 푸는데 굉장히 오랜 시간이 걸렸다. 처음에 나도 위의 풀이와 같이, 두 연결 중 하나를 최단 거리로 만들고 다른 하나를 최단거리로 이으려는 아이디어를 떠올렸다.

a를 먼저 연결하고 그다음 b를 연결한다고 가정해보자. 나는 여기서 a를 최단거리로 연결하는 경우의 수가 많기 때문에, a를 꼭짓점으로 하는 직사각형 전체를 사용한다고 가정해서, b를 연결시 이 직사각형의 vertex를 사용해서 연결하지 못하도록 하였다. 만약 b점 자체가 이 직사각형에 포함될 경우, b점에서 출발은 가능하도록 하였다. 그리고 또 b를 꼭짓점으로 하는 직사각형 전체가 a에 들어가는 경우, 반대로 b를 먼저 연결하고 a를 연결하는 경우의 수에서 해결해주었다. 이외의 다른 케이스들도 고려해주었다.

결과적으로 이러한 풀이는 틀렸고, 위의 고려사항을 a를 연결시 b를 지날 수 없게 하는 방식으로 해결이 가능했다.

해답을 확인한 이후에도 틀렸습니다가 나왔고, 도저히 해결할 수 없었다. 결국 이 알고리즘 대회의 채점 테스트케이스들을 찾아서 확인해보았고, 코드에 조그마한 오류가 있었음을 확인했다. 더욱 정확하게 코드를 작성해야 할 것이다.
