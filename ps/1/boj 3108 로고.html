#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct pos{
    int x, y;
    void print(){
        cout << x << ',' << y << endl;
    }
};
pos board[1001][1001];
int crank[1001][1001];
int n;
pos find(pos cur){
    pos parent = board[cur.x][cur.y];
    if(parent.x==cur.x && parent.y==cur.y){
        return cur;
    } else {
        pos root = find(parent);
        board[cur.x][cur.y] = root; 
        return root;
    }
}
void cunion(pos a, pos b){
    pos ap = find(a); 
    pos bp = find(b);
    if(crank[ap.x][ap.y] < crank[bp.x][bp.y]){
        swap(ap, bp);
    }   
    board[bp.x][bp.y] = ap;
    if(crank[ap.x][ap.y]==crank[bp.x][bp.y]){
        crank[ap.x][ap.y] += 1;
    }
}
void solve(){
    for(int i = 0; i<1001; i++){
        for(int j = 0; j<1001; j++){
            board[i][j] = pos {i, j};
            crank[i][j] = 1;
        }
    }
    pos from, to;
    vector<pos> check_set;
    cin >> n; 
    for(int i = 0; i<n; i++){
        cin >> from.x >> from.y >> to.x >> to.y; 
        from.x += 500;
        from.y += 500;
        to.x += 500;
        to.y += 500;
        for(int i = from.y; i<=to.y; i++){
            pos cursor1 = {from.x, i};
            pos cursor2 = {to.x, i};
            cunion(from, cursor1);
            cunion(from, cursor2);
        }
        for(int i = from.x; i<=to.x; i++){
            pos cursor1 = {i, from.y};
            pos cursor2 = {i, to.y};
            cunion(to, cursor1);
            cunion(to, cursor2);
        }
        check_set.push_back(from);
    }
    pos start = find({500,500});
    set<pair<int, int>> ans; 
    for(pos s: check_set){
        pos root = find(s);
        if(start.x != root.x || start.y != root.y)
            ans.insert({root.x, root.y}); 
    } 
    cout << ans.size();
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 문제가 조금 길게 설명되어 있지만, 결국에는 직사각형이 변과 꼭지점으로 연결되 있는 직사각형의 집합이 몇개 인지를 구해야 한다.
    • 처음에 (0,0) 에서 출발하기 때문에, 이 좌표를 지나는 직사각형은 세지 말아야 한다.
  • 일단은, 각 직사각형이 지나는 모든 좌표가 같은 집합에 들어가야 한다.
  • 여러 직사각형이 있으면, 공통의 좌표를 지나는 직사각형 들은 같은 집합에 속해있어야 한다.
  • 집합은 union-find로 해결해야 한다. 위 문제는 같은 좌표를 지나는 직사각형에 속하는 모든 다른 좌표들은 같은 집합에 속해야 하기 때문에, 2차원 좌표를 대상으로 union-find 연산을 해주면 된다.

느낀점

  • union-find 자료구조 구현이 잘 기억이 나지 않아서, 구현 방법에 대해 검색을 하여 참고했다.
