첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
vector<vector<int>> graph(501, vector<int>());
vector<vector<int>> graph_rev(501, vector<int>());
int in[501] = {0};
int in_rev[501] = {0};

void solve(){
    cin >> n >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v; 
        graph[u].push_back(v);
        in[v] += 1;
        graph_rev[v].push_back(u);
        in_rev[u] += 1;
    }

    int cur_count[501][501] = {0,};
    queue<int> q;
    for(int i = 1; i<=n; i++){
        if(in[i]==0){
            q.push(i);
        }
    }
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        for(int next: graph[cur]){
            for(int i = 1; i<=500; i++){
                if(cur_count[cur][i]==1)
                    cur_count[next][i] = 1;
            }
            cur_count[next][cur] = 1;
            in[next] -= 1; 
            if(in[next]==0)
                q.push(next);
        }
    }
    int cur_count_rev[501][501] = {0,};
    for(int i = 1; i<=n; i++){
        if(in_rev[i]==0){
            q.push(i);
        }
    }
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        for(int next: graph_rev[cur]){
            for(int i = 1; i<=500; i++){
                if(cur_count_rev[cur][i]==1)
                    cur_count_rev[next][i] = 1;
            }
            cur_count_rev[next][cur] = 1;
            in_rev[next] -= 1; 
            if(in_rev[next]==0)
                q.push(next);
            
        }
    }

    int ans = 0;
    for(int i = 1; i<=n; i++){
        int acc = 0;
        for(int j = 1; j<=n; j++){
            if(cur_count[i][j]==1 || cur_count_rev[i][j]==1) 
                acc += 1;
        }
        if(acc==n-1)
            ans += 1;
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제에서 주어진 것처럼, 키 작은 사람 -> 키 큰 사람 으로 그래프를 만들어 준다.

먼저 그래프의 정방향으로 topological sort를 하여, 각 노드에서 보다 키가 작은 노드의 개수를 구해준다. 그리고 그래프의 역방향으로(화살표 방향을 반대로 한다) topological sort를 하여, 각 노드에서 보다 키가 큰 노드의 개수를 구해준다.

노드의 개수를 찾기 위해, 각 탐색마다 다음 노드에 현재 노드 이전의 노드들을 표시해 놓은 값들을 그대로 복사해준다.

이 두 값을 더해주었을 때 n-1이라는 것은 해당 노드보다 키가 작은 노드의 개수와 키가 큰 노드의 개수를 정확히 알 수 있다는 뜻이 된다.

두번째 풀이: dfs

정방향과 역방향으로 현재 탐색중인 노드에서 도달가능한 노드의 개수를 센다. 이 둘을 더했을 때 합이 n-1이면 정확한 순서를 알 수 있는 경우의 수가 된다.

세번째 풀이: floyd-warshall

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
int fdw[501][501] = {0,};
void solve(){
    fill(&fdw[0][0], &fdw[500][501], INT_MAX);
    cin >> n >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v;
        fdw[u][v] = 1;
    }
    for(int k = 1; k <= n; k++)
        for(int i = 1; i<=n; i++)
            for(int j = 1; j<=n; j++)
                if(fdw[i][k]!=INT_MAX && fdw[k][j]!=INT_MAX && fdw[i][k] + fdw[k][j] < fdw[i][j])
                    fdw[i][j] = fdw[i][k] + fdw[k][j];


    int ans = 0;
    for(int target = 1; target <= n; target += 1){
        int acc = 0;
        for(int t = 1; t<=n; t++){
            if(fdw[target][t]!=INT_MAX || fdw[t][target]!=INT_MAX)
                acc += 1;
        }
        if(acc==n-1)
            ans += 1;
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

키 작은 사람 -> 키 큰 사람 으로 간선을 표시해주고, floyd-warshall 알고리즘을 돌린다.

이후 첫번째 사람부터 n번째 사람까지 순회하면서, 현재 사람보다 키 작은 사람과 키 큰 사람의 수를 더했을 때 n-1이면 키를 정확하게 알 수 있는 경우이다.

현재 i번째 사람을 탐색하고 있을 때, 키 작은 사람은 fdw[1~n][i] != INT_MAX인 값의 개수이고, 키 큰 사람은 fdw[i][1~n] != INT_MAX인 값의 개수이다. 도달 불가능한 경우라면 값이 INT_MAX일 것이기 때문이다.

느낀점

dfs나 floyd-warshall을 생각하지 못하고 조금 복잡하게 처리하였다. 주의하자.
