문제

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
int seq[1000] = {0,};
int dp[1000] = {0,};
int prev_idx[1000] = {0,};
void solve(){
    cin >> n; 
    for(int i = 0; i<n; i++){
        cin >> seq[i];
        dp[i] = 1;
        prev_idx[i] = i;
        for(int k = i-1; k>=0; k--){
            if(seq[k] < seq[i] && dp[i] < dp[k] + 1){
                prev_idx[i] = k;
                dp[i] = dp[k] + 1;
            }      
        }
    }
    int max_idx = max_element(dp, dp + n) - dp;
    cout << dp[max_idx] << '\n';
    vector<int> stack;
    int target = max_idx;
    while(target != prev_idx[target]){
        stack.push_back(seq[target]);
        target = prev_idx[target];
    }
    stack.push_back(seq[target]);
    while(!stack.empty()){
        int cur = stack.back();
        stack.pop_back();
        cout << cur << ' ';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

dp를 이용해서 풀이해야 한다.

  • 위에서 사용된 dp[idx] 의미는, 앞에서 부터 idx까지 가장 긴 부분 수열의 길이 이다. 그래서 주어진 수열을 탐색 할때 이전까지의 모든 수들에 대해서 dp[0] ~ dp[idx-1]를 탐색하며 현재 수에서의 가장 긴 부분 수열의 길이를 알아내야 한다. O(n^2) 의 시간 복잡도를 가진다.

  • 최대 최소 원소 구하기

느낀점

이전에 풀어봤던 문제이다. 하지만, 마지막에 전체 dp array를 한번 탐색해서 가장 긴 길이를 찾아야 하는데, 그렇게 하지 않고 n-1 의 값만 사용하는 것으로 잘 못 생각해서, 오류를 잡아내는데 시간이 좀 소요되었다.
