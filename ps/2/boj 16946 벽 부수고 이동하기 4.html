#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m;
int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};
int board[1000][1000] = {0,};
struct pos {
    int x, y;
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<m; }
    pos adj(int i){ return pos { x+dx[i%4], y+dy[i%4] }; } 
};

int visited[1000][1000] = {0,};
int id = 1;
int area[1000000] = {0,};
void bfs(pos cur){
    visited[cur.x][cur.y] = id;
    queue<pos> q;
    q.push(cur);
    int count = 1;
    while(!q.empty()){
        pos cur = q.front();
        q.pop();
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i);
            if(next.safe() && board[next.x][next.y] == 0 && visited[next.x][next.y] == 0){
                count += 1;
                visited[next.x][next.y] = id;
                q.push(next);
            }
        }
    }
    area[id] = count;
    id += 1;
}
void solve(){
    cin >> n >> m;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            int t = cin.get();
            while(t=='\r'||t=='\n'||t==' '){
                t = cin.get();
            }
            board[i][j] = t-48;
        }
    }

    for(int i = 0; i<n; i++){
        for(int j=0; j<m; j++){
            if(board[i][j]==0 && visited[i][j]==0){
                bfs(pos {i,j});  
            }
        }
    }
    
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            if(board[i][j]==1){
                pos cur = {i,j};
                set<int> ids;
                for(int k = 0; k<4; k++){
                    pos next = cur.adj(k);
                    if(next.safe() && board[next.x][next.y]==0){
                        ids.insert(visited[next.x][next.y]);
                    }
                }
                int sum = 1;
                for(auto id: ids){
                    sum += area[id];
                }
                cout << sum % 10;
            }
            else {
                cout << 0; 
            }
        }
        cout << endl;
    }
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • bfs를 통해 풀면 된다. 벽이 없는 좌표들에서 bfs를 통해 연결된 영역의 크기가 얼마인지 구하고, bfs를 도는 과정에서 id 값으로 같은 영역은 표시해둔다.
  • 이후 벽이 있는 좌표들에서 상하좌우 네 방향을 확인하는데, id 값을 이용해서 같은 영역인지 판단하고 같은 영역일 경우 한번만 영역의 크기를 더해주면 된다.
