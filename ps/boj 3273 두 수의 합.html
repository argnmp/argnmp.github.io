#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int save[100000] = {0,};
int calc[2000000] = {0,};
int n, x;
int res = 0;
void solve(){
  cin >> n; 
  for(int i = 0; i<n; i++){
    cin >> save[i];
  }
  cin >> x;
  for(int i = 0; i<n; i++){
    int t = x - save[i];
    if(t >= 0){
      res += calc[t];
    }
    calc[save[i]] += 1;
  }
  cout << res;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 입력으로 주어지는 숫자들을 하나씩 읽으면서, 필요한 숫자가 몇개 있는지 확인하고 그 수만큼 결과에 더해주면 된다.
  • 중복이 문제가 되는데, 사실 하나씩 새로 읽을 때마다 calc에 필요한 숫자를 갱신해주면 중복을 생각할 필요가 없다. 무조건 한 쌍 씩만 계산되기 때문이다.

느낀점

  • map을 써야하나 잠시 고민했지만, 복잡하게 생각하지 않아야 한다.
