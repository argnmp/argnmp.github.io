풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
struct coord{
    double x, y;
    bool operator<(const coord& t) const{
        if(y != t.y)
            return y < t.y;
        else
            return x < t.x;
    }
    void print(){
        printf("[%lf, %lf]\n", x, y);
    }
};
coord points[100000];
set<int> pset;
/*
 * u = (p2.x-p1.x, p2.y-p1.y, 0);
 * v = (p3.x-p1.x, p3.y-p1.y, 0);
 * u x v = (0, 0, (p2.x-p1.x)*(p3.y-p1.y)-(p2.y-p1.y)*(p3.x-p1.x))
 */
double ccw(coord p1, coord p2, coord p3){
    return (p2.x-p1.x)*(p3.y-p1.y)-(p2.y-p1.y)*(p3.x-p1.x);
}

struct angle_sort{
    coord cri;
    bool operator()(coord& a, coord& b){
        double cp = ccw(cri, a, b);
        if(cp!=0)
            return cp > 0;
        else
            return abs(a.x-cri.x) + abs(a.y-cri.y) < abs(b.x-cri.x) + abs(b.y-cri.y);
    }
};

void solve(){
    cin >> n; 
    for(int i = 0; i<n; i++){
        cin >> points[i].x >> points[i].y;
    }
    sort(points, points + n);
    angle_sort as = {{points[0]}};
    sort(points+1, points+n, as);
    vector<coord> stk = {points[0], points[1]};
    for(int i = 2; i<n; i++){
        while(1){
            coord p2 = stk[stk.size()-1];
            coord p1 = stk[stk.size()-2];
            coord p3 = points[i];
            double cp = ccw(p1, p2, p3);
            if(cp>0){
                stk.push_back(p3);
                break;
            }
            else if(cp<0){
                stk.pop_back();
            }
            else {
                stk.pop_back();
                stk.push_back(p3);
                break;
            }
        }
    }
    cout << stk.size();
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

convex hull에 속한 점의 개수를 구해야 하는 문제이다.

임의의 두 점을 골랐을 때, 이 두 점을 이은 직선을 기준으로 다른 모든 점이 왼쪽 또는 오른쪽에 위치하는 경우, 이 두 점은 convex hull에 속한다고 볼 수 있다. 하지만 이러한 방식으로 찾는다면 시간 내 해결이 불가능 하다.

따라서 convex hull을 찾아내는 알고리즘을 사용해야 하고, 위 코드는 graham scan을 이용하였다.

graham scan에서는 다음과 같은 방식을 사용한다. 전체 좌표 배열이 points 에 저장되어 있다고 가정한다.

  • y값이 작은 순서대로 전체 배열을 정렬하고, 0번째 index의 좌표를 시작점으로 생각한다.
  • 시작점을 제외한 다른 값(좌표)들을, 시작점에서 x축에 평행하게 그은 직선과 시작점과 좌표가 이루는 직선이 이루는 각도가 작은 순서대로 정렬한다. 만약 각도가 같다면 이는 두 좌표가 시작점과 같은 선상에 있는 것을 의미한다. 이 경우 시작점에 더 가까운 좌표가 앞으로 오도록 정렬한다.
    • 이는 굳이 새로운 벡터를 생성해서 정렬할 필요 없이, ccw를 사용해서 판단이 가능하다. a좌표와 b좌표가 있을 때, ccw(시작점, a, b) 와 ccw(시작점, b, a) 값을 이용해서 음수인지 양수인지를 확인할 수 있다.
  • 0번째 index의 좌표와 1번째 index의 좌표를 stack에 넣는다. 2번째 index 부터 끝까지 순회를 하며 다음의 작업을 수행한다.
    • stack의 상위 두 개 좌표와 현재 순회 중인 좌표에 대해 ccw 연산의 결과가 시계반대방향인지 확인한다. 만약 그러하다면 stack에 현재 순회 중인 좌표를 넣고 순회를 계속한다. 시계반대방향이 아니라면 stack을 pop하고 위 작업을 다시 수행한다.
    • 만약 ccw 연산의 결과가 0이라면 이는 일직선 상에 있는 경우이고, 문제에서는 양 끝의 점만을 포함하라고 하였으므로, stack을 pop한 뒤 현재 순회 중인 좌표를 넣고 순회를 계속한다.
