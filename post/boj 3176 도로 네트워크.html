풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, k;
struct edge{
    int v, dist;
};
vector<vector<edge>> graph(100001, vector<edge>());

struct box{
    int node;
    int min_dist;
    int max_dist;
};
box par[100001][18] = {0, };
int dep[100001] = {0, };

int visited[100001] = {0,};
void pre_lca(int node, int depth){
    dep[node] = depth;
    visited[node] = 1;
    for(edge next: graph[node]){
        if(visited[next.v]==0){
            par[next.v][0] = {node, next.dist, next.dist};
            pre_lca(next.v, depth+1); 
        } 
    }
}
void fill_lca(){
    for(int i = 1; i<18; i++){
        for(int j = 1; j<=n; j++){
            box target = par[j][i-1];
            int min_dist = min(target.min_dist, par[target.node][i-1].min_dist);
            int max_dist = max(target.max_dist, par[target.node][i-1].max_dist);
            par[j][i] = {par[target.node][i-1].node, min_dist, max_dist};
        }
    }
}

void calc(int a, int b){
    if(dep[a] > dep[b]) 
        swap(a, b);
    int min_dist = INT_MAX;
    int max_dist = INT_MIN;
    for(int i = 17; i>=0; i--){
        if(dep[b]-dep[a] >= (1<<i)){
            min_dist = min(min_dist, par[b][i].min_dist);
            max_dist = max(max_dist, par[b][i].max_dist);
            b = par[b][i].node;
        }
    }

    if(a==b){
        cout << min_dist << ' ' << max_dist << '\n';
        return;
    }
    
    for(int i = 17; i>=0; i--){
        box next_a = par[a][i];
        box next_b = par[b][i];
        if(next_a.node != next_b.node){
            a = next_a.node;
            b = next_b.node;
            min_dist = min(min_dist, next_a.min_dist);
            min_dist = min(min_dist, next_b.min_dist);
            max_dist = max(max_dist, next_a.max_dist);
            max_dist = max(max_dist, next_b.max_dist);
        }
    }
    box next_a = par[a][0];
    box next_b = par[b][0];
    min_dist = min(min_dist, next_a.min_dist);
    min_dist = min(min_dist, next_b.min_dist);
    max_dist = max(max_dist, next_a.max_dist);
    max_dist = max(max_dist, next_b.max_dist);
    cout << min_dist << ' ' << max_dist << '\n';
}



void solve(){
    cin >> n;
    int u, v, w;     
    for(int i = 0; i<n-1; i++){
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    } 
    pre_lca(1, 0);
    fill_lca();
    cin >> k;
    while(k--){
        cin >> u >> v;
        calc(u, v);
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

도시를 연결하는 N-1 개의 도로가 있고 모든 도시의 쌍에 유일한 경로가 있다고 하였으므로, 이 네트워크는 트리의 형태임을 알 수 있다.

트리에서 두 노드의 경로는 무조건 공통 조상을 거쳐가기 때문에, 이 공통 조상을 구하는 것이 핵심이다.

문제에서 주어지는 도시의 최대 개수가 100000 이므로 노드 하나씩 거슬러 올라가면 시간 초과가 발생하기 때문에, lca 알고리즘을 이용해서 공통 조상을 구해야 한다.

또한 가장 짧은 도로의 길이와 가장 긴 도로의 길이를 구하기 위해서, 앞서 구한 공통 조상 까지 노드 하나씩 거슬러가면 최악의 경우 99999번 거슬러 올라가야 하고, 문제에서 주어지는 K가 최대 100000 이므로 시간 초과가 발생한다.

따라서 마찬가지로 가장 짧은 도로의 길이와 가장 긴 도로의 길이를, lca 알고리즘에서 사용하는 sparse matrix를 만드는 과정에서 같이 저장해주어야 한다. 이후 lca로 공통 조상을 찾아가는 과정에서 O(lgN)의 시간에 답을 구할 수 있게 된다.

느낀점

lca 알고리즘 구현이 기억이 나지 않아서, 이전에 풀었던 코드를 다시 공부했다. boj 13511 트리와 쿼리 2 문제와 비슷하게 풀 수 있었다.
