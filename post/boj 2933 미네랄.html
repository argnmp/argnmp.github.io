코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int r, c, n;
char board[101][101] = {0,};
int stick[101] = {0,};
int dx[4] = { -1, 0, 1, 0};
int dy[4] = { 0, 1, 0, -1};
struct pos { 
    int x; 
    int y; 
    bool flag; 
    void print(){cout << '[' << x << ',' << y << ']' << endl;}
    bool safe(){
        return 0<=x&&x<r&&0<=y&&y<c;
    }
    pos adj(int i){
        return pos{x+dx[i%4], y+dy[i%4]};
    }
};
struct compare {
    bool operator()(pos& a, pos& b){
        if(a.y == b.y){
            return a.x < b.x;
        }
        return a.y > b.y;
    }
};

// true: from left, false: from right
pos throw_stick(int r, bool direction) {
    pos cur;
    cur.x = r;
    cur.y = 0;
    cur.flag = true;
    if(direction){
        for(int i = 0; i<c; i++){
            cur.y = i; 
            if(board[cur.x][cur.y]=='x'){
                board[cur.x][cur.y] = '.';
                return cur;
            }
        } 
    } else {
        for(int i = c-1; i>=0; i--){
            cur.y = i; 
            if(board[cur.x][cur.y]=='x'){
                board[cur.x][cur.y] = '.';
                return cur;
            }
        } 
    }
    cur.flag = false; 
    return cur;
}

void print_board(){
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            cout << board[i][j];
        }
        cout << endl;
    }
}

void process(int i) {

    pos cur = throw_stick(r-stick[i], i%2==0);
    if(!cur.flag) return;

    int colored[101][101] = {0,}; // 1: connected to ground , 2: above ground
    // color minerals connected to ground
    for(int j=0; j<c; j++){
        if(board[r-1][j]=='x'){
            queue<pos> q;
            q.push(pos{r-1,j});
            colored[r-1][j] = 1;
            while(!q.empty()){
                pos t = q.front();
                q.pop();
                for(int i = 0; i<4; i++){
                    pos next = t.adj(i);
                    if(next.safe() && colored[next.x][next.y]==0 && board[next.x][next.y]=='x'){
                        colored[next.x][next.y] = 1;
                        q.push(next);
                    }
                }           
            }
        }
    } 

    // cluster start pos, diff to ground
    vector<pos> above_cluster;

    // color minerals above ground
    for(int k=0; k<4; k++){
        pos next = cur.adj(k);
        if(next.safe() && board[next.x][next.y] == 'x' && colored[next.x][next.y] == 0){
            queue<pos> q;
            q.push(next);
            colored[next.x][next.y] = 2;
            while(!q.empty()){
                pos t = q.front();
                q.pop();
                for(int i = 0; i<4; i++){
                    pos n = t.adj(i);
                    if(n.safe() && colored[n.x][n.y]==0 && board[n.x][n.y]=='x'){
                        q.push(n);
                        colored[n.x][n.y] = 2;
                    }
                }           
            }
            above_cluster.push_back(next);
        }
    }
    
    priority_queue<pos, vector<pos>, compare> sorted_pos;
    for (auto cur: above_cluster){
        int min_diff = INT_MAX;

        queue<pos> q;
        q.push(cur);
        int visited[101][101] = {0,};
        visited[cur.x][cur.y] = 1;
        while(!q.empty()){
            pos t = q.front();
            q.pop();
            sorted_pos.push(t);

            // calculate min_diff
            for(int i=t.x+1; i<r; i++){
                if(colored[i][t.y]==2) continue;
                if(board[i][t.y]=='x'){
                    min_diff = min(min_diff, i-t.x-1);
                } else if(i==r-1) {
                    min_diff = min(min_diff, i-t.x); 
                }
            }
            for(int i = 0; i<4; i++){
                pos n = t.adj(i);
                if(n.safe() && colored[n.x][n.y]==2 && visited[n.x][n.y]==0){
                    visited[n.x][n.y] = 1;
                    q.push(n);
                }
            }           
        }
        while(!sorted_pos.empty()){
            pos t = sorted_pos.top();
            sorted_pos.pop();
            board[t.x + min_diff][t.y] = 'x';
            board[t.x][t.y] = '.';
        } 
    }
}

void solve(){
    cin >> r >> c;
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            char ch = cin.get();
            while(ch == '\n' || ch == ' ' || ch == '\r'){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    }
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> stick[i];
        process(i);
    }

    print_board();
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 문제를 단순하게 생각해야 한다.
  • 깊게 고민할 필요없이 탐색은 그냥 bfs를 해버리면 된다.
  • 막대를 던졌을 때 미네랄 클러스터가 나뉘는데, 이를 바닥에 붙어 있는 클러스터와 바닥과 연결되어 있지 않은 클러스터로 나뉘게 된다.
  • bfs를 통해서 이 두가지 클러스터를 구분하고, 바닥과 연결되어 있지 않은 클러스터의 경우, 특정 미네랄 위치에서 아래로 쭉 탐색하며 바닥에 연결되어 있는 클러스터 만을 기준으로 얼마나 떨어지는지 차이를 구한다. 바닥과 연결되어 있지 않은 클러스터를 만난 경우 무시하고 밑으로 계속 탐색한다
  • 클러스터를 밑으로 내릴 때에는 priority queue를 이용해서, 클러스터의 미네랄의 좌표들을 바닥과 가까운 순으로 정렬되게 해서 밑에 있는 것부터 차례로 옮겨야 한다.
