#include <iostream>
#include <algorithm>
#include <climits>
#include <numeric>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, l;
struct coord{
    double x, y;
    bool operator<(const coord& t) const {
        if(y!=t.y)
            return y < t.y;
        return x < t.x;
    }
    void print(){
        printf("[%lf, %lf]\n", x, y);
    }
};
coord points[1000];
double ccw(coord p1, coord p2, coord p3){
    return (p2.x-p1.x)*(p3.y-p1.y)-(p2.y-p1.y)*(p3.x-p1.x); 
}
struct angle_sort{
    coord org;
    bool operator()(coord& a, coord& b){
        double cp = ccw(org, a, b); 
        if(cp==0)
            return a < b;
        else
            return cp > 0;
    }
};
double dist(coord p1, coord p2){
    return sqrt(pow(abs(p1.x-p2.x), 2) + pow(abs(p1.y-p2.y), 2)); 
}
void solve(){
    cin >> n >> l;
    for(int i = 0; i<n; i++){
        cin >> points[i].x >> points[i].y;    
    }
    sort(points, points + n);
    angle_sort as = {points[0]};
    sort(points+1, points + n, as);
    vector<coord> stk = {points[0], points[1]};
    for(int i = 2; i<n; i++){
        while(1){
            coord p2 = stk[stk.size()-1];
            coord p1 = stk[stk.size()-2];
            coord p3 = points[i];
            double cp = ccw(p1, p2, p3);
            if(cp > 0){
                stk.push_back(p3); 
                break;
            }
            else if(cp < 0){
                stk.pop_back();
            }
            else{
                stk.pop_back();
                stk.push_back(p3);
                break;
            }
        }
    }
    double ans = 0;
    for(int i = 0; i<stk.size(); i++){
        coord m = stk[i];
        coord right = stk[(i+1>=stk.size()) ? 0 : i+1];
        ans += dist(m, right);
    }
    ans += 2*M_PI*l;
    printf("%.0lf", round(ans));
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

건물의 좌표들로 convex hull에 해당하는 좌표를 구한 뒤, 다각형의 둘레와 문제에서 주어진 L값을 반지름으로 하는 원의 원주를 더해서 계산하면 된다.

방벽을 세울 때, 곡선에 해당하는 부분의 길이를 모두 더한 값이 2*PI*r 에 해당한다.

느낀점

곡선에 해당하는 부분의 길이를 모두 더했을때 2*PI*r이라는 사실을 모르고, 각 꼭지점에서 발생하는 호의 길이를 벡터의 내적을 이용해서 각도를 구한뒤, 직접 계산해서 모두 더하는 방법을 사용했다. 이렇게 하니깐 double 형 계산에 오차가 발생해서 틀렸습니다가 나왔다. 이후 검색을 통해 이러한 사실을 알았고, 맞았습니다를 받을 수 있었다.
