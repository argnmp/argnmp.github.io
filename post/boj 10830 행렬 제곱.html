코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
ll b;
struct matrix {
  int r, c;
  vector<vector<int>> data;
};
matrix init_matrix(int r, int c){
  matrix res;
  res.r = r;
  res.c = c;
  res.data = vector(res.r, vector(res.c, 0));
  return res;
}
void print(matrix target){
  for(int i = 0; i < target.r; i++){
    for(int j = 0; j < target.c; j++){
      cout << target.data[i][j] << ' ';
    }
    cout << '\n';
  }
}
matrix operator*(matrix& a, matrix& b){
  int n = a.c;
  matrix res = init_matrix(a.r, b.c);
  for(int i = 0; i < res.r; i++){
    for(int j = 0; j < res.c; j++){
      int s = 0;
      for(int k = 0; k < n; k++){
        s += (a.data[i][k] * b.data[k][j]);
      }
      res.data[i][j] = s % 1000;
    }
  }
  return res;
}


void solve(){
  cin >> n >> b; 
  matrix target = init_matrix(n, n);
  for(int i = 0; i<n; i++){
    for(int j = 0; j<n; j++){
      cin >> target.data[i][j];
    }
  }
  matrix remainder = init_matrix(n, n);
  for(int i = 0; i<n; i++){
    remainder.data[i][i] = 1;
  }

  while(b != 1){
    if(b % 2 == 0){
      target = target * target;
    } else {
      remainder = remainder * target;
      target = target * target;
    }
    b /= 2;
  }
  target = target * remainder;
  print(target);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 3을 8번 제곱한다고 했을 때, (((3^2)^2)^2) 로 3번의 곱셈으로 계산이 가능하다. 따라서 O(log2 n) 의 시간안에 계산이 가능하다. 이걸 이용해서 풀어야 한다.
  • b 횟수만큼 제곱을 해야한다고 했을 때, b를 2로 나눠주면서 줄여나가는 방식으로 해야 한다. 2로 나눠 줄 때마다 target 행렬을 제곱 해준다. 만약 b가 홀수인 경우 항등행렬인 remainder 행렬을 만들어 놓고 현재의 target 행렬을 곱해주면 된다. 나머지를 따로 저장하는 느낌. 마지막에 target 행렬과 remainder 행렬을 곱해주면 답이 나온다.
  • 행렬 연산은 다음과 같은 struct를 사용한다 행렬
