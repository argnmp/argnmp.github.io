#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
ll n;
ll mod = 1000000007;
struct mat{
    int r, c;
    vector<vector<ll>> data;
    mat operator*(const mat& t) const {
        mat res = {r, t.c}; 
        res.data = vector(r, vector<ll>(c, 0)); 
        for(int i = 0; i<res.r; i++){
            for(int j = 0; j<res.c; j++){
                ll acc = 0; 
                for(int k = 0; k<c; k++){
                    acc = (acc + (data[i][k] * t.data[k][j]) % mod) % mod;
                }
                res.data[i][j] = acc;
            }
        }
        return res;
    }
    void print(){
        for(int i = 0; i<r; i++){
            for(int j = 0; j<c; j++){
                cout << data[i][j] << ' ';
            }
            cout << endl;
            
        }
    }
};
void solve(){
    cin >> n;
    mat t = {2, 2};
    t.data = {
        {1, 1},
        {1, 0},
    };
    mat r = {2, 2};
    r.data = {
        {1, 0},
        {0, 1},
    };
    n -= 1;
    while(n>0){
        if(n%2==1){
            r = r * t;
        }
        t = t*t;
        n /= 2;
    }

    mat first = {2, 1};
    first.data = {
        {1},
        {0},
    };
    
    r = r * first;
    cout << r.data[0][0];

}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}
