#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, k;
int mod = 1000000007;
int value[1000000] = {0,};
ll segt[2097152] = {0,}; //2^21
ll segt_0[2097152] = {0,};
void segt_init(int l, int r, int node){
    if(l==r){
        if(value[l]==0){
            segt_0[node] = 1;
            segt[node] = 1;
        }
        else {
            segt[node] = value[l];
        }
        return;    
    }
    int mid = (l+r)/2;
    segt_init(l, mid, node*2);
    segt_init(mid+1, r, node*2+1);
    segt_0[node] = segt_0[node*2] + segt_0[node*2+1];
    segt[node] = (segt[node*2] * segt[node*2+1]) % mod;
}
void segt_update(int l, int r, int node, int t, int v){
    if(l==r){
        value[t] = v;
        if(value[t]==0){
            segt_0[node] = 1;
            segt[node] = 1;
        }
        else {
            segt_0[node] = 0;
            segt[node] = value[t];
        }
        return;
    }
    
    int mid = (l+r)/2;
    if(t <= mid){
        segt_update(l, mid, node*2, t, v);
    } 
    else {
        segt_update(mid+1, r, node*2+1, t, v);
    }
    segt_0[node] = segt_0[node*2] + segt_0[node*2+1];
    segt[node] = (segt[node*2] * segt[node*2+1]) % mod;
}
ll calc(int l, int r, int node, int lt, int rt){
    if(l==lt&&r==rt){
        if(segt_0[node]>=1)
            return 0;
        else
            return segt[node];
    }
    int mid = (l+r)/2;
    if(rt <= mid)
        return calc(l, mid, node*2, lt, rt);       
    else if(mid+1 <= lt)
        return calc(mid+1, r, node*2+1, lt, rt);
    else
        return calc(l, mid, node*2, lt, mid) * calc(mid+1, r, node*2+1, mid+1, rt) % mod;
}
void solve(){
    cin >> n >> m >> k;
    for(int i = 0; i<n; i++)
        cin >> value[i];
    segt_init(0, n-1, 1);
    int a, b, c;
    for(int i = 0; i<m+k; i++){
        cin >> a >> b >> c; 
        if(a==1){
            segt_update(0, n-1, 1, b-1, c);    
        }
        else if(a==2){
            cout << calc(0, n-1, 1, b-1, c-1) << '\n'; 
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

boj 2042 구간 합 구하기 에서 처럼, segment tree를 활용한다. 다만 합 대신 곱을 한 뒤 문제에서 주어진 값으로 나눈 나머지 값을 저장한다.

그렇기 때문에 위 문제의 풀이 코드의 갱신 부분을 그대로 사용할 수 없다. boj 2042 구간 합 구하기 에서는, 루트 노드에서 부터 리프 노드까지 내려가며, 갱신되는 인덱스가 포함된 구간에 해당하는 모든 노드에서, 원래 값을 빼주고 새로운 값을 더해주는 방식을 사용하였다.

하지만 구간의 곱을 구하는 segment tree의 노드는 나머지 연산을 한 결과 값을 가지고 있기 때문에, 원래 값으로 나누어주고 새로운 값을 곱해주는 방식을 사용할 수 없다. 더군다나 0으로 나누는 경우 divide by zero exception이 발생할 것이다.

따라서 여기서는 거꾸로 리프 노드 부터 올라간다. 리프 노트 부터 올라가면서, segt_init 에서 했던 것 처럼 구간 곱 정보를 담고 있는 segment tree 노드들의 값을 다시 계산해준다.

위의 코드에서는 다른 방식을 시도해보느라 segment tree의 노드에 해당하는 구간에 들어 있는 0의 개수를 저장하는 등 불필요한 부분이 존재한다. 앞서 설명한 방식만 사용하여도 문제가 풀릴 것으로 예상한다.
