문제

https://www.acmicpc.net/problem/2213

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
int weight[10001] = {0,};
vector<vector<int>> graph(10001, vector<int>());

/*
 * type: 0 -> return highest value when root node is not selected
 * type: 1 -> return highest value when root node is selected
 * type: -1 -> return highest value by comparing type 0 and type 1
 */

int sub_type0[10001] = {0,};
int sub_type1[10001] = {0,};
int sub_type_selection[10001] = {0,};
int calc(int node, int parent, int type){
    if(sub_type0[node]==-1){
        int total = 0;
        for(int next: graph[node]){
            if(next==parent)
                continue;
            total += calc(next, node, -1);
        }
        sub_type0[node] = total;
    } 
    if(sub_type1[node]==-1){
        int total = 0; 
        for(int next: graph[node]){
            if(next==parent)
                continue;
            total += calc(next, node, 0);
        }
        sub_type1[node] = total+weight[node];
    }
    
    if(type==0)
        return sub_type0[node];
    else if(type==1)
        return sub_type1[node];
    else {
        if(sub_type0[node] < sub_type1[node]){
            sub_type_selection[node] = 1;
            return sub_type1[node];
        }
        else {
            sub_type_selection[node] = 0;
            return sub_type0[node];
        }
    }
}

vector<int> included_nodes;
void find(int node, int parent, int type){
    if(type==0){
        for(int next: graph[node]){
            if(next==parent)
                continue;
            find(next, node, -1);
        }
    } else if(type==-1){
        if(sub_type_selection[node]==1){
            included_nodes.push_back(node);
            for(int next: graph[node]){
                if(next==parent)
                    continue;
                find(next, node, 0);
            }
        } else {
            for(int next: graph[node]){
                if(next==parent)
                    continue;
                find(next, node, -1);
            }
        }
    }
}

struct box{
    int node;
    int parent;
    int skip_or_check; //skip: 0, check: 1
};
void solve(){
    memset(sub_type0, -1, sizeof(sub_type0));
    memset(sub_type1, -1, sizeof(sub_type1));
    memset(sub_type_selection, -1, sizeof(sub_type_selection));
    cin >> n;
    for(int i = 1; i<=n; i++)
        cin >> weight[i];
    int u, v;
    for(int i = 0; i<n-1; i++){
        cin >> u >> v; 
        graph[u].push_back(v);
        graph[v].push_back(u);
    }
    cout << calc(1, 1, -1) << '\n';
    find(1, 1, -1);
    sort(included_nodes.begin(), included_nodes.end());
    for(int node: included_nodes){
        cout << node << ' '; 
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

일단 트리 라는 것을 제외한 다른 그래프에 관한 정보가 없기 때문에, 어떤 vertex든 루트노드가 될 수 있다. 따라서 아무 vertex나 루트 노드로 정해서 탐색을 시작하면 된다.

루트 노드에서 부터 서브 트리를 탐색을 하는 과정에서, 다음과 같은 경우의 수로 나눌 수 있다.

    1. (서브 트리의) 루트 노드를 포함하는 경우 가장 큰 가중치의 합
    1. (서브 트리의) 루트 노드를 포함하지 않는 경우 가장 큰 가중치의 합 탐색을 하는 과정에서 이 두가지의 값을 구해주어야 하고, 이를 따로 저장해 놓는다. 첫번째 경우의 수에 대한 값을 sub_type0 에 저장하고, 두번째 경우의 수에 대한 값을 sub_type1에 저장한다.

루트 노드로 부터 dfs를 돌기 시작한다. 재귀적으로 탐색하는 과정에서 세가지 정보를 하위 서브트리에 대하여 요청할 수 있다. 나는 이것을 type 으로 구분했다.

  • type == 0: 현재 서브 트리의 루트 노드를 선택하지 않았을 때 가장 큰 값을 반환
  • type == 1: 현재 서브 트리의 루트 노드를 선택했을 때 가장 큰 값을 반환
  • type == -1: 현재 서브 트리의 루트 노드를 선택했을 때와 선택하지 않았을 때를 비교해서 더 큰 값을 반환

dfs를 통해 특정 노드에 방문했을 때,위에서 나눈 경우의 수, sub_type0과 sub_type1에 해당하는 값들을 계산해주어야 한다.

sub_type0 의 경우에는, 값을 구하기 위하여 루트 노드와 연결된 서브 트리의 루트 노드에 대해서 type==-1로 재귀 호출을 해야 한다. 현재 서브트리의 루트노드를 고르지 않았기 때문에, 연결된 노드들에 대해서 고를 수도 있고 고르지 않을 수 도 있다. 문제에서 가장 큰 가중치의 값을 구해야 한다고 했기 때문에 이 두가지의 값을 모두 고려해주어야 한다.

sub_type1의 경우에는, 값을 구하기 위하여 루트 노드와 연결된 서브 트리의 루트 노드에 대해서 type==0으로 재귀 호출을 해야 한다. 왜냐하면 현재 서브 트리의 루트 노드를 골랐기 때문에, 인접한 노드를 고를 수 없기 때문이다.

이 두가지의 값을 구하고 나면, 위에서 정의한 모든 type에 맞는 값을 리턴할 수 있다. type==-1의 경우 현재 서브 트리의 루트 노드를 선택했는지 선택하지 않았는지를 (sub_type_selection에) 기록해준다. 이후 다시 루트노드 부터 dfs를 돌면서 이 값을 확인하고, 독립 집합에 포함되는 노드를 골라낼 수 있다.

여기까지 생각해보았을 때, 사실 type == 1 에 대해서는 사용하지 않기 때문에 정의해줄 필요가 없다.

마지막으로 find함수가 하는 것 처럼, 독립 집합에 속하는 노드들을 가져온다. calc에서 한 것과 비슷하게 하면 된다. type==-1인 경우 다음과 같이 탐색한다

  • sub_type_selection 이 1인 경우, 현재 서브트리의 루트 노드를 고른 것이기 때문에 독립 집합에 넣어주고, 연결된 노드에 대하여 type==0 으로 재귀 호출한다.
  • sub_type_selection 이 0인 경우, 현재 서브트리의 루트 노드를 고르지 않은 것이기 때문에 독립 집합에 넣지 않고, 연결된 노드에 대하여 type==-1 으로 재귀 호출 한다.

type==0 인 경우 다음과 같이 탐색한다. sub_type_selection은 상관이 없다. 연결된 노드에 대하여 type==-1으로 재귀 호출한다.

느낀점

나는 dp라는 생각을 하지 않고 서브 트리에 대한 정보만 기록해주면 된다고 생각하며 풀었는데, dp에 해당하는 문제였다. 처음 부터 dp라고 문제를 구별하는 것이 좋을 듯 하다.

또한 푸는 과정에서 type==1과 같이 불필요한 부분에 대한 고려를 했다. 이런 것도 푸는 도중에 필요 없는 경우의 수라고 생각하기 보다는 처음부터 제외를 해버리고 풀도록 노력해야겠다.
