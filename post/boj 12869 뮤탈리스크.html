풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
#define dbgn(x) printf("[DBG] %s: %lld\n", #x, (ll)x);
using namespace std;
using ll = long long;

int n;
struct carrier{
    int v[3] = {0,};
    bool check(int n){
        for(int i = 0; i<n; i++)
            if(v[i]>0)
                return false;
        return true;
    }
    void sortv(){
        sort(v, v+3, greater<int>());
    }
    void norm(){
        for(int i = 0; i<3; i++)
            if(v[i]<0)
                v[i]=0;
    }
};
struct box {
    carrier c;
    int count;
};
queue<box> q;

int attack[3] = {9, 3, 1};
int visited[3] = {0,};
int dp[60][60][60] = {0,};
void perm(int sel, carrier c, int count){
    if(sel==n){
        c.sortv();
        c.norm();
        if(dp[c.v[0]][c.v[1]][c.v[2]]==0){
            dp[c.v[0]][c.v[1]][c.v[2]] = 1;
            q.push({c, count + 1});
        }
        return;
    }
    for(int i = 0; i<n; i++){
        if(visited[i]==0){
            carrier next = c;
            visited[i] = 1;
            next.v[i] -= attack[sel];
            perm(sel+1, next, count);
            visited[i] = 0;
        }
    }
}

void solve(){
    cin >> n;
    carrier t;
    for(int i=0; i<n; i++){
        cin >> t.v[i];
    }
    t.sortv();
    t.norm();
    q.push({t, 0});
    while(!q.empty()){
        box cur = q.front();
        q.pop();
        if(cur.c.check(n)){
            cout << cur.count;
            return;
        }
        perm(0, cur.c, cur.count);
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

나는 bfs로 모든 경우의 수를 탐색하였다. 그 과정에서 중복방문이 생기는 것을 고려하여, 남아있는 SCV들의 체력이 큰 순서대로 정렬하여 dp배열에서 방문하였는지를 확인해주었다.

느낀점

꽤 복잡하게 푼 것 같다. 처음에는 남은 체력이 많은 SCV의 체력을 더 깎는 방식을 사용했었는데 아니었다. 모든 경우의 수를 탐색해주어야 했다.
