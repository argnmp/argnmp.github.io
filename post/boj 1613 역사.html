첫번째 풀이: topological sort

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
#define dbgn(x) printf("[DBG] %s: %lld\n", #x, (ll)x);
#define dbgarr(r, c, x) printf("[DBG] %s\n", #x);for(int i=0; i<r; i++){for(int j=0; j<c; j++){printf("%2lld ", (ll)x[i][j]);}printf("\n");}
using namespace std;
using ll = long long;
int n, k, s;
vector<vector<int>> graph(401, vector<int>());
int in_degree[401] = {0,};
int in_degree_node[401][401] = {0,};
void solve(){
    cin >> n >> k;
    int u, v;
    for(int i = 0; i<k; i++){
        cin >> u >> v;
        graph[u].push_back(v);
        in_degree[v] += 1;
    }
    queue<int> q;
    for(int i = 1; i<=n; i++)
        if(in_degree[i]==0)
            q.push(i);

    while(!q.empty()){
        int cur = q.front();
        q.pop();
        for(int next: graph[cur]){
            in_degree_node[next][cur] = 1;
            for(int i = 1; i<=n; i++){
                if(in_degree_node[cur][i]==1)
                    in_degree_node[next][i] = 1;
            }
            in_degree[next] -= 1;
            if(in_degree[next]==0)
                q.push(next);
        }
    }
    cin >> s;
    for(int i = 0; i<s; i++){
        cin >> u >> v;
        if(in_degree_node[u][v]==1)
            cout << 1 << '\n';
        else if(in_degree_node[v][u]==1)
            cout << -1 << '\n';
        else
            cout << 0 << '\n';
            
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

topological sort를 통해서, 먼저 발생한 사건을 모두 표시해준다. 이 방식에서 특정 노드가 pop되는 시점은, 그 노드의 먼저 발생한 사건을 모두 찾았다는 의미이므로, 다음 노드에 pop된 노드의 먼저 발생한 사건을 모두 표시해주도록 한다.

노드가 최대 400개이므로 가능한 풀이이다.

두번째 풀이: floyd-warshall

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
#define dbgn(x) printf("[DBG] %s: %lld\n", #x, (ll)x);
#define dbgarr(r, c, x) printf("[DBG] %s\n", #x);for(int i=0; i<r; i++){for(int j=0; j<c; j++){printf("%2lld ", (ll)x[i][j]);}printf("\n");}
using namespace std;
using ll = long long;
int n, k, s;
void solve(){
    cin >> n >> k;
    int dist[401][401] = {0,};
    fill(&dist[0][0], &dist[400][401], INT_MAX);
    
    int u, v;
    for(int i = 0; i<k; i++){
        cin >> u >> v;
        dist[u][v] = 1;
    }
    for(int k = 1; k<=n; k++){
        for(int i = 1; i<=n; i++){
            for(int j = 1; j<=n; j++){
                if(dist[i][k]!=INT_MAX && dist[k][j]!=INT_MAX){
                    if(dist[i][k] + dist[k][j] < dist[i][j]){
                        dist[i][j] = dist[i][k] + dist[k][j];
                    }
                }
            }
        }
    }
    cin >> s;
    for(int i = 0; i<s; i++){
        cin >> u >> v;
        if(dist[u][v]!=INT_MAX)
            cout << -1 << '\n';
        else if(dist[v][u]!=INT_MAX)
            cout << 1 << '\n';
        else
            cout << 0 << '\n';
    }
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

floyd-warshall 알고리즘을 이용해서, 최단거리를 모두 찾아주면 된다.

느낀점

검색을 통해 floyd-warshall로도 풀 수 있다고 해서 다시 풀어보았지만, 틀렸습니다가 나왔다. floyd-warshall 알고리즘을 잘못 사용했다. 중간 노드가 loop의 가장 바깥쪽으로 나와야 한다.
