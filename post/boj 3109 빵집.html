#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int r, c;
char board[10000][500] = {0,};
int dx[3] = {-1, 0, 1};
int dy[3] = {1, 1, 1};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){ return 0<=x&&x<r&&0<=y&&y<c; }
};
bool dfs(pos s){
    board[s.x][s.y] = 'x';
    if(s.y==c-1){
        return true;
    }
    for(int i = 0; i<3; i++){
        pos next = s.adj(i);
        if(next.safe() && board[next.x][next.y]=='.'){
            bool res = dfs(next);
            if(res){
                return true;
            }
        }
    }
    return false;
}
void solve(){
    cin >> r >> c;
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            char ch = cin.get();
            while(ch=='\n'||ch=='\r'||ch==' ')
                ch =cin.get();
            board[i][j] = ch;
        }
    }
    int ans = 0;
    for(int i = 0; i<r; i++){
        if(dfs({i, 0})){
            ans += 1;
        }
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

첫번째 열의 맨 윗쪽 행 부터 시작하여, 가능한 마지막 열의 위쪽 행에 연결하는 것이 유리히다. 따라서 위의 행 부터 탐색을 시작하고, dfs로 방문하며 위 대각선, 오른쪽, 아래 대각선 순서대로 방문을 하도록 한다.

탐색을 하는 과정에서 마지막 열에 대한 연결의 가능여부와 관계없이 방문한 좌표를 무조건 표시해둔다. 그리고 다음 탐색부터는 표시된 곳을 방문하지 않는다. 한번 파이프가 연결된 좌표는 또 연결할 수 없고, 해당 좌표에서 연결이 불가능한 경우 이는 다른 행의 탐색에도 똑같이 적용되기 때문이다. 이렇게 해야 시간초과가 발생하지 않는다.

느낀점

그리디 방식이 합쳐진 것 같다.
