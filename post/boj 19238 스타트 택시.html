풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, fuel;
int board[20][20];
int dx[4] = {0,0,-1,1};
int dy[4] = {-1,1,0,0};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }   
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<n; }
    void print(){
        printf("[%d, %d]", x, y);
    }
};
struct target{
    int done;
    pos from;
    pos to;
    int id;
    bool operator<(const target& t) const{
        if(from.x==t.from.x){
            return from.y < t.from.y;
        }
        return from.x < t.from.x;
    }
    void print(){
        printf("id %d | ", id);
        printf("from: ");
        from.print();
        printf(", to: ");
        to.print();
        printf(" -> done: %d\n", done);
    }
};
target passenger[401] = {0,};
int finished = 0;

int check_dist(pos from, pos to){
    int visited[20][20] = {0};
    memset(visited, -1, sizeof(visited));
    queue<pos> q;
    q.push(from);
    visited[from.x][from.y] = 0;
    while(!q.empty()){
        pos cur = q.front();
        q.pop();
        if(cur.x==to.x&&cur.y==to.y){
            return visited[cur.x][cur.y];
        }
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i);
            if(next.safe() && visited[next.x][next.y]==-1 && board[next.x][next.y]!=-1){
                visited[next.x][next.y] = visited[cur.x][cur.y] + 1;
                q.push(next);
            }
        }
    }
    // destination unreachable
    return -1;
}

void solve(){
    cin >> n >> m >> fuel;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            cin >> board[i][j];
            if(board[i][j]==1)
                board[i][j] = -1;
        }
    }
    pos current;
    cin >> current.x >> current.y;
    current = {current.x-1, current.y-1};
    pos from, to;
    for(int i = 1; i<=m; i++){
        cin >> from.x >> from.y >> to.x >> to.y; 
        from = {from.x-1, from.y-1};
        to = {to.x-1, to.y-1};
        passenger[i] = {0, from, to, i};
        board[from.x][from.y] = i;
    }

    /* // check
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            printf("%2d ", board[i][j]);
        }
        cout << '\n';
    } */
    while(finished != m){
        int visited[20][20] = {0,};

        // find nearest passenger 
        memset(visited, -1, sizeof(visited));
        queue<pos> q;
        q.push(current);
        visited[current.x][current.y] = 0;
        int nearest_dist = INT_MAX;
        vector<target> cand;

        if(board[current.x][current.y]>0 && passenger[board[current.x][current.y]].done == 0){
            passenger[board[current.x][current.y]].done = 1;
            cand.push_back(passenger[board[current.x][current.y]]);
            nearest_dist = 0;
        }
        
        while(!q.empty()){
            pos cur = q.front();
            q.pop();
            if(visited[cur.x][cur.y] >= nearest_dist){
                break;
            }
            for(int i = 0; i<4; i++){
                pos next = cur.adj(i);     
                if(next.safe() && visited[next.x][next.y]==-1 && board[next.x][next.y] != -1){
                    visited[next.x][next.y] = visited[cur.x][cur.y] + 1;
                    if(board[next.x][next.y] > 0 && passenger[board[next.x][next.y]].done == 0){
                        // passenger[board[current.x][current.y]].print();
                        cand.push_back(passenger[board[next.x][next.y]]); 
                        nearest_dist = visited[next.x][next.y];
                    }
                    q.push(next);
                }
            }
        }
        sort(cand.begin(), cand.end());
        // case: all passenger are not reachable
        if(cand.size()==0){
            cout << -1;
            return;
        }
        // case; fuel is not enough
        target np = cand[0];
        // np.print();
        int fuel_usage = visited[np.from.x][np.from.y];
        fuel -= fuel_usage;
        if(fuel < 0){
            cout << -1;
            return;
        }

        int from_to_dist = check_dist(np.from, np.to);
        // from to unreachable
        if(from_to_dist==-1){
            cout << -1;
            return;
        }
        fuel -= from_to_dist;
        if(fuel < 0){
            cout << -1;
            return;
        }
        fuel += from_to_dist*2;

        finished += 1;
        passenger[np.id].done = 1;
        current = np.to;
    }
    cout << fuel;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}
