문제

https://www.acmicpc.net/problem/16933

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m, k;
char board[1000][1000] = {0,};
int dx[4] = {0,-1,1,0};
int dy[4] = {-1,0,0,1};
struct pos{
    int x, y;
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<m; }
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; } 
    void print(){
        printf("[%d, %d]\n", x, y);
    }
};
struct box{
    pos p;
    int cost;
    int able;
};
int done[1000][1000][11] = {0,}; // 0 for day, 1 for night
void solve(){
    cin >> n >> m >> k;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            char ch = cin.get();
            while(ch=='\n' || ch=='\r' || ch==' '){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    } 
    
    queue<box> q;
    q.push({{0,0}, 0, k});
    done[0][0][k] = 1; 

    while(!q.empty()){
        box cur = q.front(); 
        q.pop();
        if(cur.p.x == n-1 && cur.p.y == m-1){
            cout << cur.cost+1;
            return;
        }
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe()){
                if(board[next.x][next.y]=='0' && done[next.x][next.y][cur.able] == 0){
                    done[next.x][next.y][cur.able] = 1;
                    q.push({next, cur.cost+1, cur.able});
                }
                else if(board[next.x][next.y]=='1' && cur.able > 0 && done[next.x][next.y][cur.able-1]==0){
                    if(cur.cost % 2 == 0){
                        done[next.x][next.y][cur.able-1] = 1;
                        q.push({next, cur.cost+1, cur.able-1});
                    }
                    else {
                        q.push({cur.p, cur.cost+1, cur.able});
                    }
                }
            }
        }
    } 
    cout << -1;
    return;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

[0,0] -> [n-1, m-1] 의 좌표로 이동을 하는 문제이고, 거리가 모두 같기 때문에 bfs로 최단거리를 구할 수 있다.

벽은 낮에만 부술 수 있고, 좌표를 이동할 때마다 낮과 밤이 바뀐다. 따라서 bfs에서 현재까지의 거리를 cost라고 할 때, cost가 짝수면 낮(시작할 때 낮이므로), 홀수면 밤으로 처리해주면 된다.

bfs에서는 재방문을 방지하기 위해서, 방문한 좌표에 대한 데이터를 저장해놓는다. 여기에서는 추가로 벽을 부술 수 있는 개수 를 저장해주어야 한다. 같은 좌표에 재방문을 하더라도, 벽을 부술 수 있는 개수에 따라 이후의 방문 가능한 상태가 변하기 때문이다.

또한 bfs에서 탐색을 하는 과정에서, 벽이 있고 현재 상태가 밤이라면 (cost가 홀수이면), 현재 좌표를 다시 queue에 넣어준다.

벽을 부술 수 있는 개수 외에도 낮과 밤의 상태를 추가로 저장할 수는 있지만, 굳이 저장할 필요가 없다. 다음과 같은 상황을 가정해보자.

  • [a,b]위치에 4개를 추가로 부술 수 있는 상태에서 밤에 방문을 했다고 해보자. visited 배열에 의하여 [a,b] 좌표에는 4개를 추가로 부술 수 있는 상태에서 낮에 다시 방문을 할 수 없다. 낮에 다시 방문을 할 필요가 없다. 왜냐하면 밤에 방문을 했을 때 벽이 있어서 이동을 할 수 없는 경우 queue에 현재 좌표를 다시 넣는다. 따라서 자동으로 낮에 방문을 한 것과 같게 된다.
  • [a,b] 위치에 4개를 추가로 부술 수 있는 상태에서 낮에 방문을 했다고 해보자. visited 배열에 의하여 [a,b] 좌표에는 4개를 추가로 부술 수 있는 상태에서 밤에 다시 방문을 할 수 없다. 밤에 다시 방문을 할 필요가 없다. 낮에는 벽도 부술 수 있기 때문에 밤보다 좋고, 먼저 낮에 방문했다는 것은 그다음 밤에 방문하는 것보다 더 최단 경로여서 구태여 밤에 다시 방문할 필요가 없다.

느낀점

풀어본 것 같은 느낌이 드는 문제이다. 중복되는 경우의 수를 제거하거나 재방문을 해야하는 경우의 수를 중심으로 생각해보아야 한다. 또한 문제의 특성을 파악하여, 위의 문제의 경우 bfs는 먼저 방문하는 것이 최단거리이므로 현재까지의 최단거리를 따로 저장할 필요가 없다. 그렇기에 부술 수 있는 개수에 대해서만 고민하면 된다.
