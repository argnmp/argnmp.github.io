문제

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
struct edge {
    int v, w; 
};
vector<vector<edge>> graph(100001, vector<edge>());
struct box {
    ll node, cost;
};
box par[100001][20] = {0,};
int dep[100001] = {0,};
int visited[100001] = {0,};

void pre_lca(int node, int depth){
    visited[node] = 1; 
    dep[node] = depth;
    for(edge next: graph[node]){
        if(visited[next.v]==1)
            continue;
        par[next.v][0] = {node, next.w};
        pre_lca(next.v, depth + 1);
    }    
}
void fill_lca(){
    for(int i=1; i<=19; i++){
        for(int j=1; j<=n; j++){
            box target = par[par[j][i-1].node][i-1];
            ll node = target.node;
            ll cost = par[j][i-1].cost + target.cost;
            par[j][i] = {node, cost};
        }
    }
}
ll route_cost(int a, int b){
    if(dep[a] > dep[b])
        swap(a,b);

    ll total_cost = 0;
    for(int i = 19; i>=0; i--){
        if(dep[b]-dep[a] >= (1<<i)){
            total_cost += par[b][i].cost;
            b = par[b][i].node;
        }
    }
    if(a==b)
        return total_cost;
    
    for(int i = 19; i>=0; i--){
        if(par[a][i].node!=par[b][i].node){
            total_cost += par[a][i].cost;
            total_cost += par[b][i].cost;
            a = par[a][i].node;
            b = par[b][i].node;
        }
    }
    return total_cost + par[a][0].cost + par[b][0].cost;
}
int search_k_node(int from, int k){
    for(int i = 19; i>=0; i--){
        if(k >= (1<<i)){
            from = par[from][i].node;
            k-=(1<<i);
        }
    }
    return from;
}
int node_k(int a, int b, int k){
    int org_a = a;
    int org_b = b;
    bool is_swapped = false;
    if(dep[a] > dep[b]){
        is_swapped = true;
        swap(a, b);
    }
    int a_depth = 0;
    int b_depth = 0;
    for(int i = 19; i>=0; i--){
        if(dep[b]-dep[a] >= (1<<i)){
            b_depth += (1<<i);
            b = par[b][i].node;
        }
    }

    if(a==b){
        if(is_swapped)
            return search_k_node(org_a, k);
        else
            return search_k_node(org_b, b_depth - k);
    }
    
    for(int i = 19; i>=0; i--){
        if(par[a][i].node!=par[b][i].node){
            a = par[a][i].node;
            a_depth += (1<<i);
            b = par[b][i].node;
            b_depth += (1<<i);
        }
    }
    a = par[a][0].node;
    a_depth += 1;
    b = par[b][0].node;
    b_depth += 1;

    if(is_swapped){
        if(k > b_depth)
            return search_k_node(org_b, a_depth-(k-b_depth));
        else
            return search_k_node(org_a, k);
    }
    else {
        if(k > a_depth)
            return search_k_node(org_b, b_depth-(k-a_depth));
        else
            return search_k_node(org_a, k);
    }
}
void solve(){
    cin >> n;
    int u, v, w; 
    for(int i = 0; i<n-1; i++){
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }
    pre_lca(1, 0);
    fill_lca();
    cin >> m;
    int com;

    for(int i = 0; i<m; i++){
        cin >> com;
        if(com==1){
            cin >> u >> v; 
            cout << route_cost(u, v) << '\n';
        } 
        else{
            cin >> u >> v >> w;
            cout << node_k(u, v, w-1) << '\n';
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

문제에서 n개의 정점으로 이루어진 트리(무방향 사이클이 없는 연결 그래프) 이 주어진다. 이 그래프에서 다음의 두가지 쿼리를 수행할 수 있어야 한다

  • 1 u v: u에서 v로 가는 경로의 비용을 출력한다.
  • 2 u v k: u에서 v로 가는 경로에 존재하는 정점 중에서 k번째 정점을 출력한다. k는 u에서 v로 가는 경로에 포함된 정점의 수보다 작거나 같다.

트리에서의 한 vertex에서 다른 vertex까지의 경로는, 같은 vertex를 다시 방문하지 않는 경우 최단 경로가 된다. 따라서 문제에서 주어진 경로를 이 최단 경로로 유추할 수 있다.

두 vertex의 최단 경로는 가장 가까운 공통 조상을 무조건 거치게 된다. 따라서 lca 알고리즘을 활용하여야 한다. 그리고 루트 노드는 트리 구조이기 때문에 어떤 노드로 설정하든 상관없다.

먼저 첫번째 쿼리는 LCA 알고리즘의 DP 배열을 설정하는 과정에서 cost에 대한 추가 정보를 저장하는 방식으로 해결이 가능하다.

LCA 알고리즘은 lgN의 시간복잡도를 위해서, DP를 활용하여 특정 vertex의 2^k 에 해당하는 부모 노드 정보를 미리 저장해둔다. 이와 같은 방식으로 특정 vertex의 2^k에 해당하는 부모 노드까지의 cost 정보를 미리 저장해둔다.

이후 LCA 알고리즘을 통해 공통 부모 노드를 찾아가며, 그 과정에서 점프하게 되는, 즉2^k에 해당하는 부모 노드까지의 cost를 계속 더해주며 구할 수 있다.

이 방식은 위 코드의 route_cost 함수에 구현되어있다.

그리고 두번째 쿼리는 LCA 알고리즘의 depth 정보를 사용하여 해결할 수 있다.

       1
      / \
	 2   3
    / \   \
   4   5   6

위의 트리에서 노드 4 부터 노드 6까지의 경로에서 4번째 노드, 즉 2 4 6 4 로 쿼리가 주어졌다고 해보자.

먼저 공통 부모 노드로 부터 각 노드까지의 depth를 구한다. 공통 부모 노드 1 부터 노드 4까지의 depth는 2이고, 공통 부모 노드 1 부터 노드 6까지의 depth는 2가 된다.

4 -> 6 방향에서 4번째 노드 이므로, 노드 6으로 부터 2(공통 부모 노드에서 노드 6까지의 depth) - (4(k) - (2(공통 부모에서 노드 4까지의 depth) + 1)) 번째 부모 노드가 쿼리의 결과 값이 된다.

이러한 방식으로 각 노드의 depth 정보를 활용하여 쿼리의 값을 계산할 수 있다.

이 방식은 위 코드의 node_k 함수에 구현되어 있다.
