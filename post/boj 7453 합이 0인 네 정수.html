문제

https://www.acmicpc.net/problem/7453

풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
int A[4000];
int B[4000];
int C[4000];
int D[4000];

void solve(){
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> A[i] >> B[i] >> C[i] >> D[i];
    }
    vector<int> AB; 
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            AB.push_back(A[i]+B[j]);
        }
    }
    sort(AB.begin(), AB.end()); 
    vector<int> AB_opt;
    vector<ll> AB_opt_size;
    AB_opt.push_back(AB[0]);
    AB_opt_size.push_back(1);
    for(int i = 1; i<AB.size(); i++){
        if(AB_opt.back() == AB[i]){
            AB_opt_size[AB_opt.size()-1] += 1;
        }
        else {
            AB_opt.push_back(AB[i]);
            AB_opt_size.push_back(1);
        }
    }
    vector<int> CD;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            CD.push_back(C[i]+D[j]);
        }
    }
    sort(CD.begin(), CD.end()); 
    vector<int> CD_opt;
    vector<ll> CD_opt_size;
    CD_opt.push_back(CD[0]);
    CD_opt_size.push_back(1);
    for(int i = 1; i<CD.size(); i++){
        if(CD_opt.back() == CD[i]){
            CD_opt_size[CD_opt.size()-1] += 1;
        }
        else {
            CD_opt.push_back(CD[i]);
            CD_opt_size.push_back(1);
        }
    }
    
    ll ans = 0;
    for(int i = 0; i<AB_opt.size(); i++){
        int idx = lower_bound(CD_opt.begin(), CD_opt.end(), -AB_opt[i]) - CD_opt.begin();
        if(idx == CD_opt.size() || CD_opt[idx] != -AB_opt[i])
            continue;
        // printf("%d ~ %d\n", idx, uidx);
        ans += CD_opt_size[idx] * AB_opt_size[i];
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

A와 B배열의 모든 쌍의 합을 AB에 저장해둔다. AB를 정렬한 뒤, 중복된 원소를 제거한 AB_opt배열을 만들고 AB_opt_size에는 해당 원소의 개수를 저장한다.

C와 D배열에 대해서도 같은 방법을 적용한다.

AB_opt배열을 앞에서부터 탐색하면서, CD_opt배열에서 이분 탐색으로 0을 만들 수 있는 원소를 찾는다. 각 index에 있는 AB_opt_size와 CD_opt_size 의 값의 곱을 모두 더해주면 된다.

중복된 원소를 제거하지 않고 정렬된 배열 그대로 사용하기 위해서는 upper_bound값에서 lower_bound값을 빼서 원소의 개수를 구할 수 있다. 시간은 더 느리지만 메모리는 더 적게 쓴다.

위 풀이는 boj에서 3865ms가 소요되었다.

느낀점

투 포인터를 이용할 수도 있다. 나중에 이 방식으로 다시 풀어봐야겠다. 투 포인터 풀이가 더 빠른 것 같아 보인다.
