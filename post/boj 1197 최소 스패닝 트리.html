#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int v, e;
struct edge {
    int v;
    int w;
    void print() {
        cout << '[' << v << ',' << w << ']' << '\n';
    }
};
vector<vector<edge>> graph(10001, vector<edge>());
struct compare{
    bool operator()(edge& a, edge& b){
        return a.w > b.w;
    }
};

void solve(){
    cin >> v >> e;
    int u, v, w;
    for(int i=0; i<e; i++){
        cin >> u >> v >> w; 
        graph[u].push_back({v,w});
        graph[v].push_back({u,w});
    }
    int done[10001] = {0,};
    priority_queue<edge, vector<edge>, compare> pq;
    pq.push({1, 0});
    ll weight_sum = 0;
    while(!pq.empty()){
        edge e = pq.top();
        pq.pop();
        if(done[e.v] == 1) continue;
        done[e.v] = 1;
        weight_sum += e.w;
        for(edge n: graph[e.v]){
            if(done[n.v]==0)
                pq.push(n);
        }
    }
    cout << weight_sum;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • edge라는 struct를 정의했다. edge는 도착 vertex와 그곳 까지의 가중치 정보를 가지고 있는다.
    • 그래프
  • 이 edge를 struct 자체를 priority queue에 넣어서 mst를 구하도록 설계되었다.
    • compare을 정의하여 pq에 넘겨준다.
    • 2. Areas/blog/programming languages/cpp/priority queue
  • prim's algorithm
  • done이라는 배열은 해당 vertex에 방문한지 여부를 나타낸다
    • pq에서 pop했을 때 done으로 확인한다. done이 1인 경우는 이미 mst에 포함된 vertex임을 나타낸다.
    • 연결된 edge를 pq에 넣는 과정에서 done을 확인하여 pq에 불필요한 edge가 들어가지 않도록 한다.
