풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
#define idx(node) node < 0 ? -node + 10000 : node
using namespace std;
using ll = long long;

int n, m;
vector<vector<int>> graph(20001, vector<int>());

int id=1, graph_id[20001] = {0,};
int cid=1, graph_cid[20001] = {0,};
vector<int> stk;
int scc(int node){
    graph_id[node] = id;
    id += 1;
    int parent = graph_id[node];
    stk.push_back(node);
    for(int next: graph[node]){
        if(graph_id[next]==0){
            parent = min(parent, scc(next));
        }
        else if(graph_cid[next]==0)
            parent = min(parent, graph_id[next]);
    }
    if(parent == graph_id[node]){
        while(1){
            int cur = stk.back();
            stk.pop_back();
            graph_cid[cur] = cid;
            if(cur == node)
                break;
        }
        cid += 1;
    }
    return parent;
}
void solve(){
    cin >> n >> m; 
    int a, b;
    for(int i = 0; i<m; i++){
        cin >> a >> b;
        graph[idx(-a)].push_back(idx(b));
        graph[idx(-b)].push_back(idx(a));
    }
    for(int i = 1; i<=n; i++){
        if(graph_cid[idx(i)]==0)
            scc(idx(i));
        if(graph_cid[idx(-i)]==0)
            scc(idx(-i));
    }
    for(int i = 1; i<=n; i++){
        if(graph_cid[i]==graph_cid[idx(-i)]){
            cout << 0;
            return;
        }
    }
    cout << 1 << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제에서 주어진 2-CNF 식은 조건 명제를 통해서 표현할 수 있다.

조건 명제 http://www.ktword.co.kr/test/view/view.php?m_temp1=6266는 여기에서 설명되어 있다.

조건 명제의 진리표는 다음과 같다.

| p | q | p → q | |---|---|-------| | T | T | T | | T | F | F | | F | T | T | | F | F | T | 이 진리표를 사용해서 a OR b 를 조건 명제로 표현할 수 있다.

| a | b | a ∨ b | ¬a → b | ¬b → a | |---|---|-------|--------|--------| | T | T | T | T | T | | T | F | T | T | T | | F | T | T | T | T | | F | F | F | F | F | ¬a → b 와 ¬b → a 를 사용해서 implication graph를 그릴 수 있다.

implication graph에서는 모든 노드를 a와 ¬a 두가지로 나눈다. 그리고 위 조건 명제 두가지를 directed edge로 표현한다.

graph에서 scc를 이용해서 사이클을 찾은 뒤, a와 ¬a가 동시에 들어 있는지 확인한다. 만약 들어있다면, 이는 모순이므로 해를 구할 수 없다.
