풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
/*
 * n: 1~1000
 * m: 1001~2000
 * src: 2001
 * dest: 2002
 * weight: 2003
 */
int n, m, k;
int capacity[2004][2004] = {0,};
int flow[2004][2004] = {0,};
int src = 2001, dest = 2002, weight = 2003;
struct box{
    int node;
    int min_valid_flow;
};
void solve(){
    cin >> n >> m >> k; 
    int count, work;
    for(int i = 1; i<=n; i++){
        cin >> count;
        while(count--){
            cin >> work;
            capacity[i][work + 1000] = 1; 
        }
    }
    for(int i = 1; i<=n; i++){
        capacity[src][i] = 1;
        capacity[weight][i] = k;
    }
    capacity[src][weight] = k;
    for(int i = 1001; i<=m+1000; i++){
        capacity[i][dest] = 1;
    }
    
    while(1){
        int visited[2004] = {0,};
        queue<box> q;
        q.push({src, INT_MAX});
        while(!q.empty()){
            box cur = q.front();
            q.pop();
            if(cur.node == dest){
                int cursor = cur.node;    
                while(cursor != src){
                    flow[visited[cursor]][cursor] += cur.min_valid_flow; 
                    flow[cursor][visited[cursor]] -= cur.min_valid_flow; 
                    cursor = visited[cursor];
                }
                break;
            }
            for(int i = 1; i<=2003; i++){
                if(visited[i] == 0 && capacity[cur.node][i] - flow[cur.node][i] > 0){
                    visited[i] = cur.node;
                    q.push({i, min(cur.min_valid_flow, capacity[cur.node][i] - flow[cur.node][i])});
                }
            }
        }
        if(visited[dest]==0)
            break;
    }

    int ans = 0;
    for(int i = 1; i<=2003; i++)
        ans += flow[i][dest];
    cout << ans;
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제를 단순하게 한다면, 한 사람당 일 하나씩은 무조건 할 수 있고, 벌점 한 개당 일을 하나씩 더 할 수 있다.

처음에는 이분 매칭 알고리즘을 응용해서 greedy한 방식으로 답을 찾으려 했다. 이때 사용했던 방법은 무조건 벌점을 모두 소모하여 일을 매칭하도록 하고, 나중에 매칭을 시도하는 일이 있을 때 기존에 매칭된 사람이 벌점을 소모해서 매칭한 일이었다면 매칭을 제거하고 대신 벌점을 반납하도록 하였다. 그러나 이러한 방식은 시간 초과가 발생했다.

이후 검색을 해본 결과, 이 문제는 network flow를 사용해서 풀 수 있다. boj 17412 도시 왕복하기 1 이 문제가 network flow 알고리즘을 사용한 문제이다.

이분 매칭 문제도 network flow를 통해서 해결할 수 있다. 이 문제에서는 이분 매칭 문제를 network flow로 풀 되, 노드를 하나 생성하여 벌점의 총 개수 만큼 사람들에게 추가 유량이 흐를 수 있도록 해주어야 한다.

느낀점

굉장히 신기한 풀이법이었다. for문의 범위를 잘못 설정하여 디버깅 하는데 시간이 많이 걸렸다.
