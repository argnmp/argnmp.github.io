첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
int board[1000000] = {0,};
int sorted_nums[1000000] = {0,};
int sn_max_length[1000000] = {0,};
int sn_max_length_board_idx[1000000] = {0,};
int par[1000000] = {0,};
int segt[1<<21] = {0,};

void init_segt(int l, int r, int node){
    if(l==r){
        segt[node] = l;
        return;
    }
    int mid = (l+r)/2;
    init_segt(l, mid, node*2);
    init_segt(mid+1, r, node*2+1);
    if(sn_max_length[segt[node*2]] > sn_max_length[segt[node*2+1]]){
        segt[node] = segt[node*2];
    } 
    else {
        segt[node] = segt[node*1];
    }
}
void update(int l, int r, int node, int idx, int value, int i){
    if(l==idx && r==idx){
        if(sn_max_length[idx] < value){
            sn_max_length[idx] = value;
            sn_max_length_board_idx[idx] = i;
            segt[node] = idx;
        }
        return;
    }
    int mid = (l+r)/2;
    if(idx <= mid){
        update(l, mid, node*2, idx, value, i);
    } else {
        update(mid+1, r, node*2+1, idx, value, i);
    }
    if(sn_max_length[segt[node*2]] > sn_max_length[segt[node*2+1]]){
        segt[node] = segt[node*2];
    } 
    else {
        segt[node] = segt[node*2+1];
    }
}

int calc(int l, int r, int node, int s, int e){
    if(l==s && r==e){
        return segt[node];
    }
    int mid = (l+r)/2;
    if(e<=mid){
        return calc(l, mid, node*2, s, e);
    } else if(mid+1<=s){
        return calc(mid+1, r, node*2+1, s, e);
    } else {
        int left = calc(l, mid, node*2, s, mid);
        int right = calc(mid+1, r, node*2+1, mid+1, e);
        if(sn_max_length[left] > sn_max_length[right]){
            return left;
        } 
        else {
            return right;
        }
    }
}

void solve(){
    memset(par, -1, sizeof(par));
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> board[i];
        sorted_nums[i] = board[i];
    }
    sort(sorted_nums, sorted_nums + n);
    int prev_num = sorted_nums[0];
    int sorted_nums_size = 1;
    for(int i = 1; i<n; i++){
        if(prev_num != sorted_nums[i]){
            prev_num = sorted_nums[sorted_nums_size] = sorted_nums[i];
            sorted_nums_size += 1;
        } 
    }

    unordered_map<int, int> m;
    for(int i = 0; i<sorted_nums_size; i++){
        m.insert({sorted_nums[i], i});
    }

    update(0, sorted_nums_size-1, 1, m[board[0]], 1, 0);
    for(int i = 1; i<n; i++){
        int idx;
        if(m[board[i]]-1 < 0){
            update(0, sorted_nums_size-1, 1, m[board[i]], 1, i);
            continue;
        }

        idx = calc(0, sorted_nums_size-1, 1, 0, m[board[i]]-1);
        if(sn_max_length[idx]==0){
            update(0, sorted_nums_size-1, 1, m[board[i]], 1, i);
            continue;
        }
        par[i] = sn_max_length_board_idx[idx];
        update(0, sorted_nums_size-1, 1, m[board[i]], sn_max_length[idx]+1, i);
    }
    
    int p = calc(0, sorted_nums_size-1, 1, 0, sorted_nums_size-1);
    cout << sn_max_length[p] << '\n';
    int t = sn_max_length_board_idx[p];
    vector<int> stk;
    while(t!=-1){
        stk.push_back(board[t]);
        t = par[t];
    }
    for(int i = stk.size()-1; i>=0; i--)
        cout << stk[i] << ' ';

}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

segment tree를 이용한 풀이이다. 문제에서 주어진 예제는 다음과 같다

6
10 20 10 30 20 50

원래 가장 긴 증가하는 부분 수열을 찾기 위해서는 다음과 같은 방식을 사용했다.

맨 앞에서 부터 차례대로 순회하며 dp 배열에 현재 index까지 왔을 때 가장 긴 증가하는 부분 수열의 길이를 저장한다. 가장 긴 증가하는 부분 수열의 길이를 알아내기 위해서는 dp 배열의 맨 앞에서 부터 이전 index까지 현재 index의 수보다 작으면서 가장 긴 길이를 구한다.

위와 같은 방식은 O(n^2)의 시간이 소요되어서 이 문제에서는 최대 100000 까지의 수의 개수를 시간 내에 구할 수 없다.

dp 배열에서 구해야 하는 값은 현재 수보다 작으면서 가장 긴 길이이다. 이를 구하는데 O(n^2)의 시간이 걸린다.

이를 해결하기 위해서 segment tree를 사용할 수 있다. 위의 예제의 수에서 중복되지 않게 정렬해서 새로운 배열을 만든다.

sorted_nums: [10 20 30 50]

그리고 이와 같은 크기의 배열 두개를 생성한다.

sn_max_length: [           ]
sn_max_length_board_idx: [            ]

sn_max_length는 sorted_nums에서 같은 index의 값일 때 가장 긴 길이를 의미하고, sn_max_length_board_idx는 sn_max_length를 마지막으로 업데이트한 원래 수열에서의 index를 의미한다.

앞에서 시간이 오래걸리는 문제가 발생했던 부분은, 수열에서 처음 index 부터 이전 index까지 현재 index의 값보다 작은 것들 중 가장 긴 길이를 찾는 부분이었다. 이 부분을 segment tree를 이용해서 sn_max_length를 이용해서 찾을 수 있다.

6
10 20 10 30 20 50

이 수열에서 현재 dp배열을 계산 중인 index가 3이라고 해보자. 해당 index의 값은 30이고, sorted_nums에서는 같은 값이 index 2에 해당 된다. 따라서 index 0 ~ index 1까지는 30보다 무조건 작은 값이기 때문에, segment tree를 이용해서 최댓값을 가진 index를 얻는다. 이후 index 2에 대해서 가장 긴 길이는 sn_max_length[최댓값을 가진 index] + 1이 되고, 이를 segment tree에 업데이트 한다. 이를 수열의 끝까지 반복한다.

여기에 추가로 sn_max_length_board_idx와 par배열을 이용해서 해당 증가하는 부분 수열의 길이가 생성된 이전 index를 저장해두어서, 마지막에 수열을 출력해줄 수 있도록 한다.

boj에서 1284ms가 소요되어, 다른 풀이가 200ms인 것을 감안하면 별로 좋지 않은 풀이라고 할 수 있다.

두번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
int board[1000001] = {0,};
vector<int> lis;
int possible_idx[1000001] = {0,};

void solve(){
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> board[i];
    }

    possible_idx[0] = 0;
    lis.push_back(board[0]);
    for(int i = 1; i<n; i++){
        if(lis.back() < board[i]){
            lis.push_back(board[i]);
            possible_idx[i] = lis.size()-1;
        }
        else{
            int idx = lower_bound(lis.begin(), lis.end(), board[i]) - lis.begin();
            lis[idx] = board[i];
            possible_idx[i] = idx;
        }
    }
    cout << lis.size() << '\n';
    int search_idx = lis.size()-1;
    vector<int> stk;
    for(int i = n-1; i>=0; i--){
        if(possible_idx[i]==search_idx){
            stk.push_back(board[i]);
            search_idx -= 1;
        }
    }
    for(int i=stk.size()-1; i>=0; i--){
        cout << stk[i] << ' ';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

이분 탐색을 이용하는 풀이이다.

먼저 배열을 하나 생성한다. (위 코드에서는 lis 배열이다.) 주어진 수열을 앞에서 부터 탐색을 하면서, lis배열의 가장 뒤에 있는 수보다 크다면 lis에 push 한다. 만약 그렇지 않다면 lis 배열에서 현재 탐색 중인 수 이상인 수를 현재 탐색 중인 수로 교체해준다. 이를 마지막 수열의 수까지 반복한다.

lis배열의 길이가, 증가하는 부분 수열의 최대 길이가 된다.

문제는 lis배열에 있는 수들이 가장 긴 증가하는 부분 수열에 해당하지 않을 수도 있다는 것이다. 이를 해결하기 위해서, lis배열에서 몇번째 index에 들어갈 수 있었는지를 기록하는 배열을 추가로 사용한다. (위 코드에서는 possible_idx) 이다.

만약 lis배열의 길이가 4였다면, possible_idx배열을 뒤에서 앞으로 탐색하는 과정에서 3 -> 2 -> 1 -> 0에 해당하는 수열의 수를 찾으면 된다.

느낀점

segment tree로 풀긴 풀었지만 오래걸렸고 복잡했으며 성능도 좋지 않았다. 두번째 풀이를 잘 알아놓자.
