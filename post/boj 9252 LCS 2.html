문제

코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct pos {
    int x, y;
};
int dp[1001][1001] = {0,};
string xs, ys;
void solve(){
    cin >> xs >> ys; 
    for(int i = 1; i<=xs.length(); i++){
        for(int j = 1; j<=ys.length(); j++){
            if(xs[i-1]==ys[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else {
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
            }
        }
    } 
    pos target = {(int)xs.length(), (int)ys.length()};
    int ans = dp[target.x][target.y]; 
    cout << ans << '\n';
    vector<char> stack;
    if(ans != 0){
        while(target.x!=0&&target.y!=0){
            if(xs[target.x-1]==ys[target.y-1]){
                stack.push_back(xs[target.x-1]); 
                target = {target.x-1, target.y-1};
            } else {
                if(dp[target.x-1][target.y] < dp[target.x][target.y-1]){
                    target = {target.x, target.y-1};
                } else {
                    target = {target.x-1, target.y};
                }
                
            } 
        }
        while(!stack.empty()){
            cout << stack.back();
            stack.pop_back();
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

dp를 이용해서 풀어야 한다. dp[i][j]를 xs 문자열의 i번째 문자와 ys 문자열의 j번째 문자까지의 LCS로 정의한다. 그리고 i번째 문자와 j번째 문자가 같을 경우 dp[i][j] = dp[i-1][j-1] + 1 이고, i번째 문자와 j번째 문자가 다른 경우 dp[i][j] = max(dp[i-1][j], dp[i][j-1]) 가 된다.

이후 역으로 첫번째 문자까지 돌아가면서 dp 배열을 탐색해주면 된다.

dp 배열을 만드는 경우 index 0이 아닌 1부터 사용해주면 조건문을 달지 않고 깔끔하게 처리할 수 있다. 또한 string의 경우 index가 0부터 시작하기 때문에 xs = ' ' + xs 를 통해서 index를 1부터 시작하도록 할 수 있다.
