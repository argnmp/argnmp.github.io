문제

풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
bool calc(string sel){
    if(sel.size() == n){
        cout << sel;
        return true;
    }
    for(int i = 1; i<=3; i++){
        char c = '0' + i; 
        string next = sel + c;
        bool flag = true;
        for(int k = 1; k <= next.size()/2; k++){
            int ridx = next.size()-1;
            int lidx = ridx - k;
            bool is_safe = false;
            for(int t = 0; t<k; t++){
                if(next[lidx]!=next[ridx]){
                    is_safe = true;
                    break;
                }
                lidx -= 1;
                ridx -= 1;
            }
            if(!is_safe){
                flag = false;
                break;
            }
        }
        if(flag && calc(next))
            return true;
    } 
    return false;
}
void solve(){
    cin >> n;
    calc("");
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

앞에서부터 숫자를 하나씩 추가해주면서 해당 숫자를 추가할 수 있는지 확인해주면 된다. n이 최대 80이므로, 80*80*80이 걸려서 시간초과가 발생하지 않는다.

해당 숫자를 추가할 수 있는지 확인하는 방법은, 1~(새로운 수열의 길이/2)길이의 부분수열이 같은지 확인해주면 된다.

나는 위의 풀이에서 직접 인덱스를 찾아가면서 숫자를 비교해주었는데, string의 substr을 이용해서 문자열을 비교하는 방법도 사용할 수 있다.

느낀점

시간복잡도를 고려해서, 모든 경우의 수를 탐색하는 것인지 빠르게 파악하자.
