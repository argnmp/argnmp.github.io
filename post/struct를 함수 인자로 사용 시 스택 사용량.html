struct box{
    int x, y, z;
};
void print(box t, int k){
    if(t.x==0 || t.y==0 || t.z==0 ){
        return;
    }
    print({t.x-1, t.y-1, t.z-1}, 1);
    print({t.x-1, t.y-1, t.z-1}, 1);
    print({t.x-1, t.y-1, t.z-1}, 1);
}
int main(){
    print({1, 2, 3}, 1);
    return 0;
}

assembly로 컴파일 된 결과물을 보면, 재귀 호출에서 함수의 인자로 struct를 넘길 때 stack에 매 struct마다 공간을 할당한다는 것을 알 수 있다. 계산 결과가 한번 stack에 복사되고 이후 다시 register에 복사된 이후 재귀 호출이 발생하는 것이다.

함수가 return될 때 까지는 stack이 유지되니 한번의 재귀 함수 호출마다 struct 크기 만큼의 stack 공간이 사용되는 것이다. 위에서는 총 세번이니

struct 없이 직접 함수로 값을 넘기는 경우

void print(int x, int y, int z, int k){
    if(x==0 || y==0 || z==0 ){
        return;
    }
    print(x-1, y-1, z-1, 1);
    print(x-1, y-1, z-1, 1);
    print(x-1, y-1, z-1, 1);
}
int main(){
    print(100, 99, 98, 1);
    return 0;
}

assembly로 컴파일 된 결과를 보면, struct를 사용하는 경우와 달리 stack에 계산 결과를 복사하는 과정이 없이 바로 register에서 연산하여 재귀 호출을 하는 것을 알 수 있다.
