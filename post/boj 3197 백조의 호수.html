풀이

첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int r, c;
char board[1500][1500];
int dx[4] = {0,-1,1,0};
int dy[4] = {-1,0,0,1};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]};}
    bool safe(){ return 0<=x&&x<r&&0<=y&&y<c; }
    void print(){printf("[%d, %d]\n", x, y);}
    bool operator==(pos t){
        return x==t.x && y==t.y;
    }
};
vector<pos> target;

int cost[1500][1500] = {0,};
int visited[1500][1500] = {0,};

struct box{
    pos p; 
    int dist;
    bool operator()(box& a, box& b){
        return a.dist > b.dist;
    }   
    bool operator<(const box& t) const {
        return dist < t.dist;
    }
};

void print(){
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            printf("%2d", cost[i][j]);
        }
        cout << '\n';
    }
    cout << endl;
}

void calc_cost(){
    queue<box> q;
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            if((board[i][j]=='.'||board[i][j]=='L')){
                cost[i][j] = 0;       
                q.push({{i, j}, 0});
            }
        }
    }
    while(!q.empty()){
        box cur = q.front(); 
        q.pop(); 
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe()){
                if(board[next.x][next.y]=='.'||board[next.x][next.y]=='L')    
                    continue;
                else if(cur.dist + 1 < cost[next.x][next.y]){
                    cost[next.x][next.y] = cur.dist + 1;
                    q.push({next, cur.dist+1});
                }
            }
        } 
    }
}

queue<pos> next_q; 
bool check_reachable(int current_cost){
    pos to = target[1];

    queue<pos> q;
    while(!next_q.empty()){
        q.push(next_q.front());
        next_q.pop();
    }

    while(!q.empty()){
        pos cur = q.front();
        q.pop();
        if(cur==to)
            return true;
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i);
            if(next.safe() && visited[next.x][next.y]==0){
                if(cost[next.x][next.y]==0){
                    visited[next.x][next.y] = 1;
                    q.push(next);
                }
                else if(cost[next.x][next.y]==current_cost+1){
                    visited[next.x][next.y] = 1;
                    next_q.push(next); 
                }
            }
        }
    }
    return false;
}

int calc_dist(){
    vector<box> v;
    for(int i = 0; i<r; i++)
        for(int j = 0; j<c; j++)
            v.push_back({{i, j}, cost[i][j]});
    sort(v.begin(), v.end());
    
    box cur = v[0];
    int days = 0;  
    pos from = target[0];
    next_q.push(from);
    visited[from.x][from.y] = 1;
    for(int i = 0; i<v.size();){
        while(v[i].dist == cur.dist && i<v.size()){
            cur = v[i];
            cost[cur.p.x][cur.p.y] = 0;
            i+=1;
        } 
        if(check_reachable(cur.dist))
            return days;
        cur = v[i];
        days += 1;
    }
}

void solve(){
    memset(cost, -1, sizeof(cost));
    cin >> r >> c;
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            char c = cin.get();
            while(c=='\n'||c=='\r'||c==' ')
                c = cin.get();
            board[i][j] = c;
            if(board[i][j]=='L')
                target.push_back({i, j});
            cost[i][j] = INT_MAX;
        }
    }
    calc_cost();
    cout << calc_dist();
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

먼저 각 얼음이 녹는데 걸리는 시간을 기록한다. calc_cost 에서 이 작업을 수행해서 cost배열에 기록한다.

bfs를 사용해서 계산을 하는데, 모든 물 좌표를 먼저 queue에 넣어두고 걸리는 시간이 더 적은 경우만 같은 좌표에 재방문하도록 하여 계산한다.

아래는 문제에서 주어진 예시에 대해 걸리는 시간을 계산한 것이다.

..XXX...X..
.X.XXX...L.
....XXX..X.
X.L..XXX...

00121000100
01012100000
00001210010
10000121000

cost배열에서 걸리는 시간이 같은 좌표의 값들을 모두 바꿔주며, 순서대로 순회를 한다. 예를 들면 cost배열의 값이 1인 모든 좌표들을 0으로 바꾸어주고, 2인 좌표들에 대해 0으로 바꾸어 주고, 3인 좌표들에 대해 0으로 바꾸어주며 cost배열이 모두 0이 될 때까지 반복하는 것이다.

calc_dist에서 이 작업을 수행한다.

각 순회마다, L에서 L로 도달이 가능한지 확인한다. 하지만 단순히 도달가능한지 확인하기 위해 bfs를 매 순회마다 돌리게 되면 시간초과가 발생한다.

매 순회에서 더이상 이동할 수 없는 경우 이는 얼음이 아직 녹지 않은 경우이다.(cost배열 값이 0이 아님) 이때 녹지 않은 좌표는 다음 번에 녹게되는(cost배열이 0 이 되는) 경우이므로, 다른 L을 찾지 못했다면 다음 순회에서는 여기서부터 다시 bfs 탐색을 재개하면 된다.

check_reachable에서 이 작업을 한다. bfs를 탐색하는데, cost배열의 값을 확인해서 다음에 녹게 되는 얼음인 경우 next_q에 값을 넣어 놓는다. check_reachable 에서는 next_q의 값들을 pop하여 bfs 탐색에 사용될 queue에 집어넣는다.

이 풀이에서는 얼음이 녹는 시간을 먼저 계산해두었지만, 먼저 계산하지 않고도 여러개의 queue를 사용해서 바로 계산이 가능하게 코드를 짤 수 있을 것 같다.

느낀점

cost를 계산하는 것 까지는 빠르게 작성을 할 수 있었으나, 이후 L에서 L까지 필요한 시간을 구하는 것에서 어려움을 겪었다. 처음에 priority_queue로 작성을 했지만 메모리 초과가 발생했고, 이를 해결하기 위해서 복잡하게 고민을 했다. 하지만 얼음을 녹이며(cost를 0으로 만들며) bfs로 확인한다고 생각하니, 풀이를 떠올릴 수 있었다. 너무 복잡하게 생각하지 말고, 단순화 해서 명료하게 생각하려고 노력해야겠다.
