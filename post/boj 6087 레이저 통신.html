코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int w, h;
char board[101][101];
int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};
struct pos {
    int x, y, m, dir;
    pos next(int t){
        return {x+dx[t%4], y+dy[t%4]};
    }
    bool safe(){
        return 0<=x&&x<h&&0<=y&&y<w;
    }
    void print(){
        printf("[%d, %d]: %d, %d\n", x,y,m,dir);
    }
};
struct compare {
    bool operator()(pos& a, pos& b){
        return a.m > b.m;
    }
};
void solve(){
    cin >> w >> h; 
    vector<pos> c;
    for(int i = 0; i<h; i++){
        for(int j = 0; j<w; j++){
            char ch = cin.get();
            while(ch=='\n' || ch==' ' || ch=='\r'){
                ch = cin.get();
            }
            board[i][j] = ch;
            if(ch=='C')
                c.push_back({i, j, 0, -1});
        }
    }
    pos from = c[0];
    pos to = c[1];
    int visited[101][101][4] = {0,};
    fill(&visited[0][0][0], &visited[100][100][4], INT_MAX);
    priority_queue<pos, vector<pos>, compare> q;
    q.push(from);
    while(!q.empty()){
        pos cur = q.top();
        q.pop();
        if(cur.x == to.x && cur.y == to.y) {
            cout << cur.m;
            return;
        }
        for(int i = 0; i<4; i++){
            int new_m = (cur.dir == -1 || cur.dir == i) ? cur.m : cur.m + 1;
            pos n = cur.next(i);
            if(n.safe() && board[n.x][n.y]!='*' && new_m < visited[n.x][n.y][i]){
                visited[n.x][n.y][i] = new_m;
                n.m = new_m;
                n.dir = i;
                q.push(n);
            }
        } 
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

bfs로 풀려고 하다가, queue를 priority queue로 바꾸어서 풀었다.

pos 에는 좌표와 함께 어느 방향에서 왔는지(dir) 그리고 현재까지의 거울 개수(m)를 저장한다. 상,우,하,좌 를 탐색하는데 각각 0, 1, 2, 3 을 매핑 한다. 그리고 탐색을 계속해나가면서 어느 방향에서 해당 좌표에 왔는지를 기록하고. 만약 그다음 탐색 방향이 pos의 m 값과 같다면 거울 개수를 유지하고, 다르다면 거울 개수를 한개 늘린다.

bfs에서 visited 배열은 처음에는 4가지 방향을 추가해서, 좌표마다 각 방향마다 한번씩만 방문하도록 했다. 하지만 다음과 같은 반례가 있다

4 5
C..*
...*
...*
*.**
...C

priority queue로 하기 때문에 같은 방향의 좌표를 한번 더 방문해야 한다.

따라서 해당 좌표에서의 거울 개수를 visited에 저장하도록 해서, 같은 방향일 경우 거울 개수가 더 적은 경우에만 한번 더 확인하도록 했다.

priority queue는 거울 개수가 적은 순서대로 정렬을 하도록 해서, 목적지에 도달하면 이 값을 print하고 리턴하면 된다.

느낀점

처음에 50퍼센트에서 틀렸다고 나와서 반례를 보고 맞췄다. 예전에 못풀었어서 답을 봤었는데 그 영향도 있는 듯 하다. 여기서는 priority queue로 풀었지만, bfs로 같은 방향에 있는 모든 좌표를 먼저 방문하는 방식으로 해결할 수 있을 것 같다. 레이저 쏘는 것처럼 끝까지 가면서 queue에 모두 추가하는 것이다.
