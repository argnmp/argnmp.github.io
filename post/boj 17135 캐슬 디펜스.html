#include <iostream>
#include <algorithm>

#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, d;
int board[15][15] = {0,};

struct pos{
    int x, y, dist;
    bool safe(int x_end){ return 0<=x&&x<=x_end&&0<=y&&y<m; }
    void print(){
        printf("[%d, %d]\n", x, y);
    }
};
struct box{
    pos p;
    int dist;
    bool operator<(const box& t) const {
        if(dist == t.dist){
            return p.y < t.p.y;
        }
        return dist < t.dist;
    }
};

int ans = 0;

void print(int board[15][15]){
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            cout << board[i][j] << ' ';
        }
        cout << endl;
    }
}
void calc(vector<int>& archer){
    int board_cp[15][15];
    // copy board;
    for(int i = 0; i<n; i++)
        for(int j = 0; j<m; j++)
            board_cp[i][j] = board[i][j];

    int acc = 0;
    int visited[15][15] = {0,};
    for(int i = n-1; i>=0; i--){
        // print(board_cp);
        // cout << endl;
        queue<pos> removal;
        for(int j: archer){
            pos cursor = {i, j};
            // cout << "cursor";
            // cursor.print();
            vector<box> enimies;
            for(int r = 0; r < d; r++){
                for(int c = -(d-r-1); c <= (d-r-1); c++){
                    pos target = {cursor.x-r, cursor.y+c, r+c};
                    if(target.safe(i) && board_cp[target.x][target.y] != 0){
                        // cout << "found"; 
                        // target.print();
                        enimies.push_back({target, r+abs(c)});
                    }
                }
            }
            sort(enimies.begin(), enimies.end());
            if(enimies.size()!=0){
                pos target = enimies[0].p;
                removal.push(target);
                if(board_cp[target.x][target.y] == 1){
                    board_cp[target.x][target.y] = -1;
                    acc += 1;
                }

            }
        }
        // remove enimies;
        while(!removal.empty()){
            pos t = removal.front();
            removal.pop();
            board_cp[t.x][t.y] = 0;
        }
    }
    
    ans = max(ans, acc);
}

void solve(){
    cin >> n >> m >> d;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            cin >> board[i][j];
        }
    }

    vector<int> org;
    vector<int> ind;
    for(int i = 0; i<m; i++){
        org.push_back(i+1);
        ind.push_back((i<3)? 1 : 0);
    }
    sort(org.begin(), org.end());
    sort(ind.begin(), ind.end());
    do {
        vector<int> archer;
        for(int i = 0; i<m; i++){
            if(ind[i]==1)
                archer.push_back(i);
        } 
        calc(archer);
    } while(next_permutation(ind.begin(), ind.end()));
    /* vector<int> archer = {0, 1, 2};
    calc(archer); */
    
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

궁수가 존재할 수 있는 모든 위치에 대한 조합을 구해서, 이들 모두에 대해 시뮬레이션을 돌려 최대 개수를 구한다.

나는 처음에 궁수의 제거대상이 되는 적을 찾기 위해서, bfs와 priority queue를 사용하여 찾을 경우 바로 찾는 작업을 중단하도록 했었다. 하지만 이 로직은 문제가 있다. 바로 종료할 경우 더 왼쪽에 있는 적을 찾지 못하는 경우도 생긴다는 것이다. 이러한 것을 잘 고려하자.

느낀점

매우 오랜 시간이 걸렸다. 위의 풀이에 적은 것과 같이, 적을 찾는 로직에 대해 아예 잘못 생각하고 있었다. 그 로직이 잘못되었다고 생각하지 못하여 디버깅 하느라 많은 시간을 보냈고 결국 풀지 못했다. 또한 오타가 많이 발생하거나, 반복문에서 사용한 변수명을 다시 사용하는 등의 문제로 디버깅이 힘들었다. 조금 더 집중해서 꼼꼼히 작성하자.
