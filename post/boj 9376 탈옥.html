#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int h, w;
char board[103][103] = {};
int dx[4] = {0,1,-1,0};
int dy[4] = {1,0,0,-1};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){ return 0<=x && x<h && 0<=y && y<w; }
    void print(){
        printf("%d, %d\n", x, y);
    }
};
vector<pos> start;
int done[3][103][103] = {0,};
struct box{
    pos p;
    int cost;
    bool operator()(box& a, box& b){
        return a.cost > b.cost;
    }   
};
void dijkstra(int target){
    fill(&done[target][0][0], &done[target][102][103], INT_MAX);
    priority_queue<box, vector<box>, box> pq;
    pq.push({start[target], 0});
    // printf("dbg: %d | %d, %d, %d\n", target, target, start[target].x, start[target].y);
    done[target][start[target].x][start[target].y] = 0;
    while(!pq.empty()){
        box cur = pq.top();
        pq.pop();
        if(cur.cost > done[target][cur.p.x][cur.p.y])
            continue;
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe() && board[next.x][next.y]!='*'){
                int next_cost = (board[next.x][next.y]=='#') ? cur.cost + 1 : cur.cost;
                if(next_cost < done[target][next.x][next.y]){
                    done[target][next.x][next.y] = next_cost;
                    pq.push({next, next_cost});
                }
            }
        }
    }
}
void solve(){
    memset(done, 0, sizeof(done));
    fill(&board[0][0], &board[102][103], ' ');
    start = vector<pos>();
    cin >> h >> w;
    h += 2;
    w += 2;
    for(int i = 1; i<=h-2; i++){
        for(int j = 1; j<=w-2; j++){
            char ch = cin.get();
            while(ch=='\n'||ch=='\r'||ch==' '){
                ch = cin.get();
            }
            board[i][j] = ch;
            if(ch=='$')
                start.push_back({i, j});
        }
    }  
    start.push_back({0,0});
    
    for(int i = 0; i<3; i++){
        dijkstra(i);
    }
    int ans = INT_MAX;
    for(int i = 0; i<h; i++){
        for(int j = 0; j<w; j++){
            if(1<=i&&i<=h-2&&1<=j&&j<=w-2&&board[i][j]=='#'){
                ans = min(ans, done[0][i][j] + done[1][i][j] + done[2][i][j]-2);
            } else {
                ans = min(ans, done[0][i][j] + done[1][i][j] + done[2][i][j]);
            }
        }
    }
    cout << ans << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int test;
    cin >> test;
    while(test--){
        solve(); 
    }
    return 0;
}

두 명의 죄수에 추가로 다른 한 사람을 추가해서, 세 명이 각 좌표까지 가기 위한 열어야 하는 문의 최솟값을 구한다. 이후 모든 좌표를 확인해 보았을 때, 세 명의 최솟값의 합이 최소인 좌표가 바로 열어야 하는 문의 최솟값이 된다.

합을 구할 때 주의할 사항이 있다. # 으로 문이 있는 좌표에서는 세 명의 최솟값을 더한 값에 -2를 해주어야 한다. 마지막 도착지에 문이 있다면 세 명 모두 문을 열었을 것이기 때문이다. 해당 좌표까지 오는 과정에서 문을 겹치게 열었다면, 이는 실제로 열린 문의 값이 아니게 된다. 하지만 다른 좌표에서 세명이 겹치지 않고 오는 경로에 대한 해를 가지고 있게 되어서 구할 수 있다.
