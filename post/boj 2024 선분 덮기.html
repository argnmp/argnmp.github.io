풀이 1

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int m;
struct box{
    int l, r;
    bool operator<(const box& t) const {
        if(l==t.l){
            return r > t.r; 
        }
        return l < t.l;
    }
};
vector<box> lines;
void solve(){
    cin >> m;
    int l, r;
    while(1){
        cin >> l >> r; 
        if(l==0 && r==0)
            break;
        if(l<0)
            l = 0;
        if(l>m)
            l = m;
        if(r<0)
            r = 0;
        if(r>m)
            r = m;
        lines.push_back({l, r});
    }
    sort(lines.begin(), lines.end());
    int start_idx = 0;
    while(start_idx < lines.size() && lines[start_idx].l == lines[start_idx].r){
        start_idx += 1;
    }
    if(start_idx == lines.size()){
        cout << '0';
        return;
    }
    box cur_line = lines[start_idx];
    int buffer = -1;
    int ans = 1;
    for(int i = start_idx+1; i<lines.size(); i++){
        box line = lines[i];
        // cout << line.l << ' ' << line.r <<endl;
        if(line.l==line.r)
            continue;
        if(lines[i].l <= cur_line.r){
            if(lines[i].r > cur_line.r){
                if(buffer==-1){
                    buffer = cur_line.r;
                    ans += 1; 
                }
                else {
                    if(lines[i].l > buffer){
                        buffer = cur_line.r;
                        ans += 1;
                    }
                }
                cur_line.r = lines[i].r;
            }
        }
        else {
            if(lines[i].l <= m){
                cout << '0';
                return;
            } 
        }
    }
    // cout << cur_line.l << ' '  << cur_line.r << endl;
    if(cur_line.l==0 && cur_line.r ==m){
        cout << ans;
        
    }
    else {
        cout << '0';
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

라인 스위핑 문제이다.

[0, m] 구간에서 주어진 선으로 덮을 수 있는지를 구해야 한다. 따라서 일단 선을 입력받을 때, 이 구간의 양 끝을 넘어가는 경우 양 끝으로 범위를 바꿔준다. 예를 들어서 -10 20 이었다면 0 20으로 변경한다.

그 다음 이 선들을 왼쪽 좌표가 적은 것 부터 오름차순으로 정렬한다. 만약 왼쪽 좌표가 같다면 오른쪽 좌표가 큰 것이 앞에 오도록 한다.

그리고 순서대로 선들을 순회하면서 구간을 모두 덮기 위한 선의 최소 개수를 구해나간다. 다음의 경우를 보자.

일단 불가능한 경우이다

0 5
6 10

구간이 [0, 10]이라면 0 5를 탐색한 다음, 6 10을 탐색하는데 현재까지의 선의 오른쪽 좌표보다 왼쪽 좌표가 크기 때문에 덮을 수 없는 경우가 된다.

그 다음 경우를 보자.

0 5
0 3
2 5

0 5다음에는 0 3을 순회하게 되는데 이는 최소 개수의 선에 들어가지 않아야 하고. 이후에 2 5 또한 최소 개수의 선에 들어가지 않아야 한다.

따라서 현재까지의 선의 오른쪽 좌표보다 작다면 최소 개수의 선에 들어가지 않도록 한다.

그리고 다음의 경우를 보자

0 5
2 8
4 10

0 5 다음에는 2 8을 순회한다. 이때에는 왼쪽 좌표가 현재까지의 선의 오른쪽 좌표보다 작고, 오른쪽 좌표가 현재까지의 선의 오른쪽 좌표보다 크기 때문에 최소 개수의 선에 들어간다.

그 다음에는 4 10을 순회한다. 앞선 규칙대로라면 이 탐색에서도 최소 개수의 선에 들어가게 된다. 하지만 2 8로 덮는 대신 4 10으로 덮으면 되므로, 최소 개수의 선에 2 8이 들어가지 않아야 한다. 이를 다르게 말하면, 최소 개수를 카운팅 하는 변수가 있을 때, 2 8을 탐색할 때는 1을 올려주되, 4 10을 탐색할 때에는 올려주지 않아야 한다는 것이다. 만약 그다음에 6 100를 탐색하더라도 마찬가지로 올려주지 않아야 한다.

이를 해결하기 위해서 나는 코드에서 buffer를 사용했다. 최소 개수를 카운팅 했다면, 오른쪽 좌표를 buffer에 넣어 두어서, 그다음 탐색하는 선들의 왼쪽 좌표가 buffer보다 작다면 카운팅 하지 않는 것이다.

#include <iostream>

#define dbgarr(r, c, x) printf("[DBG] %s\n", #x);for(int i=0; i<r; i++){for(int j=0; j<c; j++){printf("%2lld ", (ll)x[i][j]);}printf("\n");}
using namespace std;
using ll = long long;
int m;
struct line{
    int u, v;
    bool operator<(const line& t) const {
        return u < t.u;
    }
    void print(){
        printf("[%d, %d]\n", u, v);
    }
};

void solve(){
    cin >> m;
    int u, v;
    vector<line> cand;
    while(1){
        cin >> u >> v;
        if(u==0&&v==0)
            break;
        if(u<0)
            u=0;
        if(v<0)
            v=0;
        cand.push_back({u, v});
    }
    sort(cand.begin(), cand.end());
    /* for(line& t: cand){
        t.print();
    } */
    line cur = {0,0};
    int end = 0;
    int cnt = 0;
    for(int i = 0; i<cand.size(); i++){
        if(cand[i].u <= end){
            cur.v = max(cur.v, cand[i].v);
        }   
        else if(cand[i].u <= cur.v) {
            end = cur.v;
            cur.v = max(cur.v, cand[i].v);
            cnt += 1;
        }
        if(cur.v >= m){
            cout << cnt + 1;
            return;
        }
    } 
    cout << 0;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}
