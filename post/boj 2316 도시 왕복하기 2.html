풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
/*
 * 3~n: original node
 * 403~400+n: duplicated node
 */
int n, p;
int capacity[801][801] = {0,};
int flow[801][801] = {0,};
int src = 1, dest = 2;

struct box{
    int node, min_valid_flow;
};
void solve(){
    cin >> n >> p;
    int u, v;
    for(int i = 0; i<p; i++){
        cin >> u >> v;
        if(u==src||u==dest){
            capacity[u][v] = 1;
            capacity[v+400][u] = 1;
        }
        else if(v==src||v==dest){
            capacity[u+400][v] = 1;
            capacity[v][u] = 1;
        }
        else{
            capacity[u+400][v] = 1; 
            capacity[v+400][u] = 1; 
        }
    }
    for(int i = 3; i<=n; i++){
        capacity[i][i+400] = 1;
    }
    while(1){
        int visited[801] = {0,};
        queue<box> q;
        q.push({src, INT_MAX});
        while(!q.empty()){
            box cur = q.front();
            q.pop();
            if(cur.node==dest){
                int cursor = cur.node;
                while(cursor!=src){
                    flow[visited[cursor]][cursor] += cur.min_valid_flow;
                    flow[cursor][visited[cursor]] -= cur.min_valid_flow;
                    cursor = visited[cursor];
                }
                break;
            }
            for(int i = 1; i<801; i++){
                if(visited[i]==0 && capacity[cur.node][i]-flow[cur.node][i]>0){
                    visited[i] = cur.node; 
                    q.push({i, min(cur.min_valid_flow, capacity[cur.node][i]-flow[cur.node][i])});
                }        
            }
        }
        if(visited[dest]==0)
            break;
    }     
    int ans = 0;
    for(int i = 1; i<801; i++){
        ans += flow[i][dest];
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

boj 17412 도시 왕복하기 1 에서 조금 달라진 문제이다. 여기서는 1번 노드와 2번 노드를, 다른 노드를 재방문하지 않고 최대 몇번 왔다 갔다 할 수 있는 지를 구해야 한다.

문제를 조금 다르게 생각해보면, 출발 노드(1번 노드)와 도착 노드(2번 노드)를 왔다 갔다 한다는 것은 결국 출발 노드에서 도착 노드 까지 같은 정점을 거치지 않는 경로의 수가 총 몇개 인지를 찾는 것과 같다. boj 17412 도시 왕복하기 1 에서는 같은 간선을 거치지 않는 경로의 수가 총 몇개 인지를 구해야 했다면, 여기서는 같은 정점을 거치지 않아야 하고 또한 무방향 그래프인 점에서 차이가 있다.

7 8
1 3
1 4
3 5
4 5
5 6
5 7
6 2
7 2

이 예제 입력의 5번 노드는, boj 17412 도시 왕복하기 1 에서는 구해야 하는 경로의 수에서 중복으로 들어가도 되었으나, 이 문제에서는 중복으로 들어가면 안된다.

network flow 알고리즘은 같은 간선을 중복으로 방문하지 않으며, 최대 경로의 개수를 찾을 수 있는 기능이 있다. (용량을 1로 해주므로써) 이를 응용해서, 정점을 중복으로 방문하지 않게 하기 위해서는 한 개의 정점을 두 개로 나누어서 용량이 1인 간선으로 연결해주면 된다. a 라는 정점이 있다면 a'정점을 추가하고 용량이 1인 간선으로 연결한다. 그리고 기존 a정점으로 들어오는 간선은 a로 연결해주면 되고, 기존 a정점에서 나가는 간선은 a'에서 나가도록 하면 된다.

무방향 그래프는, u노드와 v노드가 연결 되어 있다고 하면 u->v와 v->u의 용량 모두를 1로 설정해주는 방식으로 처리하면 된다.

느낀점

정점을 두 개의 정점으로 나누고 용량이 1인 간선으로 연결하는 방법은 떠올렸지만, 무방향 그래프를 어떻게 처리해야 할지 떠올리지 못했다.
