#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
char board[10][10] = {0,};
// up: 0, down: 1, left: 2, right: 3
int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};
struct pos{
    int x, y;
    int type; // 0: none, 1: red, 2: blue;
    // check whether current pos has higher priority
    bool direction_order(int dir, pos& t){
        if(dir==0){
            return x < t.x;
        } else if(dir==1){
            return x > t.x;
        } else if(dir==2){
            return y < t.y;
        } else {
            return y > t.y;
        }
    }
    pos next(int i){ return {x+dx[i%4], y+dy[i%4], type}; }
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<m; }
    void print(){
        printf("[%d, %d]: type: %d\n", x, y, type);
    }
};

int ans = INT_MAX;
void move(int count, pos a, pos b, pos hole){
    // up down left right
    /* cout << "++++++" << count << endl;
    a.print();
    b.print();
    cout << "++++++" << endl; */
    if(count > 10){
        return;
    }

    //p1 has higher priority than p2: p1 should move first
    for(int i = 0; i<4; i++){
        pos p1 = a, p2 = b; 
        if(!p1.direction_order(i, p2)){
            swap(p1, p2);
        }
        // move p1;
        pos next_p1 = p1.next(i);
        bool is_p1_hole = false;
        while(next_p1.safe() && board[next_p1.x][next_p1.y]!='#'){
            p1 = next_p1;
            if(p1.x == hole.x && p1.y == hole.y){
                is_p1_hole = true;
                break;
            }
            next_p1 = p1.next(i);
        }

        // move p2;
        pos next_p2 = p2.next(i);
        bool is_p2_hole = false;
        while(next_p2.safe() && board[next_p2.x][next_p2.y]!='#' && (is_p1_hole || next_p2.x != p1.x || next_p2.y != p1.y)){
            p2 = next_p2;
            if(p2.x == hole.x && p2.y == hole.y){
                is_p2_hole = true;
                break;
            }
            next_p2 = p2.next(i);
        }

        if(!is_p1_hole && !is_p2_hole){
            move(count+1, p1, p2, hole); 
        } else if((is_p1_hole && p1.type==1 && !is_p2_hole)||(is_p2_hole && p2.type==1 && !is_p1_hole)){
            /* printf("found: %d\n", count);
            cout << "------" << count << endl;
            p1.print();
            p2.print();
            cout << "------" << endl; */
            ans = min(ans, count); 
            
        } else {
            continue;
        }
    }   
}
void solve(){
    pos red, blue, hole;   
    cin >> n >> m;
    char ch;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<m; j++){
            ch = cin.get();
            while(ch=='\n'||ch=='\r'||ch==' ')
                ch = cin.get();
            board[i][j] = ch;
            if(ch=='R')
                red = {i, j, 1};
            else if(ch=='B')
                blue = {i, j, 2};
            else if(ch=='O')
                hole = {i, j};
        }
    }
    move(1, red, blue, hole);
    if(ans==INT_MAX)
        cout << "-1";
    else
        cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

일단 전체 경우의 수를 모두 탐색해주어도 시간초과에 걸리지 않는다. 4^10(기울이는 경우의 수) * 16(빨간 구슬과 파란 구슬을 움직이는 최대 횟수) = 16777216 으로 1억이 되지 않기 때문이다.

따라서 모든 경우의 수를 탐색한다. 그리고 탐색하는 과정에서 다음과 같이 생각하면 쉽다.

  • 한 구슬의 도착지에 다른 구슬이 있을 수 있다. 따라서 구슬 간의 영향을 판단 하여야 하는데, 이는 어떤 구슬이 우선순위를 가지고 먼저 굴려야 하는지를 통해 판단이 가능하다. 즉, 빨간 구슬, 파란 구슬 둘 중 어느 것을 먼저 굴려야 하는지를 확인하여 그 구슬 먼저 이동시킨다.
    • 예를 들어 위로 기울이는 경우 x좌표가 위에 있는 것의 우선순위가 높다(먼저 굴려야 한다) x좌표가 위에 있는 구슬의 경우 기울일 때 다른 구슬의 영향을 받지 않기 때문이다. 구슬을 움직이는 과정에서 구멍에 들어갔는지를 판단하고, 구멍에 들어가지 않고 특정 위치에 도달했다면 그 다음 구슬을 굴릴 때 해당 위치로 이동하지 못하도록 한다.
    • 4방향 모두에 대해서 구슬의 우선순위를 정해서 이동해주면 된다.
  • 두 구슬을 모두 굴린 뒤에는 빨간 구슬과 파란 구슬이 구멍에 빠졌는지를 판단하여, 기울이는 최소 횟수를 업데이트 해줄 수 있다.
