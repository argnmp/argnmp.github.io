첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
void solve(){
    int prev_rank[500] = {0,};
    int node_rank_map[501] = {0,};
    vector<set<int>> graph(501, set<int>());
    int in[501] = {0,};
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> prev_rank[i];
        node_rank_map[prev_rank[i]] = i;
    }
    cin >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v; 
        if(node_rank_map[u] < node_rank_map[v]){
            swap(u, v);
        }
        graph[u].insert(v);
        in[v] += 1;
    }
    
    for(int i = 0; i<n; i++){
        for(int j = i+1; j<n; j++){
            u = prev_rank[i];
            v = prev_rank[j];
            if(graph[v].find(u)==graph[v].end()){
                // printf("[%d, %d]\n", u, v);
                graph[u].insert(v); 
                in[v] += 1;
            }
        }
    }

    queue<int> q;
    for(int i = 1; i<=n; i++){
        if(in[i]==0)
            q.push(i);
    }
    vector<int> ans;
    while(!q.empty()){
        if(q.size() > 1){
            cout << "?" << '\n';
            return;
        }
        int cur = q.front();
        q.pop();
        ans.push_back(cur);
        for(int next: graph[cur]){
            in[next] -= 1; 
            if(in[next]==0)
                q.push(next);
        }
    }
    if(ans.size() == n){
        for(int v: ans){
            cout << v << ' ';
        }
    } else {
        cout << "IMPOSSIBLE";
    }
    cout << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--){
        solve(); 
    }
    return 0;
}

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
int prev_rank[500] = {0,};
int graph[501][501] = {0,};
int in[501] = {0,};
void solve(){
    memset(prev_rank, 0, sizeof(prev_rank));
    memset(graph, 0, sizeof(graph));
    memset(in, 0, sizeof(in));
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> prev_rank[i];
    }
    for(int i = 0; i<n; i++){
        for(int j = i+1; j<n; j++){
            graph[prev_rank[i]][prev_rank[j]] = 1;
            in[prev_rank[j]] += 1;
        }
    }
    cin >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v; 
        if(graph[u][v]==1){
            graph[u][v] = 0;
            graph[v][u] = 1;
            in[v] -= 1;
            in[u] += 1;
        } else {
            graph[v][u] = 0;
            graph[u][v] = 1;
            in[u] -= 1;
            in[v] += 1;
        }
    }

    queue<int> q;
    for(int i = 1; i<=n; i++){
        if(in[i]==0)
            q.push(i);
    }
    vector<int> ans;
    while(!q.empty()){
        if(q.size() > 1){
            cout << "?" << '\n';
            return;
        }
        int cur = q.front();
        q.pop();
        ans.push_back(cur);
        for(int i = 1; i<=n; i++){
            if(graph[cur][i]==1){
                in[i] -= 1;
                if(in[i]==0)
                    q.push(i);
            }
        }
    }
    if(ans.size() == n){
        for(int v: ans){
            cout << v << ' ';
        }
    } else {
        cout << "IMPOSSIBLE";
    }
    cout << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--){
        solve(); 
    }
    return 0;
}

백준 온라인 저지에서 32ms 소요

풀이

topological sort를 사용해야 한다.

처음 생각한 방식은 다음과 같다.

  1. 상대적인 등수가 바뀐 쌍에 대한 정보를 가지고, 작년 순위를 참고하여 더 높은 등수를 가진 팀을 판단하여, 그래프의 edge를 연결한다.
  2. 작년 순위를 순회하며 ({높은 등수 팀, 낮은 등수 팀} 쌍에 대한 순회) 1번에서 edge가 연결되지 않은 쌍에 대하여 그래프의 edge를 (높은 등수 팀 -> 낮은 등수 팀)으로 연결한다.
  3. 그래프를 가지고 topological sort를 수행하며 queue에 1개 이상의 후보군이 들어 있다면 ?를 출력한다.
  4. topological sort 결과로 나온 값에 (위의 코드에서 ans배열) 전체 팀이 들어 있지 않다면 IMPOSSIBLE을 출력한다.

2번에서 {높은 등수, 낮은 등수} 쌍에 대한 순회는 다음과 같이 순회하는 것을 의미한다.

작년 순위가 5 4 3 2 1 인 경우,

5, 4
5, 3
5, 2
5, 1
4, 3
4, 2
4, 1
3, 2
3, 1
2, 1

이와 같이 순회한다.

이렇게 작성한 코드가 위의 첫번째 풀이 이다. 제출 결과 맞았습니다가 나왔지만 248ms가 소요되었다.

이걸 바탕으로 다시 방식을 생각해보았다.

  1. 작년 순위를 순회하며 ({높은 등수, 낮은 등수} 쌍에 대한 순회) 그래프의 edge를 (높은 등수 팀 -> 낮은 등수 팀)으로 연결한다.
  2. 상대적인 등수가 바뀐 쌍에 대한 정보를 통해, 해당하는 위의 그래프의 edge의 방향을 바꾸어 준다.
  3. 그래프를 가지고 topological sort를 수행하며 queue에 1개 이상의 후보군이 들어 있다면 ?를 출력한다.
  4. topological sort 결과로 나온 값에 (위의 코드에서 ans배열) 전체 팀이 들어 있지 않다면 IMPOSSIBLE을 출력한다.

이렇게 작성한 코드가 위의 두번째 풀이 이다. 제출 결과 맞았습니다와 함께 32ms가 소요되었다.

느낀점

처음에 문제를 읽고 상대적인 등수가 바뀐 쌍이 (높은 등수, 낮은 등수) 순서 대로 주어진다고 잘못 이해하여, 푸는데 시간이 조금 소요되었다.

백준의 다른 제출에는 20ms가 소요된 풀이도 있어서, 위의 두번째 풀이와 어떤 차이가 있는지 확인해보자.
