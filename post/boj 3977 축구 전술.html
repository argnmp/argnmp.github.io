풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
int g_id = 1, id[100001] = {0,}, done[100001] = {0,};
vector<int> stk;
int g_cid = 1, cid[100001] = {0,}, cid_in[100001] = {0,};
vector<vector<int>> cid_group;

vector<vector<int>> graph(100001, vector<int>());

void reset(){
    g_id = 1;
    g_cid = 1;
    memset(id, 0, sizeof(id));
    memset(done, 0, sizeof(done));
    memset(cid, 0, sizeof(cid));
    memset(cid_in, 0, sizeof(cid_in));
    cid_group = vector(0, vector<int>());
    cid_group.push_back(vector<int>());
    graph = vector(100001, vector<int>());
}
int scc(int node){
    id[node] = g_id;
    g_id += 1;
    int parent = id[node];
    stk.push_back(node); 
    for(int next: graph[node]){
        if(id[next]==0)
            parent = min(parent, scc(next));
        else if(done[next]==0)
            parent = min(parent, id[next]);
    }
    if(parent == id[node]){
        vector<int> tmp;
        while(1){
            int cur = stk.back();
            stk.pop_back();
            cid[cur] = g_cid;
            done[cur] = 1;
            tmp.push_back(cur);
            if(cur == node)
                break;
        }
        cid_group.push_back(tmp);
        g_cid += 1;
    }
    return parent;
}
void solve(){
    cin >> n >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v; 
        graph[u].push_back(v);
    }
    for(int i = 0; i<n; i++){
        if(done[i]==0)
            scc(i);
    }
    
    for(int i = 0; i<n; i++){
        for(int next: graph[i]){
            if(cid[i] != cid[next]){
                cid_in[cid[next]] += 1;
            }
        }
    }
    
    vector<int> ans;
    int count = 0;
    for(int i = 1; i<g_cid; i++){
        if(cid_in[i]==0){
            count += 1;
            for(int elem: cid_group[i]){
                ans.push_back(elem);
            }
        }
    }
    sort(ans.begin(), ans.end());
    if(count!=1)
        cout << "Confused\n";
    else{
        for(int s: ans)
            cout << s << '\n';
    }
    cout << "\n";
   
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--){
        reset();
        solve(); 
    } 
    return 0;
}

boj 4196 도미노 와 같은 문제이다.

일단 각 정점이 포함된 가장 큰 사이클을 구해야 한다. 왜냐하면 사이클 내의 정점들은 서로에게 무조건 도달이 가능하기 때문에 이를 하나의 큰 정점으로 볼 수 있기 때문이다.

가장 큰 사이클은 scc 알고리즘을 통해서 구할 수 있다.

이 사이클들을 하나의 새로운 정점으로 생각하고, topological sort를 수행한다. 각 사이클의 in degree를 구한뒤, 0인 사이클의 개수를 확인한다. 2개 이상이라면 둘 중에 어떤 사이클에서 시작하더라도 다른 사이클에 도달 할 수가 없기 때문에 이는 Confused 에 해당한다. 1개인 경우 in degree가 0인 사이클 내의 모든 정점에서는 사이클 내의 다른 정점들에 도달이 가능하기 때문에, 사이클 내의 모든 정점을 출력해주면 된다.
