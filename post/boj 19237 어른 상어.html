#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>

#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, k;
struct box{
    int id;
    int left;
};
box prop[20][20] = {0,};
int state[20][20] = {0,};
int dir[401] = {0,}; // 1: up, 2: down, 3: left, 4: right
int prio[401][5][5] = {0,};
int shark_count = 0;
int ans = 0;

int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe() { return 0<=x&&x<n&&0<=y&&y<n; }
};

// return dir;
int next_pos(int id, int x, int y, int dir){
    int avail = 0; 
    pos cur = {x, y};
    for(int i = 0; i<4; i++){
        pos next = cur.adj(i);
        // printf("prop[%d][%d] - id: %d, left: %d\n", next.x, next.y, prop[next.x][next.y].id, prop[next.x][next.y].left);
        if(next.safe() && prop[next.x][next.y].left == 0){
            avail |= (1<<(i+1));
        }
    }
    if(avail > 0){
        for(int i = 1; i<=4; i++){
            if((avail & (1<<(prio[id][dir][i])))!=0){
                return prio[id][dir][i]; 
            }
        }
    }
    // printf("avail: %d\n", avail);

    avail = 0;
    cur = {x, y};
    for(int i = 0; i<4; i++){
        pos next = cur.adj(i);
        if(next.safe() && prop[next.x][next.y].id == id){
            avail |= (1<<(i+1));
        }
    }
    if(avail > 0){
        for(int i = 1; i<=4; i++){
            if((avail & (1<<(prio[id][dir][i])))!=0){
                return prio[id][dir][i]; 
            }
        }
    }
}
void print(){
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            cout << state[i][j] <<' ';
        }
        cout << '\n';
    }
}
void solve(){
    cin >> n >> m >> k; 
    shark_count = m;
    int t;
    for(int i = 0; i<n ;i++){
        for(int j = 0; j<n; j++){
            cin >> state[i][j];
            if(state[i][j]>0)
                prop[i][j] = {state[i][j], k};
        }
    }
    for(int i = 1; i<=m; i++){
        cin >> dir[i]; 
    }
    for(int k = 1; k<=m; k++){
        for(int i = 1; i <= 4; i++){
            for(int j = 1; j<=4; j++){
                cin >> prio[k][i][j];
            }
        }
    }

    while(shark_count > 1 && ans < 1000){
        // printf("ans: %d, shark_count: %d\n", ans, shark_count);
        ans += 1;   
        
        vector<pair<pos, int>> task;
        for(int i = 0; i<n; i++){
            for(int j = 0; j<n; j++){
                if(state[i][j]!=0){
                    // cout << i << ' ' << j << ": " << state[i][j] << "->" << dir[state[i][j]] << endl;
                    int next = next_pos(state[i][j], i, j, dir[state[i][j]]); 
                    task.push_back({{i, j}, next});
                    
                    /* int nx = i+dx[next-1];
                    int ny = j+dy[next-1];
                    if(prop[nx][ny].left == k+1){
                        if(state[i][j] > prop[nx][ny].id){
                            shark_count -= 1;
                            state[i][j] = 0;
                            continue;
                        }
                    } */
                }
            }
        }
        for(auto t: task){
            pos next = t.first.adj(t.second-1);
            int cur_id = state[t.first.x][t.first.y];
            if(state[next.x][next.y] != 0){
                shark_count -= 1;
                if(cur_id > prop[next.x][next.y].id){
                    state[t.first.x][t.first.y] = 0;
                    continue;
                }
            }
            state[next.x][next.y] = cur_id;
            state[t.first.x][t.first.y] = 0;
            dir[cur_id] = t.second;
            prop[next.x][next.y] = {cur_id, k+1};
        }
        // decrease time;
        for(int i = 0; i<n; i++)
            for(int j = 0; j<n; j++)
                if(prop[i][j].left > 0)
                    prop[i][j].left -= 1;
        
        // print();
    }
    if(shark_count > 1)
        cout << -1;
    else
        cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

구현 문제이다. 매 초마다 상어들을 직접 옮겨야 한다.

먼저 모든 상어가 이동해야 하는 위치를 방향과 우선순위를 고려하여 구한다. 이후 상어들을 옮겨주고 필요한 값들을 갱신해준다.

이렇게 두 단계로 해준 이유는, 상어 하나씩 옮기고 갱신을 해버리면 다른 상어가 이동해야 하는 위치를 정할 때 영향을 줄 수 도 있기 때문이다.
