#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int c;
struct coord{
    double x, y;
    bool operator<(const coord& t) const{
        if(y!=t.y)
            return y < t.y;
        else
            return x < t.x;
    }
    coord operator-(const coord& t) const{
        return {x-t.x, y-t.y};
    }
    coord operator+(const coord& t) const{
        return {x+t.x, y+t.y};
    }
    void print(){
        printf("[%lf, %lf]\n", x, y);
    }
};
coord points[100000] = {0,};
/*
 * u = (p2.x-p1.x, p2.y-p1.y, 0);
 * v = (p3.x-p1.x, p3.y-p1.y, 0);
 * u x v = (0, 0, (p2.x-p1.x)*(p3.y-p1.y)-(p2.y-p1.y)*(p3.x-p1.x));
 */
double ccw(coord p1, coord p2, coord p3){
    return (p2.x-p1.x)*(p3.y-p1.y)-(p2.y-p1.y)*(p3.x-p1.x);
}
struct angle_sort{
    coord cri;
    bool operator()(coord& a, coord& b){
        double cp = ccw(cri, a, b);
        if(cp!=0)
            return cp > 0;
        else
            return abs(a.x-cri.x)+abs(a.y-cri.y) < abs(b.x-cri.x)+abs(b.y-cri.y);
    }
};
double dist(coord a, coord b){
    return sqrt(pow(a.x-b.x, 2)+pow(a.y-b.y, 2));
}
void solve(){
    cin >> c;
    for(int i = 0; i<c; i++){
        cin >> points[i].x >> points[i].y;
    }
        
    sort(points, points+c); 
    angle_sort as = {points[0]};
    sort(points+1, points+c, as);
    vector<coord> stk = {points[0], points[1]};
    for(int i = 2; i<c; i++){
        while(1){
            coord p1 = stk[stk.size()-2]; 
            coord p2 = stk[stk.size()-1];
            coord p3 = points[i];
            double cp = ccw(p1, p2, p3);
            if(cp>0){
                stk.push_back(p3);
                break;
            }
            else if(cp<0){
                stk.pop_back(); 
            }
            else {
                stk.pop_back();
                stk.push_back(p3);
                break;
            }
        }
    }
    int size = stk.size();
    int l_idx = 0;
    int r_idx = 1;
    double ans = -1;
    while(1){
        ans = max(ans, dist(stk[l_idx], stk[r_idx]));
        coord p3 = stk[(r_idx+1)%size] + (stk[l_idx] - stk[r_idx]);
        double cp = ccw(stk[l_idx], stk[(l_idx+1)%size], p3); 
        if(cp <= 0){
            // cross product value 0 case should be included in this if clause
            // to break when only two points exists
            l_idx = (l_idx+1)%size;
            if(l_idx==0)
                break;
        }
        else {
            r_idx = (r_idx+1)%size;
        }
    }
    printf("%lf", ans);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

가장 먼 화살 사이의 거리를 구해야 한다.

convex hull은 다른 모든 점들을 포함한다. 따라서, convex hull에 속하는 임의의 두 점이 가장 먼 거리가 될 것이기 때문에, 일단은 convex hull을 먼저 구해야 한다.

boj 1708 볼록 껍질 이 문제에서 convex hull을 구하는 graham scan 알고리즘을 확인할 수 있다.

convex hull에 속하는 점의 집합을 구했다면, 여기서 가장 먼 두 점 사이의 거리를 구해야 한다. 두 개의 점의 조합을 구해서 직접 계산을 해볼 수 있지만, convex hull에 포함될 수 있는 점이 최대 100000개 이므로 총 4999950000 번의 연산이 필요하여 시간 내 해결이 불가능 할 것으로 보인다. 하지만 문제의 다른 조건에서 화살의 좌표가 정수이고 절댓값은 1000을 넘지 않는다고 되어있다. 따라서 convex hull에 100000개의 점이 모두 포함 될 수가 없기 때문에, 모든 조합을 계산해도 시간 내 해결할 수 있는 가능성이 존재한다.

그렇지만 모든 조합을 계산하지 않고 빠르게 가장 먼 거리를 찾을 수 있는 rotating calipers라는 알고리즘이 존재한다.

rotating calipers 알고리즘에서는 다각형에서 시계 반대 방향을 보고 있는 두개의 vector를 시계 반대방향으로 이동하며 가장 먼 거리를 찾아낸다. 두개의 vector에 대해 외적을 해서 시계 반대 방향에 있다면 그 다음 점이 거 먼 거리가 될 가능성이 있는 것이고 시계 방향에 있다면 그 다음 점이 더 먼 거리가 될 가능성이 없는 것임을 이용하는 것이다.

https://stonejjun.tistory.com/42 이 글을 보고 알고리즘을 이해할 수 있었다.

느낀점

rotating calipers 문제는, convex hull 문제만이 아닌 다각형과 관련된 다른 문제에서도(다각형 내 두 점의 가장 먼 거리를 찾는 등) 쓰일 수 있을 것 같다.
