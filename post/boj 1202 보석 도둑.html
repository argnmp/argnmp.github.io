풀이

첫번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, k;
struct box{
    int m, v;
    bool operator<(const box& t) const {
        return v > t.v;         
    } 
};
void solve(){
    cin >> n >> k;
    vector<box> jewels;
    vector<int> bags;
    int m, v;
    for(int i = 0; i<n; i++){
        cin >> m >> v;
        jewels.push_back({m, v});
    }
    sort(jewels.begin(), jewels.end());
    /* for(box t: jewels){
        cout << t.v << ' ' << t.m << endl;
    } */
    int t;
    for(int i = 0; i<k; i++){
        cin >> t; 
        bags.push_back(t);
    }
    sort(bags.begin(), bags.end());
    vector<int> bag_unique;
    vector<int> bag_count;

    int prev_bag = -1;
    for(int i = 0; i<k; i++){
        if(prev_bag == bags[i]){
            bag_count[bag_count.size()-1] += 1;
        }
        else {
            prev_bag = bags[i];
            bag_unique.push_back(prev_bag);
            bag_count.push_back(1);
        }
    }
    /* for(auto i: bag_unique){
        cout << i << ' ';
    }
    cout << endl;
    for(auto i: bag_count){
        cout << i << ' ';
    }
    cout << endl; */

    vector<int> bag_jumper(bag_unique.size(), -1);
    ll ans = 0;
    for(int i = 0; i<n; i++){
        box cur = jewels[i]; 
        int idx = lower_bound(bag_unique.begin(), bag_unique.end(), cur.m) - bag_unique.begin();
        if(idx != bag_unique.size()){
            int org_idx = idx;
            if(bag_jumper[org_idx]!=-1){
                idx = bag_jumper[org_idx];
            }
            while(idx < bag_count.size() && bag_count[idx]==0){
                idx += 1;
            }
            bag_jumper[org_idx] = idx;
        }
        if(idx==bag_count.size())
            continue;
        bag_count[idx] -= 1;
        ans += cur.v;
    }
    cout << ans << endl;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

이 문제는 가격이 높은 보석 부터 차례차례 가능한 한 최대 무게가 작은 가방에 넣으면 된다. 최대 보석도 300000개이고 가방도 300000이어서 그냥 무작정 탐색을 하면 시간 초과에 걸린다.

따라서 다음과 같은 방식으로 빠르게 탐색할 수 있도록 하였다.

  1. 가방의 무게를 오름차순으로 정렬한다.
  2. bag_unique에는 unique한 가방 무게만 남겨두고, 해당 무게의 가방의 개수는 bag_count에 저장한다.
  3. 이분 탐색을 통해서 보석의 무게 이상인 가방 중 가장 작은 것을 찾고, 만약 bag_count가 1개 이상이면 해당 가방을 쓸 수 있는 것이므로 개수를 줄인다.
  4. bag_count가 0이라면 index를 늘려가면서 bag_count가 1이상인 가방을 찾아서 개수를 줄인다. 만약 찾지 못한다면 보석을 넣을 수 없는 것이다.

말하자면 hash와 같은 방식이다.

하지만 이 방식 만으로는 시간 초과가 여전히 발생한다. 왜냐하면 bag_count가 0인 경우 index를 계속 늘려가야 해서 최악의 경우 O(n^2)의 시간이 소요되기 때문이다.

따라서 bag_jumper를 추가해서, bag_count가 0일 때 index를 늘려가는데 이 때 마지막 위치를 bag_jumper에 저장해서 그 자리에 방문 했을 때 바로 이동할 수 있도록 한다.

두번째 풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, k;
struct box{
    int m, v;
    bool operator<(const box& t) const {
        return m < t.m;
    }
    bool operator()(box& a, box& b){
        return a.v < b.v;
    }
};
vector<box> jewels;
vector<int> bags;
void solve(){
    cin >> n >> k;
    int m, v;
    for(int i = 0; i<n; i++){
        cin >> m >> v;
        jewels.push_back({m, v}); 
    }
    for(int i = 0; i<k; i++){
        cin >> m;
        bags.push_back(m);
    }
    sort(jewels.begin(), jewels.end());
    sort(bags.begin(), bags.end());
    priority_queue<box, vector<box>, box> pq;
    ll ans = 0;
    int idx = 0;
    for(int i = 0; i<k; i++){
        while(idx < jewels.size() && jewels[idx].m <= bags[i]){
            pq.push(jewels[idx]); 
            idx += 1;
        } 
        if(!pq.empty()){
            box highest = pq.top();
            pq.pop();
            ans += highest.v;
        }
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

보석을 무게 기준으로 오름차순으로 정렬한다. 그리고 가방도 오름차순으로 정렬한다. 이 상태에서 가방 전체를 순회하며 해당 가방에 들어갈 수 있는 보석을 모두 priority_queue(보석의 가격 기준으로 내림차순)에 넣는다. 만약 priority_queue가 비어있지 않으면 보석을 하나 꺼낸다.(가방에 보석을 넣는다)

이러한 방식이 첫번째 풀이보다 훨씬 간단하고 좋다. 첫번재 풀이와는 비슷한 시간이 소요되는 것 같다.

느낀점

첫번째 풀이로 문제를 풀었지만, 두번째 풀이와 같이 명료하게 푸는 것을 조금 더 고민해야겠다.
