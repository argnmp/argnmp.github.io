코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int r, c;
char board[25][25];
int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};
struct pos {
    int x, y;
    void print(){
        printf("[%d,%d]\n",x,y);
    }
    bool safe(){
        return 0<=x&&x<r&&0<=y&&y<c;
    }
    pos check(char ch){
        vector<pos> n;
        switch(ch){
            case 'M':
                return pos {-1, -1};
                break;
            case 'Z':
                return pos {-1, -1};
                break;
            case '|':
                n.push_back({x-1, y});
                n.push_back({x+1, y});
                break;
            case '-':
                n.push_back({x, y-1});
                n.push_back({x, y+1});
                break;
            case '+':
                n.push_back({x-1, y});
                n.push_back({x+1, y});
                n.push_back({x, y-1});
                n.push_back({x, y+1});
                break;
            case '1':
                n.push_back({x, y+1});
                n.push_back({x+1, y});
                break;
            case '2':
                n.push_back({x-1, y});
                n.push_back({x, y+1});
                break;
            case '3':
                n.push_back({x-1, y});
                n.push_back({x, y-1});
                break;
            case '4':
                n.push_back({x, y-1});
                n.push_back({x+1, y});
                break;
        }
        for(pos t: n){
            if(t.safe()){
                if(board[t.x][t.y]=='.')
                    return t;
            }
        }
        return pos {-1, -1};
    }
    pos next(int i){
        return pos {x + dx[i%4],y + dy [i%4]}; 
    }
};
char mapper(bitset<4> b){
    switch(b.to_ulong()){
        case 5:
            return '|';
        case 10:
            return '-';
        case 15:
            return '+';
        case 6: 
            return '1';
        case 3:
            return '2';
        case 9:
            return '3';
        case 12:
            return '4';
    }
}
void solve(){
    cin >> r >> c;
    for(int i = 0; i<r; i++){
        for(int j = 0; j<c; j++){
            char ch = cin.get();
            while(ch=='\n'||ch==' '||ch=='\r'){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    }
    bool flag = true;
    pos target;
    for(int i = 0; i<r && flag; i++){
        for(int j = 0; j<c && flag; j++){
            if(board[i][j]!='.'){
                pos cur = {i, j}; 
                pos res = cur.check(board[i][j]);
                if(!(res.x==-1&&res.y==-1)){
                    target = res;       
                    target.print();
                    flag = false;
                }
            }
        }
    }
    
    bitset<4> b;
    for(int i = 0; i<4; i++){
        pos n = target.next(i);
        if(n.safe() && board[n.x][n.y]!='.'){
            pos res = n.check(board[n.x][n.y]);
            if(res.x==target.x && res.y==target.y)
                b.set(i);
        }
    }
    
    cout << target.x + 1 << ' ' << target.y + 1 << ' ' << mapper(b);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • 최대 625 칸이기 때문에, 전체 탐색을 진행한다.
  • 진행하는 과정에서 배관이 나오면, 주변에 필요한 배관들이 모두 연결이 되어 있는지 확인한다. 연결이 안되어 있는 좌표가 있으면 탐색을 중단한다. 왜냐하면 문제에서 배관 블록 하나만 지웠다고 되어 있기 때문이다.
  • 지운 배관 블록의 좌표를 찾으면, 상하좌우의 배관을 보면서, 필요한 배관의 모양을 판단한다.
