#include <iostream>
#include <algorithm>
#include <climits>

#include <queue>

#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
string target;
vector<int> kmp_pi(string p){
    int m = p.size(), j = 0;
    vector<int> pi(m, 0);
    for(int i = 1; i<m; i++){
        while(j>0 && p[i]!=p[j]){
            j = pi[j-1];
        }
        if(p[i]==p[j]){
            pi[i] = ++j;
        }
    }
    return pi;
}
void solve(){
    vector<int> pi = kmp_pi(target);
    int diff = target.size() - pi[target.size()-1];
    if(pi[target.size()-1]%diff!=0)
        cout << 1 << '\n';
    else
        cout << pi[target.size()-1]/diff + 1 << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    while(1){
        cin >> target;
        if(target==".")
            break;
        solve(); 
    }
    return 0;
}

kmp 알고리즘의 pi 배열을 이용해서 풀 수 있다.

문제에서 요구하는 바는, 특정 문자열이 최대로 반복되는 횟수를 찾는 것이다.

abababab

위의 문자열의 전체 구간에서, 접두사와 접미사가 일치하는 최대 길이는 6으로, index 0~5와 index 2~7이 일치하는 구간이다. 접두사와 접미사가 일치하는 부분을 index와 함께 표현하면 아래와 같다.

	0 1 2 3 4 5 6 7
A:	a b a b a b a b
	B:	a b a b a b a b
	    0 1 2 3 4 5 6 7

다시한번, 최대 길이가 6이므로 index 2~7의 구간은 index 0~5의 구간과 무조건 일치한다.

그렇다면 일치하는 접미사 구간을 제외한 구간인, index 0~1의 구간을 보자. 편의상 위에 적은 문자열을 A라 하고, 아래에 적은 문자열을 B라고 한다. (물론 이 둘은 같은 문자열이다.)

A에서 index 0~1은 B에서 index 0~1이고, 이는 다시 A에서 index 2~3과 같아야 한다. 반복적으로 A에서 index 2~3은 B에서 index 2~3이고, 이는 다시 A에서 index 4~5와 같아야 한다. 이러한 반복이 index 6~7까지 이어진다.

즉, 접미사 부분을 제외한 앞 부분은 반복되는 구간이라고 할 수 있다. 아래와 같은 경우를 생각해보자.

0 1 2 3 4
a b a b a
    a b a b a
    0 1 2 3 4

여기에서는 접두사와 접미사가 일치하는 최대 길이는 3이다. 접미사를 제외한 index 0~1구간은 위에서 살펴본 것과 같이 반복되는 구간이고, 일치하는 최대 길이가 3이므로 마찬가지로 반복해서 따라가다보면 문자가 하나만 남기 때문에 더이상 반복이 불가능하다. 따라서 이 경우는 index 0~1이 계속 반복된다고 할 수 없으니, 최대 반복횟수는 1이라고 할 수 있다.

pi배열을 구할 때, 최대 길이를 구하기 때문에 반복되는 문자의 길이는 가능한 한 최소가 되어서, 문제에서 원하는 가장 큰 반복 횟수를 찾을 수 있다.

결론적으로 pi배열을 구해서 전체 구간에서 최대의 일치하는 접두사/접미사의 길이를 찾고, 이를 반복되는 구간(전체 구간에서 접미사를 제외한 구간)으로 나누어보아서 나누어 떨어지는 경우 반복되는 횟수를 출력하면 된다. 만약 나누어 떨어지지 않는 경우는 1이 된다.

느낀점
