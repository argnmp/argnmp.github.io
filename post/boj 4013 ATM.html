풀이

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m, s, p;
vector<vector<int>> graph(500001, vector<int>());
int atm[500001] = {0, }, is_dest[500001] = {0,}, gid = 1, graph_id[500001] = {0,}, done[500001] = {0,};
int cid = 1, graph_cid[500001] = {0,}, cid_in[500001] = {0,};
ll cid_atm[500001] = {0,}, cid_is_valid[500001] = {0,}, cid_is_dest[500001] = {0,}, cid_max_atm[500001] = {0,};
vector<int> stk;
vector<vector<int>> cgraph(500001, vector<int>());

int scc(int node){
    graph_id[node] = gid;
    gid += 1;
    int parent = graph_id[node];
    stk.push_back(node);
    for(int next: graph[node]){
        if(graph_id[next]==0)
            parent = min(parent, scc(next));
        else if(done[next]==0)
            parent = min(parent, graph_id[next]);
    }
    if(parent == graph_id[node]){
        int atm_sum = 0;
        int dest = 0;
        while(1){
            int cur = stk.back();
            stk.pop_back();
            done[cur] = 1;
            graph_cid[cur] = cid;      
            atm_sum += atm[cur];
            if(is_dest[cur]==1)
                dest = 1;
            if(cur == node)
                break;
        }
        cid_atm[cid] = atm_sum; 
        cid_is_dest[cid] = dest;
        cid += 1;
    }
    return parent;
}

void find_valid_route(int cid){
    done[cid] = 1;
    cid_is_valid[cid] = 1;
    cid_in[cid] = 0;
    for(int next: cgraph[cid]){
        if(done[next]==0)
            find_valid_route(next);
    }
}

void solve(){
    cin >> n >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v;
        graph[u].push_back(v);
    }
    for(int i = 1; i<=n; i++){
        cin >> atm[i];
    }
    cin >> s >> p;
    for(int i = 0; i<p; i++){
        cin >> u;
        is_dest[u] = 1;
    }
    for(int i = 1; i<=n; i++)
        if(done[i]==0)
            scc(i);

    for(int i = 1; i<=n; i++){
        for(int next: graph[i]){
            if(graph_cid[next] != graph_cid[i]){
                cgraph[graph_cid[i]].push_back(graph_cid[next]);
            }
        }
    }

    memset(done, 0, sizeof(done));
    memset(cid_in, -1, sizeof(cid_in));
    int target_cid = graph_cid[s];
    find_valid_route(target_cid);
    
    for(int i = 1; i<cid; i++){
        for(int next: cgraph[i]){
            if(cid_is_valid[i]==1 && cid_is_valid[next]==1){
                cid_in[next] += 1;
            }
        }
    }
    queue<int> q;
    for(int i = 1; i<cid; i++){
        if(cid_in[i]==0){
            q.push(i);
        }
    }
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        for(int next: cgraph[cur]){
            cid_max_atm[next] = max(cid_max_atm[next], cid_max_atm[cur]+cid_atm[cur]);
            cid_in[next] -= 1;
            if(cid_in[next]==0)
                q.push(next);
        }    
    }
    ll ans = -1; 
    for(int i = 1; i<cid; i++){
        if(cid_is_dest[i]==1 && cid_is_valid[i]==1){
            ans = max(ans, cid_max_atm[i] + cid_atm[i]);
        }
    }
    printf("%lld", ans);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제를 정리하면 다음과 같다.

  • 각 정점에서는 처음 방문할 때, 현금을 모두 인출한다.
  • 일부 정점이 도착지가 될 수 있고, 시작 정점 부터 한 도착지까지 이동하며 거쳐가는 정점에서 인출하는 현금의 최대값을 구해야 한다.
  • 이동 시 동일한 도로나 교차로를 여러번 지날 수 있다.

동일한 도로나 교차로를 여러번 지날 수 있기 때문에, 사이클 내에 있는 모든 정점을 지나도 문제가 되지 않는다. 현금 인출을 최대로 하기 위해서는 사이클에 도착하면 무조건 사이클 내 정점을 모두 지나가야한다. 또한 사이클 내의 정점들 중, 어떤 정점로 들어와서 어떤 정점으로 나가든 상관없다.

이러한 특성 때문에, 가장 큰 사이클을 찾아서 하나의 정점으로 묶어주어야 한다. 이를 위해서는 scc 알고리즘을 사용해야 한다.

scc 알고리즘을 그래프에 적용하면 최대 크기의 사이클이 하나의 정점이 된 DAG(directed acyclic graph) 가 생성된다. 사이클이 모두 사라지기 때문이다. 사이클 내에서는 모든 정점에 도달이 가능한 점을 고려하여 scc 알고리즘을 적용한 새로운 그래프의 정점에는, 사이클에 포함된 모든 정점의 현금의 최대값을 저장하고 사이클에 포함된 정점 중 하나가 도착지라면 새로운 그래프의 정점도 도착지로 설정한다.

새로운 그래프에서 시작지점에서 도착지로 이동하며 경로에 있는 정점의 현금의 합의 최대값을 찾아야 한다. 이를 찾기 위해서는 모든 경로를 탐색하는 것이 아닌, boj 1005 ACM Craft 에서 처럼 topological sort를 사용할 수 있다. in degree가 0인 정점부터 탐색하며, 각 정점까지의 현금의 합의 최대값을 계속 갱신해주면 된다.

한 가지 더 고려할 사항이 있다. topological sort에서는 보통 in degree가 0인 모든 정점에서 시작한다. 하지만 이 문제에서는 시작지점이 존재한다. 한 정점에서 시작하기 위해 다음과 같은 방법을 사용한다.

  • dfs를 통해서 시작지점에서 도달 가능한 모든 정점을 표시하고, 이 정점들에만 in degree 값을 0으로 세팅해준다. (새로운 그래프의 이외의 정점들에는 in degree 값을 -1로 한다.)
  • 도달 가능한 정점의 in degree 값을 구하기 위해서 모든 directed edge를 탐색하는데, 이때 directed edge의 양쪽 끝 정점이 모두 위의 도달 가능한 정점에 해당하는 경우만, in degree 값을 늘려준다.
  • in degree가 0인 모든 정점에서 시작하여, 각 정점에서의 현금의 합의 최대값을 계속 갱신해준다.
  • 이후 새로운 그래프에서, 각 도달 가능한 정점인 동시에 도착지인 정점 중 가장 최대 값을 출력한다. (새로운 그래프의 모든 정점이 아닌, 도달 가능한 정점의 값만을 비교하는 이유는, 도달 가능하지 않은 정점의 현금 값이 매우 크다면 이 정점이 가장 최대 값이 될 수도 있기 때문이다.)

이러한 방식으로 답을 구할 수 있다.

추가로 topological sort를 이용하지 않고, bfs로 시작지점 부터 최대 값이 갱신되는 경우에만 재탐색이 가능하도록 하는 방식으로 최대 값을 구할 수 있다.

느낀점

boj 4196 도미노 boj 3977 축구 전술 을 푼 이후에 풀게 된 문제여서, scc 알고리즘을 활용해야 한다는 것을 빠르게 파악할 수 있었다. 하지만 최대값을 구하는 과정에서 많은 고민을 하였다. 이전에 풀었던 boj 1005 ACM Craft 를 통해서 topological sort를 이용할 수 있다는 생각읋 했지만, dfs를 통해서 도달 가능한 정점을 먼저 확인한 뒤 특정 지점에서 topological sort를 할 수 있다는 것을 파악하기까지는 시간이 걸렸다.

scc 알고리즘은 dag 를 만들어내고, dag 에서는 topological sort를 적용할 수 있으며, dfs를 통해 특정 지점에서 시작하는 topological sort가 가능하다.
