첫번째 풀이

오랜시간 고민을 한 끝에, scc와 union-find를 활용하여 문제를 풀 수 있었다.

아래는 코드이다.

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
vector<vector<int>> graph(100001, vector<int>());

vector<int> stk;
int gid = 1;
int id[100001] = {0,};
int done[100001] = {0,};
int cid = 1;
int cycle[100001] = {0,};
int cycle_used[100001] = {0,};
int scc(int node){
    id[node] = gid; 
    gid += 1;
    stk.push_back(node);
    int parent = id[node];
    for(int next: graph[node]){
        if(id[next]==0){
            parent = min(parent, scc(next));
        }
        else if(done[next]==0){
            parent = min(parent, id[next]);
        }
    }
    if(id[node]==parent){
        while(1){
            int cur = stk.back();
            stk.pop_back();
            cycle[cur] = cid;
            done[cur] = 1;
            if(cur==node)
                break;
        } 
        cid += 1;
    }
    return parent;
}

int uf[100001] = {0};
int uf_find(int node){
    if(uf[node]==node)
        return node;
    return uf[node] = uf_find(uf[node]);
}
void uf_union(int a, int b){
    int ap = uf_find(a);
    int bp = uf_find(b);
    if(ap!=bp){
        uf[bp] = ap;
    }
}

void solve(){
    graph = vector(100001, vector<int>());
    gid = 1;
    memset(id, 0, sizeof(id));
    memset(done, 0, sizeof(done));
    cid = 1;
    memset(cycle, 0, sizeof(cycle));
    memset(cycle_used, 0, sizeof(cycle_used));
    cin >> n >> m;
    int u, v;
    for(int i = 0; i<m; i++){
        cin >> u >> v; 
        graph[u].push_back(v);
    }
    for(int i = 1; i<=n; i++){
        if(done[i]==0)

        uf[i] = i;

        for(int next: graph[i]){

                cycle_used[cycle[next]] = 1;
                uf_union(cycle[i], cycle[next]);
            }

    for(int i = 1; i<cid; i++){

문제를 읽고 처음에는 dfs를 돌며 연결된 다음 정점들과 이전 정점을 같은 집합으로 묶어주면 된다고 생각했다.

사실은 dfs로 경로를 탐색할 필요가 없다.(이는 나중에 떠오른 생각이다.) union-find를 사용하기 때문에 각 정점을 한번씩만 돌면서 연결된 정점을 같은 집합으로 묶어주어서 같은 집합은 한번에 넘어뜨릴 수 있다고 판단하는 방식으로 문제를 푼다고 생각해보자.

하지만 다음과 같은 경우를 고려해보면 그렇게 단순하지 않다.

각 정점에 연결된 정점을 같은 집합으로 묶어주면 위 그래프의 모든 정점은 같은 집합에 속하게 된다. 하지만 한번 쓰러진 도미노는 다시 쓰러뜨릴 수 없기 때문에, 정점 4에서 도미노를 넘어뜨렸다면 정점 5와 6은 각각 넘어뜨려야 하므로 다른 집합에 속해야만 한다. 따라서 각 정점은 단 한번씩만 같은 집합으로 묶어주어야 한다.

그리고 정점을 넘어뜨리는 순서도 문제가 된다. 현재 정점 1, 2, 3은 사이클을 그리고 있다. 사이클 내의 정점들 중 하나를 먼저 넘어뜨리고 다른 정점을 넘어뜨리면, 총 4번이 필요하다. 하지만 사이클 외의 정점들인 4, 5, 6 중 하나를 먼저 넘어뜨리면, 총 3번으로도 가능하다. 그렇기 때문에 사이클 외의 정점들을 먼저 넘어뜨려야 한다.

또한 이러한 그래프도 고려해보았을 때, 결국 가장 큰 사이클을 찾아내서 사이클 밖의 정점을 먼저 넘어뜨려야 함을 알 수 있다. 따라서 scc 알고리즘이 필요하다.

위의 조건들을 종합해보면 정점을 넘어뜨려야 하는 순서가 꽤 복잡하다. 하지만 약간 다르게 생각해보면 단순하게 만들 수 있다.

각 정점은 한번만 같은 집합으로 묶어주어야 하고, 가장 큰 사이클 밖에 있는 정점을 먼저 넘어뜨려야 했었다. 이는 다음과 같은 알고리즘으로 바꿀 수 있다.

  • 정점을 집합으로 묶었는지 여부를 나타내는 배열을 준비한다.
  • 각 정점을 순회하면서 연결된 정점들을, 집합으로 묶이지 않은 경우 union-find로 같은 집합으로 묶고 위 배열에 묶였다고 표시를 해준다.
  • 만약 같은 사이클에 해당하는 정점인 경우 위 배열에 묶였다고 표시를 하지 않는다

이 알고리즘을 통해서 앞선 조건들을 모두 만족하는 방식으로 집합의 형성이 가능하다. 이후 집합의 개수를 출력해주면 된다.

위의 코드에서는 먼저 tarjan 알고리즘을 통해 모든 사이클을 찾고, 이 사이클에 unique id를 부여하여 각 정점에 id를 표시해둔다. (사이클이 없는 한 개의 정점도, unique id를 가지도록 한다) 사이클 id가 집합으로 묶였는지 여부를 저장하는 배열을 만든다. 이후 각 정점을 순회하면서 연결된 정점의 사이클 id가 집합으로 묶이지 않은 경우, 두 정점을 union-find로 같은 집합으로 묶는다. 만약 묶은 두 정점의 사이클 id가 같다면 집합으로 묶였는지 여부를 저장하는 배열에 묶였다고 표시하지 않고, 그 이외의 경우에는 표시한다. 마지막으로 union-find로 묶은 정점의 집합 개수를 출력해주면 된다.

두번째 풀이

문제를 풀고 다른 풀이를 검색해보니, scc로 만든 사이클을 topological sort를 통해 in degree가 0인 사이클의 개수를 출력해주는 방법이 있다.

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n, m;
vector<vector<int>> graph(100001, vector<int>());

int gid = 1;
int id[100001] = {0,};
int cid = 1, cycle[100001] = {0,}, cid_in[100001] = {0,}; // 1~(cid-1)
int done[100001] = {0,};
vector<int> stk;

int scc(int node){
    stk.push_back(node);
    id[node] = gid;
    gid += 1;
    int parent = id[node];
    for(int next: graph[node]){
        if(id[next]==0)
            parent = min(parent, scc(next));
        else if(done[next]==0)
            parent = min(parent, id[next]);
    }
    if(parent == id[node]){
        while(1){
            int cur = stk.back();
            stk.pop_back();
            done[cur] = 1;
            cycle[cur] = cid;
            if(cur == node)
                break;
        } 
        cid += 1;
    }
    return parent;
}
void solve(){
    graph = vector(100001, vector<int>());
    gid = 1;
    memset(id, 0, sizeof(id));
    memset(done, 0, sizeof(done));
    cid = 1;
    memset(cycle, 0, sizeof(cycle));
    memset(cid_in, 0, sizeof(cid_in));
    cin >> n >> m;
    int u, v;
    for(int i=0; i<m; i++){
        cin >> u >> v; 
        graph[u].push_back(v);
    }
    for(int i=1; i<=n; i++)
        if(done[i]==0)
            scc(i);

    for(int i=1; i<=n; i++){
        for(int next: graph[i]){
            if(cycle[i] != cycle[next]){
                cid_in[cycle[next]] += 1;
            }
        }
    }
    int ans = 0;
    for(int i = 1; i<=cid-1; i++){
        if(cid_in[i]==0)
            ans += 1;
    }
    cout << ans << '\n';
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--)
        solve(); 
    return 0;
}

사이클 내의 정점들은 어떤 정점을 넘어뜨려도, 모든 정점들이 넘어지게 된다. 따라서 scc로 구한 사이클을 한 단위로 해서, in degree가 0인 사이클의 개수를 구한다. in degree가 1 이상인 사이클은 in degree가 0인 사이클에 의하여 무조건 넘어지기 때문이다.
