#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
int board[20][20] = {0,};
int sum = 0;
// up right down left
int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){
        return 0<=x&&x<n&&0<=y&&y<n;
    }
    void print(){
        printf("[%d, %d]\n", x, y);
    }
};

int ans = INT_MAX;

void calc(int x, int y, int d1, int d2){
    pos a = {x, y};
    pos b = {x+d1, y-d1};
    pos c = {x+d2, y+d2};
    pos d = {x+d1+d2, y-d1+d2};

    // sector 1
    int sec1 = 0;
    int lim1 = a.x-1;
    for(int j = a.y; j>=b.y; j--){
        for(int i = 0; i<=lim1; i++){
            sec1 += board[i][j]; 
        }
        lim1 += 1;
    }
    for(int j = b.y-1; j>=0; j--){
        for(int i = 0; i<=b.x-1; i++){
            sec1 += board[i][j];
        }
    }

    // sector 2
    int sec2 = 0;
    int lim2 = c.y+1; 
    for(int i = c.x; i>=a.x; i--){
        for(int j = n-1; j>=lim2; j--){
            sec2 += board[i][j];  
        }
        lim2 -= 1;
    }
    for(int i = a.x-1; i>=0; i--){
        for(int j = n-1; j>=a.y+1; j--){
            sec2 += board[i][j];
        }
    } 

    // sector 3
    int sec3 = 0;
    int lim3 = d.x+1;
    for(int j = d.y; j<=c.y; j++){
        for(int i = n-1; i>=lim3; i--){
            sec3 += board[i][j];
        }
        lim3 -= 1;
    }
    for(int j = c.y+1; j<=n-1; j++){
        for(int i = n-1; i>=c.x+1; i--){
            sec3 += board[i][j];
        }
    }

    // section 4
    int sec4 = 0;
    int lim4 = b.y-1;
    for(int i = b.x; i<=d.x; i++){
        for(int j = 0; j<=lim4; j++){
            sec4 += board[i][j];
        }
        lim4 += 1;
    }
    for(int i = d.x+1; i<=n-1; i++){
        for(int j = 0; j<=d.y-1; j++){
            sec4 += board[i][j];
        }
    }

    int sec5 = sum - (sec1 + sec2 + sec3 + sec4);
    vector<int> acc = {sec1, sec2, sec3, sec4, sec5};
    sort(acc.begin(), acc.end());
    ans = min(ans, acc[4]-acc[0]);
}
void solve(){
    cin >> n;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            cin >> board[i][j];
            sum += board[i][j];
        }
    }
    for(int x = 0; x<n; x++){
        for(int y = 0; y<n; y++){
            for(int d1 = 1; x+d1<n && y-d1>=0; d1++){
                for(int d2 = 1; x+d2<n && y+d2<n; d2++){
                    calc(x, y, d1, d2); 
                }
            }
        }
    }
    cout << ans;
    // calc(2, 2, 1, 1);
     
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

백준에 있는 예제이다. 파란색으로 나누고 다시 빨간색과 주황색으로 나누어 계산해주면 된다.

느낀점

처음에 전체 경우의 수를 탐색하면 시간 초과가 발생할 것이라 생각했지만, 검색해보니 전체 경우의 수를 탐색하는 문제였다. 가능할 것 같아보이는 풀이라면 시도해보자.
