코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n, m, from, to;
struct edge {
    int v;
    int w;
    vector<int> route;
    void print(){
        printf("[%d,%d]\n", v, w);
    }
};
vector<vector<edge>> graph(1001, vector<edge>());
struct compare {
    bool operator()(edge& a,edge& b){
        return a.w > b.w;
    }
};

void solve(){
    cin >> n >> m; 
    int u, v, w;
    for(int i=0; i<m; i++){
        cin >> u >> v >> w;
        graph[u].push_back({v,w});
    }
    cin >> from >> to;
    
    int done[1001] = {0,};
    // this [edge] means the total route cost to that vertex
    priority_queue<edge, vector<edge>, compare> pq;
    pq.push({from, 0}); 
    while(!pq.empty()){
        edge r = pq.top();
        pq.pop();
        if(done[r.v]==1) continue;
        done[r.v] = 1;
        if(r.v == to){
            // found destination
            cout << r.w;
            return;
        }
        for(edge n: graph[r.v]){
            if(done[n.v]==0)
                pq.push({n.v, r.w + n.w});
        }
    }
    return;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

  • dijkstra 알고리즘을 사용하면 된다.
  • edge struct의 weight을 현재까지의 route의 비용 으로 생각하여 priority queue에 그대로 넣을 수 있다.
    • 그래프
  • done 배열을 사용해서 해당 vertex까지의 최소비용이 나왔는지 여부를 설정할 수 있다.
    • queue에서 pop을 했을 때 done을 확인한다
    • 인접한 vertex를 탐색할 때 done을 확인한다
  • done 배열의 갱신은, priority queue에서 pop을 한 다음에 이루어져야 한다!!! BFS의 경우에는 탐색하는 과정에서 done(visitied) 배열을 갱신한다.

느낀점

  • 예전에 풀었던 문제이다. dijkstra가 기억이 안나서 예전 코드를 보았다.
  • 따로 배열을 설정하지 않아도 edge struct를 그대로 유지해서 dijkstra 결과를 얻을 수 있다.
