풀이

#include <iostream>
	#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
#define dbgn(x) printf("[DBG] %s: %lld\n", #x, (ll)x);
using namespace std;
using ll = long long;

int n, m;
/*
 * 7 means empty
 */
int board[20][20] = {0,};
int dx[4] = {0,0,-1,1};
int dy[4] = {-1,1,0,0};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){ return 0<=x&&x<n&&0<=y&&y<n; }
    void print(){
        printf("[%d, %d]", x, y);
    }
};
struct block{
    pos p;
    int color;
    void print(){
        p.print();
        cout << ": " << color << '\n';
    }
    bool operator<(const block& t) const {
        if(color == t.color){
            if(p.x == t.p.x){
                return p.y < t.p.y;
            }
            return p.x < t.p.x;
        }
        return color > t.color;
    }
    
};
struct block_group{
    // first block is standard block
    vector<block> blocks;
    int rainbow_count;
    bool operator<(const block_group& t) const {
        if(blocks.size() == t.blocks.size()){
            if(rainbow_count == t.rainbow_count){
                if(blocks[0].p.x == t.blocks[0].p.x){
                    return blocks[0].p.y > t.blocks[0].p.y;
                }
                return blocks[0].p.x > t.blocks[0].p.x;
            }
            return rainbow_count > t.rainbow_count;
        }
        return blocks.size() > t.blocks.size();
    }
    void print(){
        for(block& b: blocks){
            b.print();
        }
    }
};

void print(){
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            if(board[i][j]==7){
                printf(" .");
                continue;

            }
            printf("%2d", board[i][j]);
        }
        cout << endl;
    }
}
void gravity(){
    for(int j = 0; j<n; j++){
        for(int i = n-1; i>=0; i--){
            if(board[i][j]==7){
                for(int k = i-1; k>=0; k--){
                    if(board[k][j]==-1){
                        break; 
                    }
                    else if(board[k][j]>=0 && board[k][j]!=7){
                        board[i][j] = board[k][j];
                        board[k][j] = 7;
                        break;
                    }
                }
            }
        }
    }
}

int ans = 0;

block_group get_block_group(pos s, vector<vector<int>>& visited){
    int color = board[s.x][s.y];

    block_group grp;
    grp.blocks.push_back({s, color});
    
    visited[s.x][s.y] = 1; 
    queue<pos> q;
    q.push(s);
    while(!q.empty()){
        pos cur = q.front();
        q.pop();
        for(int i = 0; i<4; i++){
            pos next = cur.adj(i); 
            if(next.safe() && board[next.x][next.y]!=-1 && board[next.x][next.y]!=7 && visited[next.x][next.y]==0){
                if(board[next.x][next.y]==0 || board[next.x][next.y] == color){
                    visited[next.x][next.y] = 1; 
                    grp.blocks.push_back({next, board[next.x][next.y]}); 
                    q.push(next);
                }
            }
        }
    }
    sort(grp.blocks.begin(), grp.blocks.end());
    
    int count = 0;
    for(block& b: grp.blocks){
        if(b.color == 0){
            count += 1;
            visited[b.p.x][b.p.y] = 0;
        }
    }
    grp.rainbow_count = count;

    return grp;
}

bool simul(){
    vector<block_group> bg;
    vector<vector<int>> visited(20, vector<int>(20, 0));
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            if(board[i][j]>0 && board[i][j]!=7 && visited[i][j]==0){
                block_group grp = get_block_group({i, j}, visited);
                if(grp.blocks.size() >= 2)
                    bg.push_back(grp);
            }
        }
    }
    if(bg.size()==0)
        return false;
    
    sort(bg.begin(), bg.end());
    block_group& target = bg[0];
    for(block& b: target.blocks){
        board[b.p.x][b.p.y] = 7;
    }
    // target.print();
    ans += target.blocks.size() * target.blocks.size();
    /* cout << "remove" << endl;
    print();
    cout << endl; */
    gravity();
    /* cout << "grapvity" << endl;
    print();
    cout << endl; */
    int tmp[20][20] = {0,};
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            tmp[n-j-1][i] = board[i][j]; 
        }
    }
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            board[i][j] = tmp[i][j]; 
        }
    }
    /* cout << "rotate" << endl;
    print();
    cout << endl; */
    gravity();
    /* cout << "grapvity" << endl;
    print();
    cout << endl; */
    
    return true; 
}
void solve(){
    cin >> n >> m;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            cin >> board[i][j];
        }
    }
    while(simul());
    cout << ans;
    
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

문제 그대로 구현해주면 된다.

pos, block, block_group 구조체를 선언해주었다.

기준 블록을 찾고 제거 대상이 되는 블록 그룹을 찾기 위해서 sort를 이용했다. 연산자 오버로딩을 통해, sort 결과물에서 찾고자 하는 블록과 블록 그룹이 가장 앞으로 오도록 했다.

또한 무지개 블록은 어느 블록 그룹에도 들어갈 수 있으므로, 한번 블록 그룹을 찾은 이후에는 visited 값을 초기화해주었다. 그 외의 색깔 블록은 다시 방문할 필요가 없으므로 visited에 표시해주었다.
