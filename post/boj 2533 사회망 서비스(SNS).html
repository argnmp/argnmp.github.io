코드

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
int n;
vector<vector<int>> graph(1000001, vector<int>());

/*
 * type==1 : must select current node
 * type==-1 : compare whether select or not current node
 */

/*
 * dp[node][0] : lowest value of subtree not selecting the node
 * dp[node][1] : lowest value of subtree selecting the node
 */

int dp[1000001][2] = {0,};
int calc(int node, int parent, int type){
    if(dp[node][0]==-1){
        int count = 0; 
        for(int next: graph[node]){
            if(next == parent)
                continue;
            count += calc(next, node, 1); 
        } 
        dp[node][0] = count;
    }
    if(dp[node][1]==-1){
        int count = 0;
        for(int next: graph[node]){
            if(next == parent)
                continue;
            count += calc(next, node, -1);
        }
        dp[node][1] = count + 1;
    }
    if(type==1){
        return dp[node][1];
    } else {
        return min(dp[node][0], dp[node][1]);
    }
}
void solve(){
    memset(dp, -1, sizeof(dp));
    cin >> n;
    int u, v;
    for(int i = 0; i<n-1; i++){
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u);
    }
    cout << calc(1, 1, -1);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

boj 2213 트리의 독립집합 과 같은 풀이이다. 여기에서는 얼리어답터이거나 아닌 경우의 수 두가지로 나누어서 생각해보면 된다.

dfs를 돌 때에는, boj 2213 트리의 독립집합 과는 반대이다. 여기에서는 무조건 현재 노드를 골라야 하거나, 고를지 말지 선택하는, 이 두가지 경우를 고려하여 탐색을 해야 한다.

느낀점

boj 2213 트리의 독립집합 을 풀고 이 문제를 풀어서 그런가 빠르게 풀 수 있었다.

트리에서는 n개의 vertex가 있을 때, 무조건 n-1개의 edge가 존재하여야 한다. 트리에서 루트 노드를 제외한 모든 노드는 parent가 있다. 따라서 어떤 vertex를 루트 노드로 선택을 하더라도, 그 vertex를 제외한 나머지 노드는 무조건 parent가 한 개씩 있으므로 n-1개가 된다.k
