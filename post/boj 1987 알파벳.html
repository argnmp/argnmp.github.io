문제

풀이

첫번째 풀이: 비트마스킹

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int r, c;
char board[20][20];
int dx[4] = {0,0,-1,1};
int dy[4] = {-1,1,0,0};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){ return 0<=x&&x<r&&0<=y&&y<c; }
};
struct box{
    pos p;
    int bit;
    int dist;
};
void solve(){
    cin >> r >> c;
    for(int i=0; i<r; i++){
        for(int j=0; j<c; j++){
            char ch = cin.get();
            while(ch=='\n'||ch=='\r'||ch==' '){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    }
    int ans = INT_MIN;
    vector<vector<unordered_set<int>>> visited(20, vector(20, unordered_set<int>()));
    queue<box> q;
    q.push({{0,0}, (1<<(board[0][0]-'A')), 1});
    visited[0][0].insert(1<<(board[0][0]-'A'));
    while(!q.empty()){
        box cur = q.front();
        q.pop();
        ans = max(ans, cur.dist); 
        for(int i = 0; i<4; i++){
            pos next = cur.p.adj(i);
            if(next.safe() && ((cur.bit & (1<<(board[next.x][next.y]-'A')))==0) && visited[next.x][next.y].find(cur.bit | 1<<(board[next.x][next.y]-'A')) == visited[next.x][next.y].end()){
                visited[next.x][next.y].insert(cur.bit | (1<<(board[next.x][next.y]-'A')));
                q.push({next, cur.bit | (1<<(board[next.x][next.y]-'A')), cur.dist + 1});
            }
        }
    }
    cout << ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

bfs로 탐색을 하면서 지금까지의 거쳐간 알파벳을 비트마스킹으로 표시한다. 그리고 visited배열에는 unordered_set을 이용해서 알파벳 사용 여부를 나타내는 비트를 저장하여, bfs로 다시 탐색하지 않도록 한다.

처음에는 unordered_set을 사용하지 않고 무조건 다 bfs 탐색을 했는데 메모리 초과가 발생했다. queue에 너무 많은 요소들이 들어가기 때문이었다. visited 배열을 만들어 주어야 했는데, visited[20][20][1<<26]은 매우 큰 수라서 만들 수 가 없었다. 그래서 unordered_set을 사용해서 저장했더니 문제가 풀렸다.

하지만 이 방식은 문제에서 요구하는 방식이 아닐 뿐더러, 정확성도 떨어지는 풀이라고 생각된다. 그래서 검색해보니 dfs를 이용한 풀이가 있었다.

두번째 풀이: dfs

#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int r, c;
char board[20][20];
int dx[4] = {0,0,-1,1};
int dy[4] = {-1,1,0,0};
struct pos{
    int x, y;
    pos adj(int i){ return {x+dx[i%4], y+dy[i%4]}; }
    bool safe(){ return 0<=x&&x<r&&0<=y&&y<c; }
};
struct box{
    pos p;
    int bit;
    int dist;
};
int visited[26] = {0,};
int ans = INT_MIN;
void dfs(pos cur, int dist){
    visited[board[cur.x][cur.y]-'A'] = 1;
    for(int i = 0; i<4; i++){
        pos next = cur.adj(i);
        if(next.safe() && visited[board[next.x][next.y]-'A']==0){
            visited[board[next.x][next.y]-'A'] = 1;
            dfs(next, dist+1);
            visited[board[next.x][next.y]-'A'] = 0;
        }
    }
    ans = max(ans, dist);
}
void solve(){
    cin >> r >> c;
    for(int i=0; i<r; i++){
        for(int j=0; j<c; j++){
            char ch = cin.get();
            while(ch=='\n'||ch=='\r'||ch==' '){
                ch = cin.get();
            }
            board[i][j] = ch;
        }
    }
    dfs({0,0}, 1);
    cout << ans; 
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

[0,0]부터 dfs로 탐색을 시작하여, 해당 알파벳을 밟지 않은 경우 다음 좌표를 방문한다.

느낀점

무조건 비트마스킹으로 생각하지 말고, 다른 방식으로도 생각해보자.
