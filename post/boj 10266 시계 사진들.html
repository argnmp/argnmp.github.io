#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>
#include <map>
#include <unordered_map>
#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
#include <cmath>
using namespace std;
using ll = long long;
int n;
int degrees[360000] = {0,};
vector<int> kmp_pi(string p){
    int j = 0, m = p.size();
    vector<int> pi(m, 0);
    for(int i = 1; i<m; i++){
        while(j>0 && p[i]!=p[j])
            j = pi[j-1];
        if(p[i]==p[j])
            pi[i] = ++j;        
    }
    return pi;
}
vector<int> kmp(string s, string p){
    int j = 0, n = s.size(), m = p.size();
    vector<int> pi = kmp_pi(p);
    vector<int> res;
    for(int i = 0; i<n; i++){
        while(j>0 && s[i]!=p[j])
            j = pi[j-1];
        if(s[i]==p[j]){
            if(j==m-1){
                res.push_back(i-m-1);
                j = pi[j];
            }
            else {
                ++j;
            }
        }
    }
    return res;
}
void solve(){
    string s, p;
    cin >> n;
    int t;
    for(int i = 0; i<n; i++){
        cin >> t; 
        degrees[t] = 1;
    }
    for(int i = 0; i<360000; i++){
        if(degrees[i]==1)
            s += '1';
        else
            s += '0';
    }
    s += s;

    memset(degrees, 0, sizeof(degrees));
    for(int i = 0; i<n; i++){
        cin >> t;
        degrees[t] = 1;
    }
    for(int i = 0; i<360000; i++){
        if(degrees[i]==1)
            p += '1';
        else
            p += '0';
    }

    vector<int> search_result = kmp(s, p);
    if(search_result.size() > 0){
        cout << "possible";
    }
    else {
        cout << "impossible";
    }

}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

kmp 알고리즘을 사용하는 문제이다.

string s, p; 의 두개의 360000 길이의 string을 준비한다.

첫번째 시계의 각도가 주어지면, 해당 각도를 index로 하여 s에 1로 기록해준다. 그 외의 string 의 값들은 0으로 한다. 그리고 s를 한번 반복하여 뒤에 붙여준다.

두번째 시계의 각도가 주어지면, 해당 각도를 index로 하여 p에 1로 기록해준다. 그 외의 string 의 값들은 0으로 한다.

그리고 s에서 p패턴이 존재하는 지를 kmp 알고리즘을 사용하여 찾으면 된다. 첫번째 시계의 문자열을 한번 반복하여 붙여준 이유는, 시계에서 회전하여 일치하는 경우를 찾기 위함이다. 이는 360도 이전 까지만 돌려보면 충분하므로 한번만 붙여주는 것이다.
