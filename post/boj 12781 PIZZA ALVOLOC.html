#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
#include <queue>

#include <set>
#include <unordered_set>
#include <bitset>
#include <cstring>
using namespace std;
using ll = long long;
struct coord{
    double x, y;
};
/*
 * u = (p2.x-p1.x, p2.y-p1.y, 0);
 * v = (p3.x-p1.x, p3.y-p1.y, 0);
 * u x v = (0, 0, (p2.x-p1.x)*(p3.y-p1.y) - (p2.y-p1.y)*(p3.x-p1.x))
 */
double ccw(coord p1, coord p2, coord p3){
    return (p2.x-p1.x)*(p3.y-p1.y) - (p2.y-p1.y)*(p3.x-p1.x);
}

void solve(){
    coord p1, p2, p3, p4;
    cin >> p1.x >> p1.y >> p2.x >> p2.y >> p3.x >> p3.y >> p4.x >> p4.y; 
    double check1 = ccw(p1, p2, p3) * ccw(p1, p2, p4);
    double check2 = ccw(p3, p4, p1) * ccw(p3, p4, p2);
    if(check1 < 0 && check2 < 0)
        cout << 1;
    else
        cout << 0;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    solve(); 
    return 0;
}

풀이

boj 11758 CCW 벡터의 외적을 이용하는 문제이다.

문제의 제약 조건으로, 모든 점이 겹치지 않고 볼록 다각형 위에 있다고 주어졌다. 네 조각으로 나누어지기 위해서는, 자르게 되는 한 선분을 중심으로 다른 두 점이 각각 왼쪽가 오른쪽에 위치해야 한다. 둘다 왼쪽 or 오른쪽에 위치하게 되면 네 조각으로 자를 수 없다.

이를 판단하기 위해서는 벡터의 외적 결과가 하나는 시계방향 다른 하나는 반시계 방향이 나와야 하고, 이는 두 외적 결과값이 나왔을 때 음수 인 것으로 판단하면 된다.

두 선분 모두에 대해서 계산을 해주어야 한다.
