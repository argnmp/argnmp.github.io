개요

passport.js는 express.js에서 자주 사용되는 사용자 인증 라이브러리이다. 그 중에서 passport.js를 이용한 인증을 적용시켜 보고자 한다.

passport.js를 이용한 로그인 까지는 완성이 되었다고 가정한다.

로그인 시 jwt token 발급하기

router.post('/signin', wrapAsync(authController.signin));

const signin = async (req, res, next) => {
    try {
        passport.authenticate('local', (err, user, info) => {
            if(err){
                next({status: 500, message: 'internal server error'});
                return;
            }
            if(!user){
                next({status: 400, message: info.message});
                return;
            }
            const token = jwt.sign({ id: user.userId }, process.env.JWT_KEY, { expiresIn: '1m' });
            res.cookie('jwt', token);
            res.status(200).json({ message: "your token was generated" });
        })(req,res);        
    } catch (e) {
        console.error(e);
        next({status: 500, message: 'internal server error'});
    }
}

/signin으로 post 요청이 들어오면 signin 함수를 실행한다. 아이디와 패스워드가 일치했을 시 jwt.sign 으로 토큰을 발급한다. 첫번째 인자는 payload로, 나중에 jwt을 이용해 요청이 들어올 때 검증하기 위한 정보를 추가해 놓는다. 두번째 인자는 secret key 이다. 아무거나 지정해도 되지만 나중에 jwt를 검증할 때 사용해야 하므로 같이 보관하는 편이 좋다. 이 프로젝트에서 dotenv를 사용중이기 때문에 .env 파일에서 불러오는 방식을 사용했다. 세번째는 옵션이다. 대표적으로 expiresIn 옵션은 token의 제한시간을 설정한다.

res.cookie('jwt', token) 으로 cookie에 token을 넣어서 클라이언트에게 응답한다.

jwt strategy 생성하기

//jwt strategy
const cookieExtractor = (req)=>{
    let token = null;
    if (req&&req.cookies) token = req.cookies['jwt'];
    return token;
};
const jwtConfig = {
    jwtFromRequest: cookieExtractor, 
    secretOrKey: process.env.JWT_KEY
};
const jwtVerify = async (payload, done) => {
    try {
        const user = await mockUserCheck(payload.id); 
        if(!user){
            done(null, false, {message: "not valid authentication"});
            return;
        }
        done(null, user);
    } catch (e) {
        console.error(e);
        done(e);
    }
}

module.exports = {
    initialize : () => {
        passport.use('local', new LocalStrategy(localConfig, localVerify));
        passport.use('jwt', new JWTStrategy(jwtConfig, jwtVerify));
    },
    authenticate : (req, res, next) => {
        return passport.authenticate('jwt', {session: false}, (err, user, info) => {
            if(err){
                next({status: 500, message: 'internal server error'});
                return;
            }
            if(!user){
                next({status: 401, message: info.message});
                return;
            }
            req.user = user;
            next();
        })(req,res,next);
    }
}

구조가 약간 복잡해 보일 수 있지만 핵심은 똑같다. 일단 strategy를 생성하는 이유는 요청을 받을 때 이 strategy로 쉽게 검증을 하기 위해서 이다. passport 에 strategy를 등록하는 코드는 바로 passport.use('jwt', new JWTStrategy(jwtConfig, jwtVerify)); 이 부분이다. 나중에 passport.authenticate('jwt'.... 를 통해서 해당 strategy를 사용할 수 있게 된다.

passport.use('jwt', new JWTStrategy(jwtConfig, jwtVerify)) 를 통해서 jwt strategy를 등록할 수 있다. jwtConfig에는 두가지 정보가 무조건 들어가야 한다.

첫번째는 클라이언트로 부터 받은 jwt token 이다. 앞서서 클라이언트에게 토큰을 발급할 때 쿠키에 저장하도록 했었다. 따라서 다시 클라이언트의 쿠키로 부터 가져와야 한다. 그 코드가 바로 cookieExtractor 함수 이다. jwtConfig.jwtFromRequest에 함수를 등록해주면 첫번째 인자로 req 객체를 넣어준다. 그걸 통해서 token을 리턴하면 된다.

두번째는 jwt를 발급할 때 사용했던 secretKey이다. 나는 dotenv를 이용해서 secretkey를 가져왔었다. 똑같이 secretkey를 입력해주면 된다.

다음으로 jwtVerify는 jwt로 인증이 완료되면 실행하는 콜백 함수이다. passport.js의 다른 statrategy를 등록할 때와 같다고 생각하면 된다. jwt로 부터 payload를 뽑아낼 수 있는데, 그 payload를 통해서 실제 유저인지를 확인한 뒤 done을 통해서 필요한 user의 정보를 넘겨주면 된다.

보통은 라우트에서 바로 router.get('/jwttest',passport.authenticate('jwt', {session: false},wrapAsync(authController.jwttest)); 를 호출하는데, 조금 커스텀을 해주기 위해서 그 부분을 분리해 놓았다. (두번째 코드) done()를 호출하면 err, user, info 순서대로 done의 인자들이 들어가게 된다. 여기서 user 정보를 req.user에 넣어서 다음 미들웨어를 호출하였다.

로그아웃

앞선 코드에서 cookie를 통해서 token을 제공했으므로 이 cookie를 비워주면 된다. 하지만 로그아웃 전에 토큰을 따로 보관한다면 로그아웃을 해도 다시 그 토큰을 이용할 수 있기 때문에 이 경우는 접근을 제한 할 수는 없다.

다른 방법은 서버에서 로그아웃 한 토큰을 남은 시간 후에는 자동으로 삭제되도록 디비에 저장해 놓고, 그 토큰으로의 접근을 방지하는 것이다.
